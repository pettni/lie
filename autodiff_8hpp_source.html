52 template<typename _F, typename _Wrt>
53 auto dr_autodiff(_F && f, _Wrt && x)
54 {
55  using Result = decltype(std::apply(f, x));
56  using Scalar = ::smooth::Scalar<Result>;
57 
58  static_assert(Manifold<Result>, "f(x) is not a Manifold");
59 
60  using AdScalar = autodiff::Dual<Scalar, Scalar>;
61 
62  Result fval = std::apply(f, x);
63 
64  static constexpr Eigen::Index Nx = wrt_dof<_Wrt>::value;
65  static constexpr Eigen::Index Ny = Dof<Result>;
66  const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
67 
68  // cast fval and x to ad types
69  const auto x_ad = wrt_cast<AdScalar>(x);
70  const CastT<AdScalar, Result> fval_ad = cast<AdScalar>(fval);
71 
72  // zero-valued tangent element
73  Eigen::Matrix<AdScalar, Nx, 1> a_ad = Eigen::Matrix<AdScalar, Nx, 1>::Zero(nx);
74 
75  Eigen::Matrix<Scalar, Ny, Nx> jac = autodiff::jacobian(
76  [&f, &fval_ad, &x_ad](Eigen::Matrix<AdScalar, Nx, 1> & var) -> Eigen::Matrix<AdScalar, Ny, 1> {
77  return rminus<CastT<AdScalar, Result>>(std::apply(f, wrt_rplus(x_ad, var)), fval_ad);
78  },
79  autodiff::wrt(a_ad),
80  autodiff::at(a_ad));
81 
82  return std::make_pair(std::move(fval), std::move(jac));
83 }
