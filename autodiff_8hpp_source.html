66 template<std::size_t K = 1>
67  requires(K >= 1 && K <= 2)
68 auto dr_autodiff(auto && f, auto && x)
69 {
70  using Result = decltype(std::apply(f, x));
71  using Scalar = ::smooth::Scalar<Result>;
72  using Eigen::Matrix;
73 
74  static_assert(Manifold<Result>, "f(x) is not a Manifold");
75 
76  using AdScalar = autodiff::HigherOrderDual<K, Scalar>;
77 
78  Result F = std::apply(f, x);
79 
80  static constexpr Eigen::Index Nx = wrt_Dof<decltype(x)>();
81  static constexpr Eigen::Index Ny = Dof<Result>;
82  const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
83  const Eigen::Index ny = dof(F);
84 
85  // cast F and x to ad types
86  const auto x_ad = wrt_cast<AdScalar>(x);
87  CastT<AdScalar, Result> F_ad = cast<AdScalar>(F);
88 
89  if constexpr (K == 1) {
90  // zero-valued tangent element
91  Matrix<AdScalar, Nx, 1> a_ad = Matrix<AdScalar, Nx, 1>::Zero(nx);
92 
93  // function to differentiate
94  const auto f_ad = [&](Matrix<AdScalar, Nx, 1> & var) -> Matrix<AdScalar, Ny, 1> {
95  return rminus<CastT<AdScalar, Result>>(std::apply(f, wrt_rplus(x_ad, var)), F_ad);
96  };
97 
98  Matrix<Scalar, Ny, Nx> J(ny, nx);
99 
100  J = autodiff::jacobian(f_ad, autodiff::wrt(a_ad), autodiff::at(a_ad));
101  return std::make_pair(std::move(F), std::move(J));
102  } else if constexpr (K == 2) {
103  // function to differentiate
104  const auto f_ad = [&f, &x_ad, &F_ad](
105  Matrix<AdScalar, Nx, 1> & var1,
106  Matrix<AdScalar, Nx, 1> & var2) -> Matrix<AdScalar, Ny, 1> {
107  return rminus(std::apply(f, wrt_rplus(wrt_rplus(x_ad, var1), var2)), F_ad);
108  };
109 
110  Matrix<Scalar, Ny, Nx> J(ny, nx);
111  Matrix<Scalar, Nx, std::min(Nx, Ny) == -1 ? -1 : Nx * Ny> H(nx, nx * ny);
112 
113  // zero-valued tangent elements
114  Matrix<AdScalar, Nx, 1> a_ad1 = Matrix<AdScalar, Nx, 1>::Zero(nx);
115  Matrix<AdScalar, Nx, 1> a_ad2 = Matrix<AdScalar, Nx, 1>::Zero(nx);
116 
117  const auto a_wrt1 = autodiff::wrt(a_ad1);
118  const auto a_wrt2 = autodiff::wrt(a_ad2);
119 
120  autodiff::detail::ForEachWrtVar(
121  a_wrt1, [&](auto && i, auto && xi) constexpr {
122  autodiff::detail::ForEachWrtVar(
123  a_wrt2, [&](auto && j, auto && xj) constexpr {
124  const auto u =
125  autodiff::detail::eval(f_ad, autodiff::at(a_ad1, a_ad2), autodiff::wrt(xi, xj));
126  for (auto k = 0u; k < ny; ++k) {
127  J(k, i) = static_cast<double>(autodiff::detail::derivative<1>(u[k]));
128  H(j, k * nx + i) = autodiff::detail::derivative<2>(u[k]);
129  }
130  });
131  });
132 
133  return std::make_tuple(std::move(F), std::move(J), std::move(H));
134  }
135 }
