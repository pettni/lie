50template<std::size_t K, typename Scalar>
51constexpr StaticMatrix<Scalar, 1, K + 1> monomial_derivative(Scalar u, std::size_t p = 0)
52{
53 StaticMatrix<Scalar, 1, K + 1> ret;
54
55 if (p > K) { return ret; }
56
57 for (auto i = 0u; i < p; ++i) { ret[0][i] = Scalar(0); }
58 Scalar P1 = 1;
59 std::size_t P2 = 1;
60 for (auto j = 2u; j <= p; ++j) { P2 *= j; }
61 ret[0][p] = P1 * P2;
62 for (auto i = p + 1; i <= K; ++i) {
63 P1 *= u;
64 P2 *= i;
65 P2 /= i - p;
66 ret[0][i] = P1 * P2;
67 }
68
69 return ret;
70}

98template<std::size_t K, typename Scalar = double>
99constexpr StaticMatrix<Scalar, K + 1, K + 1> bspline_basis()
100{
101 StaticMatrix<Scalar, K + 1, K + 1> ret;
102 if constexpr (K == 0) {
103 ret[0][0] = 1;
104 return ret;
105 } else {
106 const auto coeff_mat_km1 = bspline_basis<K - 1, Scalar>();
107 StaticMatrix<Scalar, K + 1, K> low, high;
108 StaticMatrix<Scalar, K, K + 1> left, right;
109
110 for (std::size_t i = 0; i != K; ++i) {
111 for (std::size_t j = 0; j != K; ++j) {
112 low[i][j] = coeff_mat_km1[i][j];
113 high[i + 1][j] = coeff_mat_km1[i][j];
114 }
115 }
116
117 for (std::size_t k = 0; k != K; ++k) {
118 left[k][k + 1] = static_cast<Scalar>(K - (k + 1)) / static_cast<Scalar>(K);
119 left[k][k] = Scalar(1) - left[k][k + 1];
120
121 right[k][k + 1] = Scalar(1) / static_cast<Scalar>(K);
122 right[k][k] = -right[k][k + 1];
123 }
124
125 return low * left + high * right;
126 }
127}

133template<std::size_t K, typename Scalar = double>
134constexpr StaticMatrix<Scalar, K + 1, K + 1> bernstein_basis()
135{
136 StaticMatrix<Scalar, K + 1, K + 1> ret;
137 if constexpr (K == 0) {
138 ret[0][0] = 1;
139 return ret;
140 } else {
141 constexpr auto coeff_mat_km1 = bernstein_basis<K - 1, Scalar>();
142 StaticMatrix<Scalar, K + 1, K> low, high;
143 StaticMatrix<Scalar, K, K + 1> left, right;
144
145 for (std::size_t i = 0; i != K; ++i) {
146 for (std::size_t j = 0; j != K; ++j) {
147 low[i][j] = coeff_mat_km1[i][j];
148 high[i + 1][j] = coeff_mat_km1[i][j];
149 }
150 }
151
152 for (std::size_t k = 0; k != K; ++k) {
153 left[k][k] = Scalar(1);
154
155 right[k][k] = Scalar(-1);
156 right[k][k + 1] = Scalar(1);
157 }
158
159 return low * left + high * right;
160 }
161}

167template<std::size_t K, typename Scalar = double>
168constexpr StaticMatrix<Scalar, K + 1, K + 1> hermite_basis()
169{
170 StaticMatrix<Scalar, K + 1, K + 1> ret;
171 ret[0][0] = 1;
172
173 if constexpr (K > 0) { ret[1][1] = 2; }
174
175 if constexpr (K > 1) {
176 for (auto k = 2u; k < K + 1; ++k) {
177 for (auto i = 0u; i < k; ++i) { ret[i + 1][k] += 2 * ret[i][k - 1]; }
178 for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= 2 * (k - 1) * ret[i][k - 2]; }
179 }
180 }
181 return ret;
182}

188template<std::size_t K, typename Scalar = double>
189constexpr StaticMatrix<Scalar, K + 1, K + 1> laguerre_basis()
190{
191 StaticMatrix<Scalar, K + 1, K + 1> ret;
192 ret[0][0] = 1;
193
194 if constexpr (K > 0) {
195 ret[0][1] = 1.;
196 ret[1][1] = -1;
197 }
198
199 if constexpr (K > 1) {
200 for (auto k = 2u; k < K + 1; ++k) {
201 for (auto i = 0u; i < k; ++i) {
202 ret[i][k] += (2 * k - 1) * ret[i][k - 1] / k;
203 ret[i + 1][k] -= ret[i][k - 1] / k;
204 }
205 for (auto i = 0u; i + 1 < k; ++i) {
206 ret[i][k] -= static_cast<double>(k - 1) * ret[i][k - 2] / k;
207 }
208 }
209 }
210 return ret;
211}

224template<std::size_t K, typename Scalar = double>
225constexpr StaticMatrix<Scalar, K + 1, K + 1> jacobi_basis(double alpha, double beta)
226{
227 StaticMatrix<Scalar, K + 1, K + 1> ret;
228 ret[0][0] = 1;
229
230 if constexpr (K > 0) {
231 ret[0][1] = alpha + 1. - (alpha + beta + 2) / 2;
232 ret[1][1] = (alpha + beta + 2.) / 2;
233 }
234
235 if constexpr (K > 1) {
236 for (auto k = 2u; k < K + 1; ++k) {
237 const auto frac = 1. / ((2 * k) * (k + alpha + beta) * (2 * k + alpha + beta - 2));
238 const auto c1 = (2 * k + alpha + beta - 1) * (alpha * alpha - beta * beta);
239 const auto c2 =
240 (2 * k + alpha + beta - 1) * (2 * k + alpha + beta) * (2 * k + alpha + beta - 2);
241 const auto c3 = 2 * (k + alpha - 1) * (k + beta - 1) * (2 * k + alpha + beta);
242 for (auto i = 0u; i < k; ++i) {
243 ret[i][k] += c1 * ret[i][k - 1] * frac;
244 ret[i + 1][k] += c2 * ret[i][k - 1] * frac;
245 }
246 for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= c3 * ret[i][k - 2] * frac; }
247 }
248 }
249
250 return ret;
251}

290template<PolynomialBasis Basis, std::size_t K, typename Scalar = double>
291constexpr StaticMatrix<Scalar, K + 1, K + 1> polynomial_basis()
292{
293 if constexpr (Basis == PolynomialBasis::Monomial) {
294 StaticMatrix<Scalar, K + 1, K + 1> ret;
295 for (auto k = 0u; k <= K; ++k) { ret[k][k] = Scalar(1); }
296 return ret;
297 }
298 if constexpr (Basis == PolynomialBasis::Bernstein) {
299 return detail::bernstein_basis<K, Scalar>();
300 }
301 if constexpr (Basis == PolynomialBasis::Laguerre) { return detail::laguerre_basis<K, Scalar>(); }
302 if constexpr (Basis == PolynomialBasis::Hermite) { return detail::hermite_basis<K, Scalar>(); }
303 if constexpr (Basis == PolynomialBasis::Legendre) {
304 return detail::jacobi_basis<K, Scalar>(0, 0);
305 }
306 if constexpr (Basis == PolynomialBasis::Chebyshev1st) {
307 auto ret = detail::jacobi_basis<K, Scalar>(-0.5, -0.5);
308 const auto fac = monomial_derivative<K>(1.) * ret;
309 for (auto k = 0u; k < K + 1; ++k) {
310 for (auto r = 0u; r < K + 1; ++r) { ret[r][k] /= fac[0][k]; }
311 }
312 return ret;
313 }
314 if constexpr (Basis == PolynomialBasis::Chebyshev2nd) {
315 auto ret = detail::jacobi_basis<K, Scalar>(0.5, 0.5);
316 const auto fac = monomial_derivative<K>(1.) * ret;
317 for (auto k = 0u; k < K + 1; ++k) {
318 for (auto r = 0u; r < K + 1; ++r) { ret[r][k] *= static_cast<Scalar>(k + 1) / fac[0][k]; }
319 }
320 return ret;
321 }
322 if constexpr (Basis == PolynomialBasis::Bspline) { return detail::bspline_basis<K, Scalar>(); }
323}

343template<
344 std::size_t K,
345 std::ranges::random_access_range R,
346 typename Scalar = std::ranges::range_value_t<R>>
347constexpr StaticMatrix<Scalar, K + 1, K + 1> lagrange_basis(const R & ts)
348{
349 StaticMatrix<Scalar, K + 1, K + 1> ret;
350
351 for (auto row = 0u; row < K + 1; ++row) {
352 ret[row][0] = 1;
353 for (auto col = 0u; col < K + 1; ++col) {
354 if (col != row) {
355 const auto row_copy = ret[row];
356 ret[row].fill(0);
357 for (auto i = 0u; i <= col - (col > row ? 1 : 0); ++i) {
358 ret[row][i + 1] += row_copy[i] / (*(ts.begin() + row) - *(ts.begin() + col));
359 ret[row][i] -=
360 *(ts.begin() + col) * row_copy[i] / (*(ts.begin() + row) - *(ts.begin() + col));
361 }
362 }
363 }
364 }
365
366 return ret.transpose();
367}

380template<std::size_t K, std::size_t N, typename Scalar, std::ranges::sized_range R>
381constexpr StaticMatrix<Scalar, K + 1, N>
382polynomial_basis_derivatives(const StaticMatrix<Scalar, K + 1, K + 1> & B, const R & ts)
383{
384 assert(std::ranges::size(ts) == N);
385
386 StaticMatrix<Scalar, K + 1, N> ret;
387
388 for (auto j = 0u; const auto t : ts) {
389 const auto U = monomial_derivative<K>(t, 1);
390 const auto U_B = U * B;
391 for (auto i = 0u; i < K + 1; ++i) { ret[i][j] = U_B[0][i]; }
392 ++j;
393 }
394
395 return ret;
396}

416template<PolynomialBasis Basis, std::size_t K, typename Scalar = double>
417constexpr StaticMatrix<Scalar, K + 1, K + 1> polynomial_cumulative_basis()
418{

421 for (std::size_t j = 0; j != K; ++j) { M[i][K - 1 - j] += M[i][K - j]; }

434template<std::size_t K, std::size_t P, typename Scalar = double>
435constexpr StaticMatrix<Scalar, K + 1, K + 1> monomial_integral()
436{
437 StaticMatrix<Scalar, K + 1, K + 1> ret;
438 for (auto i = 0u; i <= K; ++i) {
439 for (auto j = i; j <= K; ++j) {
440 if (i >= P && j >= P) {
441 std::size_t c = 1;
442 for (auto _i = i - P + 1; _i <= i; ++_i) { c *= _i; }
443 for (auto _j = j - P + 1; _j <= j; ++_j) { c *= _j; }
444 ret[i][j] = static_cast<Scalar>(c) / (i + j - 2 * P + 1);
445 } else {
446 ret[i][j] = 0;
447 }
448 ret[j][i] = ret[i][j];
449 }
450 }
451 return ret;
452}

462inline constexpr double
463integrate_absolute_polynomial(double t0, double t1, double A, double B, double C)
464{
465 // location of first zero (if any)
466 double mid1 = std::numeric_limits<double>::infinity();
467 // location of second zero (if any)
468 double mid2 = std::numeric_limits<double>::infinity();
469
470 if (std::abs(A) < 1e-9 && std::abs(B) > 1e-9) {
471 // linear non-constant function
472 mid1 = std::clamp(-C / B, t0, t1);
473 } else if (std::abs(A) > 1e-9) {
474 // quadratic function
475 const double res = B * B / (4 * A * A) - C / A;
476
477 if (res > 0) {
478 mid1 = -B / (2 * A) - std::sqrt(res);
479 mid2 = -B / (2 * A) + std::sqrt(res);
480 }
481 }
482
483 const auto integ = [&](double u) { return A * u * u * u / 3 + B * u * u / 2 + C * u; };
484
485 const double mid1cl = std::clamp(mid1, t0, t1);
486 const double mid2cl = std::clamp(mid2, t0, t1);
487
488 return std::abs(integ(t1) - integ(t0) + 2 * integ(mid1cl) - 2 * integ(mid2cl));
489}
