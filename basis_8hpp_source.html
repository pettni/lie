52 template<std::size_t K, typename Scalar>
53 constexpr StaticMatrix<Scalar, 1, K + 1> monomial_derivative(Scalar u, std::size_t p = 0)
54 {
55  StaticMatrix<Scalar, 1, K + 1> ret;
56 
57  if (p > K) { return ret; }
58 
59  for (auto i = 0u; i < p; ++i) { ret[0][i] = Scalar(0); }
60  Scalar P1 = 1;
61  std::size_t P2 = 1;
62  for (auto j = 2u; j <= p; ++j) { P2 *= j; }
63  ret[0][p] = P1 * P2;
64  for (auto i = p + 1; i <= K; ++i) {
65  P1 *= u;
66  P2 *= i;
67  P2 /= i - p;
68  ret[0][i] = P1 * P2;
69  }
70 
71  return ret;
72 }

132 template<std::size_t K, typename Scalar = double>
133 constexpr StaticMatrix<Scalar, K + 1, K + 1> bspline_basis()
134 {
135  StaticMatrix<Scalar, K + 1, K + 1> ret;
136  if constexpr (K == 0) {
137  ret[0][0] = 1;
138  return ret;
139  } else {
140  constexpr auto coeff_mat_km1 = bspline_basis<K - 1, Scalar>();
141  StaticMatrix<Scalar, K + 1, K> low, high;
142  StaticMatrix<Scalar, K, K + 1> left, right;
143 
144  for (std::size_t i = 0; i != K; ++i) {
145  for (std::size_t j = 0; j != K; ++j) {
146  low[i][j] = coeff_mat_km1[i][j];
147  high[i + 1][j] = coeff_mat_km1[i][j];
148  }
149  }
150 
151  for (std::size_t k = 0; k != K; ++k) {
152  left[k][k + 1] = static_cast<Scalar>(K - (k + 1)) / static_cast<Scalar>(K);
153  left[k][k] = Scalar(1) - left[k][k + 1];
154 
155  right[k][k + 1] = Scalar(1) / static_cast<Scalar>(K);
156  right[k][k] = -right[k][k + 1];
157  }
158 
159  return low * left + high * right;
160  }
161 }

201 template<std::size_t K, typename Scalar = double>
202 constexpr StaticMatrix<Scalar, K + 1, K + 1> hermite_basis()
203 {
204  StaticMatrix<Scalar, K + 1, K + 1> ret;
205  ret[0][0] = 1;
206 
207  if constexpr (K > 0) { ret[1][1] = 2; }
208 
209  if constexpr (K > 1) {
210  for (auto k = 2u; k < K + 1; ++k) {
211  for (auto i = 0u; i < k; ++i) { ret[i + 1][k] += 2 * ret[i][k - 1]; }
212  for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= 2 * (k - 1) * ret[i][k - 2]; }
213  }
214  }
215  return ret;
216 }

223 constexpr StaticMatrix<Scalar, K + 1, K + 1> laguerre_basis()
224 {
225  StaticMatrix<Scalar, K + 1, K + 1> ret;
226  ret[0][0] = 1;

229  ret[0][1] = 1.;
230  ret[1][1] = -1;

258 template<std::size_t K, typename Scalar = double>
259 constexpr StaticMatrix<Scalar, K + 1, K + 1> jacobi_basis(double alpha, double beta)
260 {
261  StaticMatrix<Scalar, K + 1, K + 1> ret;
262  ret[0][0] = 1;
263 
264  if constexpr (K > 0) {
265  ret[0][1] = alpha + 1. - (alpha + beta + 2) / 2;
266  ret[1][1] = (alpha + beta + 2.) / 2;
267  }
268 
269  if constexpr (K > 1) {
270  for (auto k = 2u; k < K + 1; ++k) {
271  const auto frac = 1. / ((2 * k) * (k + alpha + beta) * (2 * k + alpha + beta - 2));
272  const auto c1 = (2 * k + alpha + beta - 1) * (alpha * alpha - beta * beta);
273  const auto c2 =
274  (2 * k + alpha + beta - 1) * (2 * k + alpha + beta) * (2 * k + alpha + beta - 2);
275  const auto c3 = 2 * (k + alpha - 1) * (k + beta - 1) * (2 * k + alpha + beta);
276  for (auto i = 0u; i < k; ++i) {
277  ret[i][k] += c1 * ret[i][k - 1] * frac;
278  ret[i + 1][k] += c2 * ret[i][k - 1] * frac;
279  }
280  for (auto i = 0u; i + 1 < k; ++i) { ret[i][k] -= c3 * ret[i][k - 2] * frac; }
281  }
282  }
283 
284  return ret;
285 }

324 template<PolynomialBasis Basis, std::size_t K, typename Scalar = double>
325 constexpr StaticMatrix<Scalar, K + 1, K + 1> polynomial_basis()
326 {
327  if constexpr (Basis == PolynomialBasis::Monomial) {
328  StaticMatrix<Scalar, K + 1, K + 1> ret;
329  for (auto k = 0u; k <= K; ++k) { ret[k][k] = Scalar(1); }
330  return ret;
331  }
332  if constexpr (Basis == PolynomialBasis::Bernstein) {
333  return detail::bernstein_basis<K, Scalar>();
334  }
335  if constexpr (Basis == PolynomialBasis::Laguerre) { return detail::laguerre_basis<K, Scalar>(); }
336  if constexpr (Basis == PolynomialBasis::Hermite) { return detail::hermite_basis<K, Scalar>(); }
337  if constexpr (Basis == PolynomialBasis::Legendre) {
338  return detail::jacobi_basis<K, Scalar>(0, 0);
339  }
340  if constexpr (Basis == PolynomialBasis::Chebyshev1st) {
341  auto ret = detail::jacobi_basis<K, Scalar>(-0.5, -0.5);
342  const auto fac = monomial_derivative<K>(1.) * ret;
343  for (auto k = 0u; k < K + 1; ++k) {
344  for (auto r = 0u; r < K + 1; ++r) { ret[r][k] /= fac[0][k]; }
345  }
346  return ret;
347  }
348  if constexpr (Basis == PolynomialBasis::Chebyshev2nd) {
349  auto ret = detail::jacobi_basis<K, Scalar>(0.5, 0.5);
350  const auto fac = monomial_derivative<K>(1.) * ret;
351  for (auto k = 0u; k < K + 1; ++k) {
352  for (auto r = 0u; r < K + 1; ++r) { ret[r][k] *= static_cast<Scalar>(k + 1) / fac[0][k]; }
353  }
354  return ret;
355  }
356  if constexpr (Basis == PolynomialBasis::Bspline) { return detail::bspline_basis<K, Scalar>(); }
357 }

377 template<
378  std::size_t K,
379  std::ranges::random_access_range R,
380  typename Scalar = std::ranges::range_value_t<R>>
381 constexpr StaticMatrix<Scalar, K + 1, K + 1> lagrange_basis(const R & ts)
382 {
383  StaticMatrix<Scalar, K + 1, K + 1> ret;
384 
385  for (auto row = 0u; row < K + 1; ++row) {
386  ret[row][0] = 1;
387  for (auto col = 0u; col < K + 1; ++col) {
388  if (col != row) {
389  const auto row_copy = ret[row];
390  ret[row].fill(0);
391  for (auto i = 0u; i <= col - (col > row ? 1 : 0); ++i) {
392  ret[row][i + 1] += row_copy[i] / (*(ts.begin() + row) - *(ts.begin() + col));
393  ret[row][i] -=
394  *(ts.begin() + col) * row_copy[i] / (*(ts.begin() + row) - *(ts.begin() + col));
395  }
396  }
397  }
398  }
399 
400  return ret.transpose();
401 }

468 template<std::size_t K, std::size_t P, typename Scalar = double>
469 constexpr StaticMatrix<Scalar, K + 1, K + 1> monomial_integral()
470 {
471  StaticMatrix<Scalar, K + 1, K + 1> ret;
472  for (auto i = 0u; i <= K; ++i) {
473  for (auto j = i; j <= K; ++j) {
474  if (i >= P && j >= P) {
475  std::size_t c = 1;
476  for (auto _i = i - P + 1; _i <= i; ++_i) { c *= _i; }
477  for (auto _j = j - P + 1; _j <= j; ++_j) { c *= _j; }
478  ret[i][j] = static_cast<Scalar>(c) / (i + j - 2 * P + 1);
479  } else {
480  ret[i][j] = 0;
481  }
482  ret[j][i] = ret[i][j];
483  }
484  }
485  return ret;
486 }

505 template<PolynomialBasis Basis, std::size_t K, typename Scalar, std::ranges::range R>
506 auto evaluate_polynomial(const R & x, const Scalar & u, int p = 0)
507 {
508  assert(std::ranges::size(x) == K + 1);
509 
510  constexpr auto B_s = polynomial_basis<Basis, K, double>();
511  const auto U_s = monomial_derivative<K>(u, p);
512 
513  Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> B(B_s[0].data());
514  Eigen::Map<const Eigen::Matrix<Scalar, K + 1, 1>> U(U_s[0].data());
515 
516  using RT = std::ranges::range_value_t<R>;
517  using RetT = std::conditional_t<
518  std::is_base_of_v<Eigen::MatrixBase<RT>, RT>,
519  Eigen::Matrix<Scalar, Dof<RT>, 1>,
520  Scalar>;
521 
522  RetT ret = Default<RetT>();
523 
524  const Eigen::Matrix<Scalar, 1, K + 1> w = U.transpose() * B.template cast<Scalar>();
525  for (auto i = 0u; const auto & xi : x) { ret += w[i++] * xi; }
526 
527  return ret;
528 }

538 inline double integrate_absolute_polynomial(double t0, double t1, double A, double B, double C)
539 {
540  // location of first zero (if any)
541  double mid1 = std::numeric_limits<double>::infinity();
542  // location of second zero (if any)
543  double mid2 = std::numeric_limits<double>::infinity();
544 
545  if (std::abs(A) < 1e-9 && std::abs(B) > 1e-9) {
546  // linear non-constant function
547  mid1 = std::clamp(-C / B, t0, t1);
548  } else if (std::abs(A) > 1e-9) {
549  // quadratic function
550  const double res = B * B / (4 * A * A) - C / A;
551 
552  if (res > 0) {
553  mid1 = -B / (2 * A) - std::sqrt(res);
554  mid2 = -B / (2 * A) + std::sqrt(res);
555  }
556  }
557 
558  const auto integ = [&](double u) { return A * u * u * u / 3 + B * u * u / 2 + C * u; };
559 
560  const double mid1cl = std::clamp(mid1, t0, t1);
561  const double mid2cl = std::clamp(mid2, t0, t1);
562 
563  return std::abs(integ(t1) - integ(t0) + 2 * integ(mid1cl) - 2 * integ(mid2cl));
564 }
