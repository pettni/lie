143 template<std::size_t N, LieGroup G>
144 class PiecewiseBezier
145 {
146 public:
150  PiecewiseBezier() : knots_{0, 1}, segments_{Bezier<N, G>{}} {}
151 
158  PiecewiseBezier(std::vector<double> && knots, std::vector<Bezier<N, G>> && segments)
159  : knots_(std::move(knots)), segments_(std::move(segments))
160  {}
161 
168  template<std::ranges::range Rt, std::ranges::range Rs>
169  PiecewiseBezier(const Rt & knots, const Rs & segments)
170  : knots_(std::ranges::begin(knots), std::ranges::end(knots)),
171  segments_(std::ranges::begin(segments), std::ranges::end(segments))
172  {}
173 
175  PiecewiseBezier(const PiecewiseBezier &) = default;
177  PiecewiseBezier(PiecewiseBezier &&) = default;
179  PiecewiseBezier & operator=(const PiecewiseBezier &) = default;
181  PiecewiseBezier & operator=(PiecewiseBezier &&) = default;
183  ~PiecewiseBezier() = default;
184 
186  double t_min() const { return knots_.front(); }
187 
189  double t_max() const { return knots_.back(); }
190 
199  G operator()(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
200  {
201  // target condition
202  // knots_[istar] <= t < knots_[istar + 1]
203 
204  std::size_t istar = 0;
205 
206  auto it = utils::binary_interval_search(knots_, t);
207  if (it != std::ranges::end(knots_)) {
208  istar = std::min<std::size_t>(it - knots_.begin(), knots_.size() - 2);
209  }
210 
211  const double T = knots_[istar + 1] - knots_[istar];
212  const double u = (t - knots_[istar]) / T;
213  const G g = segments_[istar](u, vel, acc);
214 
215  if (vel.has_value()) { vel.value() /= T; }
216  if (acc.has_value()) { acc.value() /= (T * T); }
217 
218  return g;
219  }
220 
221 private:
222  std::vector<double> knots_;
223  std::vector<Bezier<N, G>> segments_;
224 
225  friend class Curve<G>;
226 };

306 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
307 PiecewiseBezier<2, G> fit_quadratic_bezier(const Rt & tt, const Rg & gg)
308 {
309  assert(std::ranges::size(tt) >= 2 && std::ranges::size(gg) >= 2);
310  assert(std::ranges::adjacent_find(tt, std::ranges::greater_equal()) == tt.end());
311 
312  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
313 
314  std::vector<Bezier<2, G>> segments(N);
315 
316  auto it_g = std::ranges::begin(gg);
317  auto it_t = std::ranges::begin(tt);
318 
319  Tangent<G> v0 = rminus(*(it_g + 1), *it_g) / (*(it_t + 1) - *it_t);
320 
321  for (auto i = 0u; i != N; ++i, ++it_t, ++it_g) {
322  const double dt = *(it_t + 1) - *it_t;
323 
324  // scaled velocity
325  const Tangent<G> va = v0 * dt;
326 
327  // create segment
328  const Tangent<G> v1 = va / 2;
329  const Tangent<G> v2 = rminus(*(it_g + 1), composition(*it_g, ::smooth::exp<G>(va / 2)));
330 
331  segments[i] = Bezier<2, G>(*it_g, std::array<Tangent<G>, 2>{v1, v2});
332 
333  // unscaled end velocity for interval
334  v0 = v2 * 2 / dt;
335  }
336 
337  auto take_view = tt | std::views::take(N + 1);
338  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
339 
340  return PiecewiseBezier<2, G>(std::move(knots), std::move(segments));
341 }

354 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
355 PiecewiseBezier<3, G> fit_cubic_bezier(const Rt & tt,
356  const Rg & gg,
357  std::optional<Tangent<G>> v0 = {},
358  std::optional<Tangent<G>> v1 = {})
359 {
360  assert(std::ranges::size(tt) >= 2 && std::ranges::size(gg) >= 2);
361  assert(std::ranges::adjacent_find(tt, std::ranges::greater_equal()) == tt.end());
362 
363  // number of intervals
364  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
365 
366  std::size_t NumVars = Dof<G> * 3 * N;
367 
368  Eigen::SparseMatrix<Scalar<G>> lhs;
369  lhs.resize(NumVars, NumVars);
370  Eigen::Matrix<int, -1, 1> nnz = Eigen::Matrix<int, -1, 1>::Constant(NumVars, 3);
371  nnz.head(Dof<G>).setConstant(2);
372  nnz.tail(Dof<G>).setConstant(2);
373  lhs.reserve(nnz);
374 
375  Eigen::Matrix<Scalar<G>, -1, 1> rhs(NumVars);
376  rhs.setZero();
377 
378  // variable layout:
379  //
380  // [ v_{1, 0}; v_{2, 0}; v_{3, 0}; v_{1, 1}; v_{2, 1}; v_{3, 1}; ...]
381  //
382  // where v_ji is a Dof-length vector
383 
384  const auto idx = [&](int j, int i) { return 3 * Dof<G> * i + Dof<G> * (j - 1); };
385 
386  std::size_t row_counter = 0;
387 
389 
390  const std::size_t v10_start = idx(1, 0);
391  const std::size_t v20_start = idx(2, 0);
392 
393  if (v0.has_value()) {
394  for (auto n = 0u; n != Dof<G>; ++n) { lhs.insert(row_counter + n, v10_start + n) = 3; }
395  rhs.segment(row_counter, Dof<G>) = v0.value();
396  } else {
397  // zero second derivative at start:
398  // v_{1, 0} = v_{2, 0}
399  for (auto n = 0u; n != Dof<G>; ++n) {
400  lhs.insert(row_counter + n, v10_start + n) = 1;
401  lhs.insert(row_counter + n, v20_start + n) = -1;
402  }
403  }
404  row_counter += Dof<G>;
405 
407 
408  auto it_t = std::ranges::begin(tt);
409  auto it_g = std::ranges::begin(gg);
410 
411  for (auto i = 0u; i + 1 < N; ++i, ++it_t, ++it_g) {
412  const std::size_t v1i_start = idx(1, i);
413  const std::size_t v2i_start = idx(2, i);
414  const std::size_t v3i_start = idx(3, i);
415 
416  const std::size_t v1ip_start = idx(1, i + 1);
417  const std::size_t v2ip_start = idx(2, i + 1);
418 
419  // segment lengths
420  const Scalar<G> Ti = *(it_t + 1) - *it_t;
421  const Scalar<G> Tip = *(it_t + 2) - *(it_t + 1);
422 
423  // pass through control points
424  // v_{1, i} + v_{2, i} + v_{3, i} = x_{i+1} - x_i
425  for (auto n = 0u; n != Dof<G>; ++n) {
426  lhs.insert(row_counter + n, v1i_start + n) = 1;
427  lhs.insert(row_counter + n, v2i_start + n) = 1;
428  lhs.insert(row_counter + n, v3i_start + n) = 1;
429  }
430  rhs.segment(row_counter, Dof<G>) = rminus(*(it_g + 1), *it_g);
431  row_counter += Dof<G>;
432 
433  // velocity continuity
434  // v_{3, i} = v_{1, i+1}
435  for (auto n = 0u; n != Dof<G>; ++n) {
436  lhs.insert(row_counter + n, v3i_start + n) = 1 * Tip;
437  lhs.insert(row_counter + n, v1ip_start + n) = -1 * Ti;
438  }
439  row_counter += Dof<G>;
440 
441  // acceleration continuity (approximate for Lie groups)
442  // v_{2, i} - v_{3, i} = v_{2, i+1} - v_{1, i+1}
443  for (auto n = 0u; n != Dof<G>; ++n) {
444  lhs.insert(row_counter + n, v2i_start + n) = 1 * (Tip * Tip);
445  lhs.insert(row_counter + n, v3i_start + n) = -1 * (Tip * Tip);
446  lhs.insert(row_counter + n, v1ip_start + n) = -1 * (Ti * Ti);
447  lhs.insert(row_counter + n, v2ip_start + n) = 1 * (Ti * Ti);
448  }
449  row_counter += Dof<G>;
450  }
451 
453 
454  const std::size_t v1_nm_start = idx(1, N - 1);
455  const std::size_t v2_nm_start = idx(2, N - 1);
456  const std::size_t v3_nm_start = idx(3, N - 1);
457 
458  // end at last control point
459  // v_{1, n-1} + v_{2, n-1} v_{3, n-1} = x_{n} - x_{n-1}
460  for (auto n = 0u; n != Dof<G>; ++n) {
461  lhs.insert(row_counter + n, v1_nm_start + n) = 1;
462  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
463  lhs.insert(row_counter + n, v3_nm_start + n) = 1;
464  }
465  rhs.segment(row_counter, Dof<G>) = rminus(*(it_g + 1), *it_g);
466  row_counter += Dof<G>;
467 
468  if (v1.has_value()) {
469  for (auto n = 0u; n != Dof<G>; ++n) { lhs.insert(row_counter + n, v3_nm_start + n) = 3; }
470  rhs.segment(row_counter, Dof<G>) = v1.value();
471  } else {
472  // zero second derivative at end:
473  // v_{2, n-1} = v_{3, n-1}
474  for (auto n = 0u; n != Dof<G>; ++n) {
475  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
476  lhs.insert(row_counter + n, v3_nm_start + n) = -1;
477  }
478  }
479 
481 
482  lhs.makeCompressed();
483 
485 
486  Eigen::SparseLU<decltype(lhs), Eigen::COLAMDOrdering<int>> solver(lhs);
487  Eigen::VectorXd result = solver.solve(rhs);
488 
490 
491  std::vector<Bezier<3, G>> segments;
492  segments.reserve(N);
493 
494  it_g = std::ranges::begin(gg);
495 
496  for (auto i = 0u; i != N; ++i, ++it_g) {
497  const std::size_t v1i_start = idx(1, i);
498  const std::size_t v3i_start = idx(3, i);
499 
500  Tangent<G> v1 = result.template segment<Dof<G>>(v1i_start);
501  Tangent<G> v3 = result.template segment<Dof<G>>(v3i_start);
502  // re-compute v2 to compensate for linearization
503  // this ensures points are interpolated, but the cost is
504  // potential non-continuity of the second derivative
505  Tangent<G> v2 = rminus(composition(*(it_g + 1), ::smooth::exp<G>(-v3)),
506  smooth::composition(*it_g, ::smooth::exp<G>(v1)));
507 
508  segments.emplace_back(
509  *it_g, std::array<Tangent<G>, 3>{std::move(v1), std::move(v2), std::move(v3)});
510  }
511 
512  auto take_view = tt | std::views::take(N + 1);
513  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
514 
515  return PiecewiseBezier<3, G>(std::move(knots), std::move(segments));
516 }

530 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
531 PiecewiseBezier<3, G> fit_cubic_bezier_local(const Rt & tt,
532  const Rg & gg,
533  std::optional<Tangent<G>> v0 = {},
534  std::optional<Tangent<G>> v1 = {})
535 {
536  assert(std::ranges::size(tt) >= 2 && std::ranges::size(gg) >= 2);
537  assert(std::ranges::adjacent_find(tt, std::ranges::greater_equal()) == tt.end());
538 
539  // number of intervals
540  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
541 
542  std::vector<Bezier<3, G>> segments(N);
543  std::vector<Tangent<G>> vel(N + 1);
544 
545  auto t_it = std::ranges::begin(tt);
546  auto g_it = std::ranges::begin(gg);
547 
548  vel[0] = v0.value_or(rminus(*(g_it + 1), *g_it) / (*(t_it + 1) - *t_it));
549 
550  ++t_it;
551  ++g_it;
552 
553  for (auto i = 1u; i < N; ++i) {
554  vel[i] = rminus(*(g_it + 1), *(g_it)) / (*(t_it + 1) - *(t_it));
555  ++t_it;
556  ++g_it;
557  }
558 
559  vel[N] = v1.value_or(rminus(*g_it, *(g_it - 1)) / (*t_it - *(t_it - 1)));
560 
561  t_it = std::ranges::begin(tt);
562  g_it = std::ranges::begin(gg);
563 
564  for (auto i = 0u; i < N; ++i) {
565  std::array<Tangent<G>, 3> vs;
566  vs[0] = (*(t_it + 1) - *t_it) * vel[i] / 3;
567  vs[2] = (*(t_it + 1) - *t_it) * vel[i + 1] / 3;
568  vs[1] = rminus(composition(*(g_it + 1), ::smooth::exp<G>(-vs[2])),
569  smooth::composition(*g_it, ::smooth::exp<G>(vs[0])));
570 
571  segments[i] = Bezier<3, G>(*g_it, std::move(vs));
572 
573  ++t_it;
574  ++g_it;
575  }
576 
577  std::vector<double> knots(std::ranges::begin(tt), std::ranges::end(tt));
578  return PiecewiseBezier<3, G>(std::move(knots), std::move(segments));
579 }
