205  G eval(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
206  {
207  // target condition
208  // knots_[istar] <= t < knots_[istar + 1]
209 
210  auto it = utils::binary_interval_search(knots_, t);
211 
212  std::size_t istar;
213  if (it != std::ranges::end(knots_)) {
214  istar = it - std::ranges::begin(knots_);
215  } else if (t < knots_.front()) {
216  istar = 0;
217  } else {
218  istar = knots_.size() - 2;
219  }
220 
221  const double T = knots_[istar + 1] - knots_[istar];
222  const double u = (t - knots_[istar]) / T;
223  const G g = segments_[istar].eval(u, vel, acc);
224 
225  if (vel.has_value()) { vel.value() /= T; }
226  if (acc.has_value()) { acc.value() /= (T * T); }
227 
228  return g;
229  }
230 
231 private:
232  std::vector<double> knots_;
233  std::vector<Bezier<N, G>> segments_;
234 
235  friend class Curve<G>;
236 };

249 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
250 PiecewiseBezier<1, G> fit_linear_bezier(const Rt & tt, const Rg & gg)
251 {
252  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
253  throw std::invalid_argument("fit_linear_bezier: Not enough points");
254  }
255 
256  if (std::ranges::adjacent_find(tt, std::ranges::greater_equal()) != tt.end()) {
257  throw std::invalid_argument("fit_linear_bezier: Interpolation times not strictly increasing");
258  }
259 
260  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
261 
262  std::vector<Bezier<1, G>> segments(N);
263 
264  auto it_g = std::ranges::begin(gg);
265  auto it_t = std::ranges::begin(tt);
266 
267  for (auto i = 0u; i != N; ++i, ++it_t, ++it_g) {
268  segments[i] = Bezier<1, G>(*it_g, std::array<typename G::Tangent, 1>{(*(it_g + 1) - *it_g)});
269  }
270 
271  auto take_view = tt | std::views::take(N + 1);
272  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
273 
274  return PiecewiseBezier<1, G>(std::move(knots), std::move(segments));
275 }

289 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
290 PiecewiseBezier<2, G> fit_quadratic_bezier(const Rt & tt, const Rg & gg)
291 {
292  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
293  throw std::invalid_argument("fit_quadratic_bezier: Not enough points");
294  }
295 
296  if (std::ranges::adjacent_find(tt, std::ranges::greater_equal()) != tt.end()) {
297  throw std::invalid_argument(
298  "fit_quadratic_bezier: Interpolation times not strictly increasing");
299  }
300 
301  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
302 
303  std::vector<Bezier<2, G>> segments(N);
304 
305  auto it_g = std::ranges::begin(gg);
306  auto it_t = std::ranges::begin(tt);
307 
308  typename G::Tangent v0 = (*(it_g + 1) - *it_g) / (*(it_t + 1) - *it_t);
309 
310  for (auto i = 0u; i != N; ++i, ++it_t, ++it_g) {
311  const double dt = *(it_t + 1) - *it_t;
312 
313  // scaled velocity
314  const typename G::Tangent va = v0 * dt;
315 
316  // create segment
317  const typename G::Tangent v1 = va / 2;
318  const typename G::Tangent v2 = *(it_g + 1) - (*it_g * G::exp(va / 2));
319 
320  segments[i] = Bezier<2, G>(*it_g, std::array<typename G::Tangent, 2>{v1, v2});
321 
322  // unscaled end velocity for interval
323  v0 = v2 * 2 / dt;
324  }
325 
326  auto take_view = tt | std::views::take(N + 1);
327  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
328 
329  return PiecewiseBezier<2, G>(std::move(knots), std::move(segments));
330 }

343 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
344 PiecewiseBezier<3, G> fit_cubic_bezier(const Rt & tt,
345  const Rg & gg,
346  std::optional<typename G::Tangent> v0 = {},
347  std::optional<typename G::Tangent> v1 = {})
348 {
349  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
350  throw std::invalid_argument("fit_cubic_bezier: Not enough points");
351  }
352 
353  if (std::ranges::adjacent_find(tt, std::ranges::greater_equal()) != tt.end()) {
354  throw std::invalid_argument("fit_cubic_bezier: Interpolation times not strictly increasing");
355  }
356 
357  // number of intervals
358  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
359 
360  std::size_t NumVars = G::Dof * 3 * N;
361 
362  Eigen::SparseMatrix<typename G::Scalar> lhs;
363  lhs.resize(NumVars, NumVars);
364  Eigen::Matrix<int, -1, 1> nnz = Eigen::Matrix<int, -1, 1>::Constant(NumVars, 3);
365  nnz.head(G::Dof).setConstant(2);
366  nnz.tail(G::Dof).setConstant(2);
367  lhs.reserve(nnz);
368 
369  Eigen::Matrix<typename G::Scalar, -1, 1> rhs(NumVars);
370  rhs.setZero();
371 
372  // variable layout:
373  //
374  // [ v_{1, 0}; v_{2, 0}; v_{3, 0}; v_{1, 1}; v_{2, 1}; v_{3, 1}; ...]
375  //
376  // where v_ji is a Dof-length vector
377 
378  const auto idx = [&](int j, int i) { return 3 * G::Dof * i + G::Dof * (j - 1); };
379 
380  std::size_t row_counter = 0;
381 
383 
384  const std::size_t v10_start = idx(1, 0);
385  const std::size_t v20_start = idx(2, 0);
386 
387  if (v0.has_value()) {
388  for (auto n = 0u; n != G::Dof; ++n) { lhs.insert(row_counter + n, v10_start + n) = 3; }
389  rhs.segment(row_counter, G::Dof) = v0.value();
390  } else {
391  // zero second derivative at start:
392  // v_{1, 0} = v_{2, 0}
393  for (auto n = 0u; n != G::Dof; ++n) {
394  lhs.insert(row_counter + n, v10_start + n) = 1;
395  lhs.insert(row_counter + n, v20_start + n) = -1;
396  }
397  }
398  row_counter += G::Dof;
399 
401 
402  auto it_t = std::ranges::begin(tt);
403  auto it_g = std::ranges::begin(gg);
404 
405  for (auto i = 0u; i + 1 < N; ++i, ++it_t, ++it_g) {
406  const std::size_t v1i_start = idx(1, i);
407  const std::size_t v2i_start = idx(2, i);
408  const std::size_t v3i_start = idx(3, i);
409 
410  const std::size_t v1ip_start = idx(1, i + 1);
411  const std::size_t v2ip_start = idx(2, i + 1);
412 
413  // segment lengths
414  const typename G::Scalar Ti = *(it_t + 1) - *it_t;
415  const typename G::Scalar Tip = *(it_t + 2) - *(it_t + 1);
416 
417  // pass through control points
418  // v_{1, i} + v_{2, i} + v_{3, i} = x_{i+1} - x_i
419  for (auto n = 0u; n != G::Dof; ++n) {
420  lhs.insert(row_counter + n, v1i_start + n) = 1;
421  lhs.insert(row_counter + n, v2i_start + n) = 1;
422  lhs.insert(row_counter + n, v3i_start + n) = 1;
423  }
424  rhs.segment(row_counter, G::Dof) = *(it_g + 1) - *(it_g);
425  row_counter += G::Dof;
426 
427  // velocity continuity
428  // v_{3, i} = v_{1, i+1}
429  for (auto n = 0u; n != G::Dof; ++n) {
430  lhs.insert(row_counter + n, v3i_start + n) = 1 * Tip;
431  lhs.insert(row_counter + n, v1ip_start + n) = -1 * Ti;
432  }
433  row_counter += G::Dof;
434 
435  // acceleration continuity (approximate for Lie groups)
436  // v_{2, i} - v_{3, i} = v_{2, i+1} - v_{1, i+1}
437  for (auto n = 0u; n != G::Dof; ++n) {
438  lhs.insert(row_counter + n, v2i_start + n) = 1 * (Tip * Tip);
439  lhs.insert(row_counter + n, v3i_start + n) = -1 * (Tip * Tip);
440  lhs.insert(row_counter + n, v1ip_start + n) = -1 * (Ti * Ti);
441  lhs.insert(row_counter + n, v2ip_start + n) = 1 * (Ti * Ti);
442  }
443  row_counter += G::Dof;
444  }
445 
447 
448  const std::size_t v1_nm_start = idx(1, N - 1);
449  const std::size_t v2_nm_start = idx(2, N - 1);
450  const std::size_t v3_nm_start = idx(3, N - 1);
451 
452  // end at last control point
453  // v_{1, n-1} + v_{2, n-1} v_{3, n-1} = x_{n} - x_{n-1}
454  for (auto n = 0u; n != G::Dof; ++n) {
455  lhs.insert(row_counter + n, v1_nm_start + n) = 1;
456  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
457  lhs.insert(row_counter + n, v3_nm_start + n) = 1;
458  }
459  rhs.segment(row_counter, G::Dof) = *(it_g + 1) - *it_g;
460  row_counter += G::Dof;
461 
462  if (v1.has_value()) {
463  for (auto n = 0u; n != G::Dof; ++n) { lhs.insert(row_counter + n, v3_nm_start + n) = 3; }
464  rhs.segment(row_counter, G::Dof) = v1.value();
465  } else {
466  // zero second derivative at end:
467  // v_{2, n-1} = v_{3, n-1}
468  for (auto n = 0u; n != G::Dof; ++n) {
469  lhs.insert(row_counter + n, v2_nm_start + n) = 1;
470  lhs.insert(row_counter + n, v3_nm_start + n) = -1;
471  }
472  }
473 
475 
476  lhs.makeCompressed();
477 
479 
480  Eigen::SparseLU<decltype(lhs), Eigen::COLAMDOrdering<int>> solver(lhs);
481  Eigen::VectorXd result = solver.solve(rhs);
482 
484 
485  std::vector<Bezier<3, G>> segments;
486  segments.reserve(N);
487 
488  it_g = std::ranges::begin(gg);
489 
490  for (auto i = 0u; i != N; ++i, ++it_g) {
491  const std::size_t v1i_start = idx(1, i);
492  const std::size_t v3i_start = idx(3, i);
493 
494  typename G::Tangent v1 = result.template segment<G::Dof>(v1i_start);
495  typename G::Tangent v3 = result.template segment<G::Dof>(v3i_start);
496  // re-compute v2 to compensate for linearization
497  // this ensures points are interpolated, but the cost is
498  // potential non-continuity of the second derivative
499  typename G::Tangent v2 = *(it_g + 1) * G::exp(-v3) - (*it_g) * G::exp(v1);
500 
501  segments.emplace_back(
502  *it_g, std::array<typename G::Tangent, 3>{std::move(v1), std::move(v2), std::move(v3)});
503  }
504 
505  auto take_view = tt | std::views::take(N + 1);
506  std::vector<double> knots(std::ranges::begin(take_view), std::ranges::end(take_view));
507 
508  return PiecewiseBezier<3, G>(std::move(knots), std::move(segments));
509 }

523 template<std::ranges::range Rt, std::ranges::range Rg, LieGroup G = std::ranges::range_value_t<Rg>>
524 PiecewiseBezier<3, G> fit_cubic_bezier_local(const Rt & tt,
525  const Rg & gg,
526  std::optional<typename G::Tangent> v0 = {},
527  std::optional<typename G::Tangent> v1 = {})
528 {
529  if (std::ranges::size(tt) < 2 || std::ranges::size(gg) < 2) {
530  throw std::invalid_argument("fit_cubic_bezier_local: Not enough points");
531  }
532 
533  if (std::ranges::adjacent_find(tt, std::ranges::greater_equal()) != tt.end()) {
534  throw std::invalid_argument(
535  "fit_cubic_bezier_local: Interpolation times not strictly increasing");
536  }
537 
538  // number of intervals
539  const std::size_t N = std::min<std::size_t>(std::ranges::size(tt), std::ranges::size(gg)) - 1;
540 
541  std::vector<Bezier<3, G>> segments(N);
542  std::vector<typename G::Tangent> vel(N + 1);
543 
544  auto t_it = std::ranges::begin(tt);
545  auto g_it = std::ranges::begin(gg);
546 
547  vel[0] = v0.value_or((*(g_it + 1) - *g_it) / (*(t_it + 1) - *t_it));
548 
549  ++t_it;
550  ++g_it;
551 
552  for (auto i = 1u; i < N; ++i) {
553  vel[i] = (*(g_it + 1) - *(g_it)) / (*(t_it + 1) - *(t_it));
554  ++t_it;
555  ++g_it;
556  }
557 
558  vel[N] = v1.value_or((*g_it - *(g_it - 1)) / (*t_it - *(t_it - 1)));
559 
560  t_it = std::ranges::begin(tt);
561  g_it = std::ranges::begin(gg);
562 
563  for (auto i = 0u; i < N; ++i) {
564  std::array<typename G::Tangent, 3> vs;
565  vs[0] = (*(t_it + 1) - *t_it) * vel[i] / 3;
566  vs[2] = (*(t_it + 1) - *t_it) * vel[i + 1] / 3;
567  vs[1] = *(g_it + 1) * G::exp(-vs[2]) - (*g_it) * G::exp(vs[0]);
568 
569  segments[i] = Bezier<3, G>(*g_it, std::move(vs));
570 
571  ++t_it;
572  ++g_it;
573  }
574 
575  std::vector<double> knots(std::ranges::begin(tt), std::ranges::end(tt));
576  return PiecewiseBezier<3, G>(std::move(knots), std::move(segments));
577 }
