126 template<typename S = double>
127 CastT<S, G>
128 operator()(const S & t, OptTangent<CastT<S, G>> vel = {}, OptTangent<CastT<S, G>> acc = {}) const
129 {
130 // index of relevant interval
131 int64_t istar = static_cast<int64_t>((static_cast<double>(t) - t0_) / dt_);
132
133 S u;
134 // clamp to end of range if necessary
135 if (istar < 0) {
136 istar = 0;
137 u = S(0);
138 } else if (istar + static_cast<int64_t>(K + 1) > static_cast<int64_t>(ctrl_pts_.size())) {
139 istar = static_cast<int64_t>(ctrl_pts_.size() - K - 1);
140 u = S(1);
141 } else {
142 u = std::clamp<S>((t - S(t0_) - S(istar) * S(dt_)) / S(dt_), S(0.), S(1.));
143 }
144
145 CastT<S, G> g = cspline_eval_gs<K>(
146 // clang-format off
147 ctrl_pts_
148 | std::views::drop(istar)
149 | std::views::take(int64_t(K + 1)) // gcc 11.1 bug can't handle uint64_t
150 | std::views::transform([](const auto & glocal) { return cast<S>(glocal); }),
151 // clang-format on
152 B_.template cast<S>(),
153 u,
154 vel,
155 acc);
156
157 if (vel.has_value()) { vel.value() /= S(dt_); }
158 if (acc.has_value()) { acc.value() /= S(dt_ * dt_); }
159
160 return g;
161 }
