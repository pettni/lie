148 template<typename S = double>
149 CastT<S, G> operator()(
150 const S & t,
151 detail::OptTangent<CastT<S, G>> vel = {},
152 detail::OptTangent<CastT<S, G>> acc = {}) const
153 {
154 // index of relevant interval
155 int64_t istar = static_cast<int64_t>((static_cast<double>(t) - t0_) / dt_);
156
157 S u;
158 // clamp to end of range if necessary
159 if (istar < 0) {
160 istar = 0;
161 u = S(0);
162 } else if (istar + K + 1 > ctrl_pts_.size()) {
163 istar = ctrl_pts_.size() - K - 1;
164 u = S(1);
165 } else {
166 u = std::clamp<S>((t - S(t0_) - S(istar * dt_)) / S(dt_), S(0.), S(1.));
167 }
168
169 CastT<S, G> g = cspline_eval<K>(
170 // clang-format off
171 ctrl_pts_
172 | std::views::drop(istar)
173 | std::views::take(int64_t(K + 1)) // gcc 11.1 bug can't handle uint64_t
174 | std::views::transform([](const auto & glocal) { return cast<S>(glocal); }),
175 // clang-format on
176 B_.template cast<S>(),
177 u,
178 vel,
179 acc);
180
181 if (vel.has_value()) { vel.value() /= S(dt_); }
182 if (acc.has_value()) { acc.value() /= S(dt_ * dt_); }
183
184 return g;
185 }
186
187private:
188 // cumulative basis functions
189 static constexpr auto B_s_ = polynomial_cumulative_basis<PolynomialBasis::Bspline, K, double>();
190 inline static const Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> B_ =
191 Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>>(B_s_[0].data());
192
193 double t0_, dt_;
194 std::vector<G> ctrl_pts_;
