144  G operator()(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
145  {
146  // index of relevant interval
147  int64_t istar = static_cast<int64_t>((t - t0_) / dt_);
148 
149  double u;
150  // clamp to end of range if necessary
151  if (istar < 0) {
152  istar = 0;
153  u = 0;
154  } else if (istar + K + 1 > ctrl_pts_.size()) {
155  istar = ctrl_pts_.size() - K - 1;
156  u = 1;
157  } else {
158  u = std::clamp<double>((t - t0_ - istar * dt_) / dt_, 0., 1.);
159  }
160 
161  // gcc 11.1 bug can't handle uint64_t
162  G g = cspline_eval<K>(
163  ctrl_pts_ | std::views::drop(istar) | std::views::take(int64_t(K + 1)), B_, u, vel, acc);
164 
165  if (vel.has_value()) { vel.value() /= dt_; }
166  if (acc.has_value()) { acc.value() /= (dt_ * dt_); }
167 
168  return g;
169  }
