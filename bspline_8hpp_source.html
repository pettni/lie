148  G operator()(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
149  {
150  // index of relevant interval
151  int64_t istar = static_cast<int64_t>((t - t0_) / dt_);
152 
153  double u;
154  // clamp to end of range if necessary
155  if (istar < 0) {
156  istar = 0;
157  u = 0;
158  } else if (istar + K + 1 > ctrl_pts_.size()) {
159  istar = ctrl_pts_.size() - K - 1;
160  u = 1;
161  } else {
162  u = (t - t0_ - istar * dt_) / dt_;
163  }
164 
165  constexpr auto Mstatic = detail::cum_coefmat<CSplineType::BSPLINE, double, K>().transpose();
166  Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> M(Mstatic[0].data());
167 
168  G g = cspline_eval<K>(
169  ctrl_pts_ | std::views::drop(istar) | std::views::take(K + 1), M, u, vel, acc);
170 
171  if (vel.has_value()) { vel.value() /= dt_; }
172  if (acc.has_value()) { acc.value() /= (dt_ * dt_); }
173 
174  return g;
175  }

195 template<std::size_t K,
196  std::ranges::range Rt,
197  std::ranges::range Rg,
198  LieGroup G = std::ranges::range_value_t<Rg>>
199 BSpline<K, G> fit_bspline(const Rt & tt, const Rg & gg, double dt)
200 {
201  static_assert(std::is_same_v<std::ranges::range_value_t<Rt>, double>, "Rt value type is double");
202 
203  assert(std::ranges::adjacent_find(tt, std::ranges::greater_equal()) == tt.end());
204 
205  auto [tmin_ptr, tmax_ptr] = std::minmax_element(std::ranges::begin(tt), std::ranges::end(tt));
206 
207  const double t0 = *tmin_ptr;
208  const double t1 = *tmax_ptr;
209 
210  const std::size_t NumData = std::min(std::ranges::size(tt), std::ranges::size(gg));
211  const std::size_t NumPts = K + static_cast<std::size_t>((t1 - t0 + dt) / dt);
212 
213  constexpr auto Mstatic = detail::cum_coefmat<CSplineType::BSPLINE, double, K>().transpose();
214  Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> M(Mstatic[0].data());
215 
216  auto f = [&](const auto & var) {
217  Eigen::VectorXd ret(Dof<G> * NumData);
218 
219  Eigen::SparseMatrix<double, Eigen::RowMajor> Jac;
220  Jac.resize(Dof<G> * NumData, Dof<G> * NumPts);
221  Jac.reserve(Eigen::Matrix<int, -1, 1>::Constant(Dof<G> * NumData, Dof<G> * (K + 1)));
222 
223  auto t_iter = std::ranges::begin(tt);
224  auto g_iter = std::ranges::begin(gg);
225 
226  for (auto i = 0u; i != NumData; ++t_iter, ++g_iter, ++i) {
227  const int64_t istar = static_cast<int64_t>((*t_iter - t0) / dt);
228  const double u = (*t_iter - t0 - istar * dt) / dt;
229 
230  Eigen::Matrix<double, Dof<G>, (K + 1) * Dof<G>> d_vali_pts;
231  auto g_spline = cspline_eval<K>(
232  var | std::views::drop(istar) | std::views::take(K + 1), M, u, {}, {}, d_vali_pts);
233 
234  const Tangent<G> resi = rminus(g_spline, *g_iter);
235 
236  ret.segment<Dof<G>>(i * Dof<G>) = resi;
237 
238  const Eigen::Matrix<double, Dof<G>, Dof<G>> d_resi_vali = dr_expinv<G>(resi);
239  const Eigen::Matrix<double, Dof<G>, (K + 1) * Dof<G>> d_resi_pts = d_resi_vali * d_vali_pts;
240 
241  for (auto r = 0u; r != Dof<G>; ++r) {
242  for (auto c = 0u; c != Dof<G> * (K + 1); ++c) {
243  Jac.insert(i * Dof<G> + r, istar * Dof<G> + c) = d_resi_pts(r, c);
244  }
245  }
246  }
247 
248  Jac.makeCompressed();
249 
250  return std::make_pair(std::move(ret), Eigen::MatrixXd(Jac));
251  };
252 
253  // create optimization variable
254  ManifoldVector<G> ctrl_pts(NumPts);
255 
256  // create initial guess
257  auto t_iter = std::ranges::begin(tt);
258  auto g_iter = std::ranges::begin(gg);
259  for (auto i = 0u; i != NumPts; ++i) {
260  const double t_target = t0 + (i - static_cast<double>(K - 1) / 2) * dt;
261  while (t_iter + 1 < std::ranges::end(tt)
262  && std::abs(t_target - *(t_iter + 1)) < std::abs(t_target - *t_iter)) {
263  ++t_iter;
264  ++g_iter;
265  }
266  ctrl_pts[i] = *g_iter;
267  }
268 
269  // fit to data with loose convergence criteria
270  MinimizeOptions opts;
271  opts.ftol = 1e-3;
272  opts.ptol = 1e-3;
273  opts.max_iter = 10;
274  minimize<diff::Type::ANALYTIC>(f, smooth::wrt(ctrl_pts), opts);
275 
276  return BSpline<K, G>(t0, dt, std::move(ctrl_pts));
277 }
