167template<LieImplemented... _Gs>
168class Bundle : public BundleBase<Bundle<_Gs...>>
169{

174public:
178 template<typename... S>
179 requires(std::is_assignable_v<_Gs, S> &&...)
180 Bundle(S &&... gs)
181 {
182 const auto tpl = std::forward_as_tuple(gs...);
183 utils::static_for<sizeof...(_Gs)>(
184 [this, &tpl](auto i) { Base::template part<i>() = std::get<i>(tpl); });
185 }
