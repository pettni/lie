99 Scalar angle() const
100 {
101 using std::atan2;
102
103 return atan2(
104 static_cast<const _Derived &>(*this).coeffs().x(),
105 static_cast<const _Derived &>(*this).coeffs().y());
106 }
107
111 Scalar scaling() const
112 {
113 using std::sqrt;
114
115 return sqrt(
116 static_cast<const _Derived &>(*this).coeffs().x()
117 * static_cast<const _Derived &>(*this).coeffs().x()
118 + static_cast<const _Derived &>(*this).coeffs().y()
119 * static_cast<const _Derived &>(*this).coeffs().y());
120 }
