80 auto dr_ceres(auto && f, auto && x)
81 {
82  // There is potential to improve thie speed of this by reducing casting.
83  // The ceres Jet type supports binary operations with e.g. double, but currently
84  // the Lie operations require everything to have a uniform scalar type. Enabling
85  // plus and minus for different scalars would thus save some casts.
86  using Result = decltype(std::apply(f, x));
87  using Scalar = ::smooth::Scalar<Result>;
88 
89  static_assert(Manifold<Result>, "f(x) is not a Manifold");
90 
91  Result fval = std::apply(f, x);
92 
93  static constexpr Eigen::Index Nx = wrt_Dof<decltype(x)>();
94  static constexpr Eigen::Index Ny = Dof<Result>;
95  const Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x);
96  const Eigen::Index ny = dof<Result>(fval);
97 
98  static_assert(Nx > 0, "Ceres autodiff does not support dynamic sizes");
99 
100  Eigen::Matrix<Scalar, Nx, 1> a = Eigen::Matrix<Scalar, Nx, 1>::Zero(nx);
101  Eigen::Matrix<Scalar, Ny, 1> b(ny);
102  Eigen::Matrix<Scalar, Ny, Nx, (Nx == 1) ? Eigen::ColMajor : Eigen::RowMajor> jac(ny, nx);
103 
104  const auto f_deriv = [&]<typename T>(const T * in, T * out) {
105  Eigen::Map<const Eigen::Matrix<T, Nx, 1>> mi(in, nx);
106  Eigen::Map<Eigen::Matrix<T, Ny, 1>> mo(out, ny);
107  mo =
108  rminus<CastT<T, Result>>(std::apply(f, wrt_rplus(wrt_cast<T>(x), mi)), cast<T, Result>(fval));
109  return true;
110  };
111  const Scalar * a_ptr[1] = {a.data()};
112  Scalar * jac_ptr[1] = {jac.data()};
113 
114  ceres::internal::AutoDifferentiate<Dof<Result>, ceres::internal::StaticParameterDims<Nx>>(
115  f_deriv, a_ptr, b.size(), b.data(), jac_ptr);
116 
117  return std::make_pair(std::move(fval), Eigen::Matrix<Scalar, Ny, Nx>(jac));
118 }
