71 template<std::size_t K, LieGroup G, typename Derived>
72 inline G cspline_eval_diff(
73  const std::ranges::sized_range auto & diff_points,
74  const Eigen::MatrixBase<Derived> & Bcum,
75  Scalar<G> u,
76  detail::OptTangent<G> vel = {},
77  detail::OptTangent<G> acc = {},
78  detail::OptJacobian<G, K> der = {}) noexcept
79 {
80  assert(std::ranges::size(diff_points) == K);
81  assert(Bcum.cols() == K + 1);
82  assert(Bcum.rows() == K + 1);
83 
84  const auto U = monomial_derivatives<K, 2, Scalar<G>>(u);
85 
86  Eigen::Map<const Eigen::Vector<Scalar<G>, K + 1>> uvec(U[0].data());
87  Eigen::Map<const Eigen::Vector<Scalar<G>, K + 1>> duvec(U[1].data());
88  Eigen::Map<const Eigen::Vector<Scalar<G>, K + 1>> d2uvec(U[2].data());
89 
90  if (vel.has_value()) { vel.value().setZero(); }
91  if (acc.has_value()) { acc.value().setZero(); }
92 
93  const Eigen::Index xdof = dof(*std::ranges::cbegin(diff_points));
94 
95  G g = Identity<G>(xdof);
96 
97  for (auto j = 1u; const auto & v : diff_points) {
98  const Scalar<G> Btilde = uvec.dot(Bcum.col(j));
99 
100  const G exp_Bt_v = ::smooth::exp<G>(Btilde * v);
101 
102  g = composition(g, exp_Bt_v);
103 
104  if (vel.has_value() || acc.has_value()) {
105  const Scalar<G> dBtilde = duvec.dot(Bcum.col(j));
106  const auto Adj = Ad(inverse(exp_Bt_v));
107 
108  if (vel.has_value()) {
109  vel.value().applyOnTheLeft(Adj);
110  vel.value().noalias() += dBtilde * v;
111  }
112 
113  if (acc.has_value()) {
114  const Scalar<G> d2Btilde = d2uvec.dot(Bcum.col(j));
115  acc.value().applyOnTheLeft(Adj);
116  acc.value().noalias() += dBtilde * ad<G>(vel.value()) * v;
117  acc.value().noalias() += d2Btilde * v;
118  }
119  }
120 
121  ++j;
122  }
123 
124  if (der.has_value()) {
125  der.value().setZero();
126 
127  G z2inv = Identity<G>(xdof);
128 
129  for (auto jrev = 0u; jrev <= K; ++jrev) {
130  const auto j = K - jrev; // j: K -> 0
131 
132  if (j != K) {
133  const Scalar<G> Btilde_jp = uvec.dot(Bcum.col(j + 1));
134  const Tangent<G> & vjp = *std::ranges::next(std::ranges::cbegin(diff_points), j);
135  const Tangent<G> sjp = Btilde_jp * vjp;
136 
137  der.value().template block<Dof<G>, Dof<G>>(0, j * Dof<G>).noalias() -=
138  Btilde_jp * Ad(z2inv) * dr_exp<G>(sjp) * dl_expinv<G>(vjp);
139 
140  z2inv = composition(z2inv, ::smooth::exp<G>(-sjp));
141  }
142 
143  const Scalar<G> Btilde_j = uvec.dot(Bcum.col(j));
144 
145  if (j != 0u) {
146  const Tangent<G> & vj = *std::ranges::next(std::ranges::cbegin(diff_points), j - 1);
147  der.value().template block<Dof<G>, Dof<G>>(0, j * Dof<G>).noalias() +=
148  Btilde_j * Ad(z2inv) * dr_exp<G>(Btilde_j * vj) * dr_expinv<G>(vj);
149  } else {
150  der.value().template block<Dof<G>, Dof<G>>(0, j * Dof<G>).noalias() += Btilde_j * Ad(z2inv);
151  }
152  }
153  }
154 
155  return g;
156 }

174 template<
175  std::size_t K,
176  std::ranges::range R,
177  typename Derived,
178  LieGroup G = std::ranges::range_value_t<R>>
179 inline G cspline_eval(
180  const R & gs,
181  const Eigen::MatrixBase<Derived> & Bcum,
182  Scalar<G> u,
183  detail::OptTangent<G> vel = {},
184  detail::OptTangent<G> acc = {},
185  detail::OptJacobian<G, K> der = {}) noexcept
186 {
187  assert(std::ranges::size(gs) == K + 1);
188 
189  auto b1 = std::ranges::cbegin(gs);
190  auto b2 = std::ranges::cbegin(gs) + 1;
191  std::array<Tangent<G>, K> diff_pts;
192  for (auto i = 0u; i != K; ++i) { diff_pts[i] = rminus(*b2++, *b1++); }
193 
194  return composition(
195  *std::ranges::begin(gs), cspline_eval_diff<K, G>(diff_pts, Bcum, u, vel, acc, der));
196 }
