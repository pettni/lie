71template<std::size_t K, LieGroup G, typename Derived>
72inline G cspline_eval_diff(
73 std::ranges::sized_range auto && diff_points,
74 const Eigen::MatrixBase<Derived> & Bcum,
75 Scalar<G> u,
76 detail::OptTangent<G> vel = {},
77 detail::OptTangent<G> acc = {},
78 detail::OptJacobian<G, K> der = {}) noexcept
79{
80 assert(std::ranges::size(diff_points) == K);
81 assert(Bcum.cols() == K + 1);
82 assert(Bcum.rows() == K + 1);
83
84 const auto U = monomial_derivatives<K, 2, Scalar<G>>(u);
85
86 Eigen::Map<const Eigen::Vector<Scalar<G>, K + 1>> uvec(U[0].data());
87 Eigen::Map<const Eigen::Vector<Scalar<G>, K + 1>> duvec(U[1].data());
88 Eigen::Map<const Eigen::Vector<Scalar<G>, K + 1>> d2uvec(U[2].data());
89
90 if (vel.has_value()) { vel.value().setZero(); }
91 if (acc.has_value()) { acc.value().setZero(); }
92
93 const Eigen::Index xdof = dof(*std::ranges::cbegin(diff_points));
94
95 G g = Identity<G>(xdof);
96
97 for (const auto & [j, v] : utils::zip(std::views::iota(1u), diff_points)) {
98 const Scalar<G> Btilde = uvec.dot(Bcum.col(j));
99
100 const G exp_Bt_v = ::smooth::exp<G>(Btilde * v);
101
102 g = composition(g, exp_Bt_v);
103
104 if (vel.has_value() || acc.has_value()) {
105 const Scalar<G> dBtilde = duvec.dot(Bcum.col(j));
106 const auto Adj = Ad(inverse(exp_Bt_v));
107
108 if (vel.has_value()) {
109 vel.value().applyOnTheLeft(Adj);
110 vel.value().noalias() += dBtilde * v;
111 }
112
113 if (acc.has_value()) {
114 const Scalar<G> d2Btilde = d2uvec.dot(Bcum.col(j));
115 acc.value().applyOnTheLeft(Adj);
116 acc.value().noalias() += dBtilde * ad<G>(vel.value()) * v;
117 acc.value().noalias() += d2Btilde * v;
118 }
119 }
120 }
121
122 if (der.has_value()) {
123 der.value().setZero();
124
125 G z2inv = Identity<G>(xdof);
126
127 for (const auto j : std::views::iota(0u, K + 1) | std::views::reverse) {
128 // j : K -> 0 (inclusive)
129
130 if (j != K) {
131 const Scalar<G> Btilde_jp = uvec.dot(Bcum.col(j + 1));
132 const Tangent<G> & vjp = *std::ranges::next(std::ranges::cbegin(diff_points), j);
133 const Tangent<G> sjp = Btilde_jp * vjp;
134
135 der.value().template block<Dof<G>, Dof<G>>(0, j * Dof<G>).noalias() -=
136 Btilde_jp * Ad(z2inv) * dr_exp<G>(sjp) * dl_expinv<G>(vjp);
137
138 z2inv = composition(z2inv, ::smooth::exp<G>(-sjp));
139 }
140
141 const Scalar<G> Btilde_j = uvec.dot(Bcum.col(j));
142
143 if (j != 0u) {
144 const Tangent<G> & vj = *std::ranges::next(std::ranges::cbegin(diff_points), j - 1);
145 der.value().template block<Dof<G>, Dof<G>>(0, j * Dof<G>).noalias() +=
146 Btilde_j * Ad(z2inv) * dr_exp<G>(Btilde_j * vj) * dr_expinv<G>(vj);
147 } else {
148 der.value().template block<Dof<G>, Dof<G>>(0, j * Dof<G>).noalias() += Btilde_j * Ad(z2inv);
149 }
150 }
151 }
152
153 return g;
154}

172template<
173 std::size_t K,
174 std::ranges::range R,
175 typename Derived,
176 LieGroup G = std::ranges::range_value_t<R>>
177inline G cspline_eval(
178 R && gs,
179 const Eigen::MatrixBase<Derived> & Bcum,
180 Scalar<G> u,
181 detail::OptTangent<G> vel = {},
182 detail::OptTangent<G> acc = {},
183 detail::OptJacobian<G, K> der = {}) noexcept
184{
185 assert(std::ranges::size(gs) == K + 1);
186
187 constexpr auto sub = [](const auto & x1, const auto & x2) { return rminus(x2, x1); };
188 const auto diff_pts = gs | utils::views::pairwise_transform(sub);
189
190 return composition(
191 *std::ranges::begin(gs), cspline_eval_diff<K, G>(diff_pts, Bcum, u, vel, acc, der));
192}
