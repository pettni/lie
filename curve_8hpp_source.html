306  G operator()(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
307  {
308  if (empty() || t < 0) {
309  if (vel.has_value()) { vel.value().setZero(); }
310  if (acc.has_value()) { acc.value().setZero(); }
311  return Identity<G>();
312  }
313 
314  if (t > t_max()) {
315  if (vel.has_value()) { vel.value().setZero(); }
316  if (acc.has_value()) { acc.value().setZero(); }
317  return end_g_.back();
318  }
319 
320  const auto istar = find_idx(t);
321 
322  const double ta = istar == 0 ? 0 : end_t_[istar - 1];
323  const double T = end_t_[istar] - ta;
324 
325  const double Del = seg_Del_[istar];
326  const double u = std::clamp<double>(seg_T0_[istar] + Del * (t - ta) / T, 0, 1);
327 
328  constexpr auto Mstatic = detail::cum_coefmat<CSplineType::BEZIER, double, 3>().transpose();
329  Eigen::Map<const Eigen::Matrix<double, 3 + 1, 3 + 1, Eigen::RowMajor>> M(Mstatic[0].data());
330 
331  G g0 = istar == 0 ? Identity<G>() : end_g_[istar - 1];
332 
333  // compensate for cropped intervals
334  if (seg_T0_[istar] > 0) {
335  g0 = composition(g0, inverse(cspline_eval_diff<3, G>(vs_[istar], M, seg_T0_[istar])));
336  }
337 
338  const G g = composition(g0, cspline_eval_diff<3, G>(vs_[istar], M, u, vel, acc));
339 
340  if (vel.has_value()) { vel.value() *= Del / T; }
341  if (acc.has_value()) { acc.value() *= Del * Del / (T * T); }
342 
343  return g;
344  }

357  Tangent<G> arclength(double t) const
358  {
359  Tangent<G> ret = Tangent<G>::Zero();
360 
361  for (auto i = 0u; i < end_t_.size(); ++i) {
362  // check if we have reached t
363  if (i > 0 && t <= end_t_[i - 1]) { break; }
364 
365  // polynomial coefficients: dx/dt = At^2 + Bt + C
366  const Tangent<G> A = 3 * vs_[i][0] - 6 * vs_[i][1] + 3 * vs_[i][2];
367  const Tangent<G> B = -6 * vs_[i][0] + 6 * vs_[i][1];
368  const Tangent<G> C = 3 * vs_[i][0];
369 
370  const double ta = i == 0 ? 0 : end_t_[i - 1];
371  const double tb = end_t_[i];
372 
373  const double ua = seg_T0_[i];
374  const double ub = ua + seg_Del_[i] * (std::min<double>(t, tb) - ta) / (tb - ta);
375 
376  // integrate | A u2 + B u + C | for u \in [ua, ub]
377  for (auto k = 0u; k < Dof<G>; ++k) {
378  ret(k) += detail::integrate_absolute_polynomial(ua, ub, A(k), B(k), C(k));
379  }
380  }
381 
382  return ret;
383  }

395  Curve crop(double ta, double tb = std::numeric_limits<double>::infinity()) const
396  {
397  ta = std::max<double>(ta, 0);
398  tb = std::min<double>(tb, t_max());
399 
400  if (tb <= ta) { return Curve(); }
401 
402  const std::size_t i0 = find_idx(ta);
403  std::size_t Nseg = find_idx(tb) + 1 - i0;
404 
405  // prevent last segment from being empty
406  if (Nseg >= 2 && end_t_[i0 + Nseg - 2] == tb) { --Nseg; }
407 
408  // state at new from beginning of curve
409  const G ga = operator()(ta);
410 
411  std::vector<double> end_t(Nseg);
412  std::vector<G> end_g(Nseg);
413  std::vector<std::array<Tangent<G>, 3>> vs(Nseg);
414  std::vector<double> seg_T0(Nseg), seg_Del(Nseg);
415 
416  // copy over all relevant segments
417  for (auto i = 0u; i != Nseg; ++i) {
418  if (i == Nseg - 1) {
419  end_t[i] = tb - ta;
420  end_g[i] = composition(inverse(ga), operator()(tb));
421  } else {
422  end_t[i] = end_t_[i0 + i] - ta;
423  end_g[i] = composition(inverse(ga), end_g_[i0 + i]);
424  }
425  vs[i] = vs_[i0 + i];
426  seg_T0[i] = seg_T0_[i0 + i];
427  seg_Del[i] = seg_Del_[i0 + i];
428  }
429 
430  // crop first segment
431  {
432  const double tta = 0;
433  const double ttb = end_t_[i0];
434  const double sa = ta;
435  const double sb = ttb;
436 
437  seg_T0[0] += seg_Del[0] * (sa - tta) / (ttb - tta);
438  seg_Del[0] *= (sb - sa) / (ttb - tta);
439  }
440 
441  // crop last segment
442  {
443  const double tta = Nseg == 1 ? ta : end_t_[Nseg - 2];
444  const double ttb = end_t_[Nseg - 1];
445  const double sa = tta;
446  const double sb = tb;
447 
448  seg_T0[Nseg - 1] += seg_Del[Nseg - 1] * (sa - tta) / (ttb - tta);
449  seg_Del[Nseg - 1] *= (sb - sa) / (ttb - tta);
450  }
451 
452  // create new curve with appropriate body velocities
453  Curve<G> ret;
454  ret.end_t_ = std::move(end_t);
455  ret.end_g_ = std::move(end_g);
456  ret.vs_ = std::move(vs);
457  ret.seg_T0_ = std::move(seg_T0);
458  ret.seg_Del_ = std::move(seg_Del);
459  return ret;
460  }

463  std::size_t find_idx(double t) const
464  {
465  // target condition:
466  // end_t_[istar - 1] <= t < end_t_[istar]
467 
468  std::size_t istar = 0;
469 
470  auto it = utils::binary_interval_search(end_t_, t);
471  if (it != end_t_.end()) {
472  istar = std::min<std::size_t>((it - end_t_.begin()) + 1, end_t_.size() - 1);
473  }
474 
475  return istar;
476  }
477 
478  // segment i is defined by
479  //
480  // - time interval: end_t_[i-1], end_t_[i]
481  // - g interval: end_g_[i-1], end_g_[i]
482  // - velocities: vs_[i]
483  // - crop: seg_T0_[i], seg_Del_[i]
484  //
485  // s.t.
486  //
487  // u(t) = T0[i] + Del[i] * (t - end_t[i-1]) / (end_t[i] - end_t[i-1])
488  //
489  // x(t) = xu(u(t)) where xu is spline defined in [0, 1]
490 
491  // segment end times
492  std::vector<double> end_t_;
493 
494  // segment end points
495  std::vector<G> end_g_;
496 
497  // segment bezier velocities
498  std::vector<std::array<Tangent<G>, 3>> vs_;
499 
500  // segment crop information
501  std::vector<double> seg_T0_, seg_Del_;

508 class Reparameterization
509 {
510 public:
512  struct Data
513  {
515  double t;
517  double s;
519  double v;
521  double a;
522  };
523 
529  Reparameterization(double smax, std::vector<Data> && d) : smax_(smax), d_(std::move(d)) {}
530 
532  double t_min() const { return 0; }
533 
535  double t_max() const { return d_.back().t; }
536 
543  double operator()(double t, double & ds, double & d2s) const
544  {
545  if (d_.size() == 1) {
546  ds = 0;
547  d2s = 0;
548  return std::min(d_.front().s, smax_);
549  }
550 
551  auto it =
552  utils::binary_interval_search(d_, t, [](const Data & d, double t) { return d.t <=> t; });
553  const double tau = t - it->t;
554 
555  ds = it->v + it->a * tau;
556  d2s = it->a;
557 
558  return std::clamp(it->s + it->v * tau + it->a * tau * tau / 2, 0., smax_);
559  }
560 
561 private:
562  double smax_;
563  std::vector<Data> d_;
564 };

586 template<LieGroup G>
587 Reparameterization reparameterize_curve(const Curve<G> & curve,
588  const Tangent<G> & vel_min,
589  const Tangent<G> & vel_max,
590  const Tangent<G> & acc_min,
591  const Tangent<G> & acc_max,
592  double start_vel = 1,
593  double end_vel = std::numeric_limits<double>::infinity(),
594  bool slower_only = false,
595  double dt = 0.05,
596  double eps = 1e-2)
597 {
598  // BACKWARDS PASS WITH MINIMAL ACCELERATION
599 
600  Eigen::Vector2d state(curve.t_max(), end_vel); // s, v
601  std::vector<double> xx, yy;
602 
603  do {
604  Tangent<G> vel, acc;
605  curve(state.x(), vel, acc);
606 
607  // clamp velocity to not exceed constraints
608  for (auto i = 0u; i != Dof<G>; ++i) {
609  // regular velocity constraint
610  if (vel(i) > eps) {
611  state.y() = std::min<double>(state.y(), vel_max(i) / vel(i));
612  } else if (vel(i) < -eps) {
613  state.y() = std::min<double>(state.y(), vel_min(i) / vel(i));
614  }
615 
616  // ensure a = 0 is feasible for acceleration constraint
617  if (acc(i) > eps) {
618  state.y() = std::min<double>(state.y(), std::sqrt(acc_max(i) / acc(i)));
619  } else if (acc(i) < -eps) {
620  state.y() = std::min<double>(state.y(), std::sqrt(acc_min(i) / acc(i)));
621  }
622  }
623 
624  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
625  // ensure v stays positive
626  state.y() = std::max(state.y(), eps);
627 
628  if (xx.empty() || state.x() < xx.back()) {
629  xx.push_back(state.x());
630  yy.push_back(state.y());
631  }
632 
633  double a = 0;
634  if (vel.cwiseAbs().maxCoeff() > eps) {
635  // figure minimal allowed acceleration
636  a = -std::numeric_limits<double>::infinity();
637  const Tangent<G> upper = (acc_max - acc * state.y() * state.y()).cwiseMax(Tangent<G>::Zero());
638  const Tangent<G> lower = (acc_min - acc * state.y() * state.y()).cwiseMin(Tangent<G>::Zero());
639  for (auto i = 0u; i != Dof<G>; ++i) {
640  if (vel(i) > eps) {
641  a = std::max<double>(a, lower(i) / vel(i));
642  } else if (vel(i) < -eps) {
643  a = std::max<double>(a, upper(i) / vel(i));
644  }
645  }
646  }
647 
648  state.x() -= state.y() * dt - a * dt * dt / 2;
649  state.y() -= dt * a;
650  } while (state.x() > 0);
651 
652  if (xx.empty() || state.x() < xx.back()) {
653  xx.push_back(state.x());
654  yy.push_back(state.y());
655  }
656 
657  // fit a spline to get v(s)
658 
659  std::reverse(xx.begin(), xx.end());
660  std::reverse(yy.begin(), yy.end());
661  auto v_func = fit_linear_bezier(xx, yy); // linear guarantees positive velocities
662 
663  // FORWARD PASS WITH MAXIMAL ACCELERATION
664 
665  std::vector<Reparameterization::Data> dd;
666  state << 0, start_vel;
667 
668  // clamp velocity to not exceed upper bound
669  state.y() = std::clamp<double>(state.y(), eps, v_func(0));
670  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
671 
672  do {
673  Tangent<G> vel, acc;
674  curve(state.x(), vel, acc);
675 
676  double a = 0;
677  if (vel.cwiseAbs().maxCoeff() > eps) {
678  // figure maximal allowed acceleration
679  a = std::numeric_limits<double>::infinity();
680  const Tangent<G> upper = (acc_max - acc * state.y() * state.y()).cwiseMax(Tangent<G>::Zero());
681  const Tangent<G> lower = (acc_min - acc * state.y() * state.y()).cwiseMin(Tangent<G>::Zero());
682  for (auto i = 0u; i != Dof<G>; ++i) {
683  if (vel(i) > eps) {
684  a = std::min<double>(a, upper(i) / vel(i));
685  } else if (vel(i) < -eps) {
686  a = std::min<double>(a, lower(i) / vel(i));
687  }
688  }
689 
690  const double new_s = state.x() + state.y() * dt + a * dt * dt / 2;
691  double new_v = state.y() + dt * a;
692 
693  // clamp velocity to not exceed upper bound
694  new_v = std::clamp<double>(new_v, eps, v_func(new_s));
695  if (slower_only) { new_v = std::min<double>(new_v, 1); }
696 
697  a = (new_v - state.y()) / dt;
698  }
699 
700  dd.push_back({
701  .t = dd.empty() ? 0 : dd.back().t + dt,
702  .s = state.x(),
703  .v = state.y(),
704  .a = a,
705  });
706 
707  state.x() += dd.back().v * dt + dd.back().a * dt * dt / 2;
708  state.y() += dd.back().a * dt;
709  } while (state.x() < curve.t_max());
710 
711  dd.push_back({
712  .t = dd.empty() ? 0 : dd.back().t + dt,
713  .s = state.x(),
714  .v = state.y(),
715  .a = 0,
716  });
717 
718  return Reparameterization(curve.t_max(), std::move(dd));
719 }
