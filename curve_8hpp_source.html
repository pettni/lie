306  G eval(double t, detail::OptTangent<G> vel = {}, detail::OptTangent<G> acc = {}) const
307  {
308  const auto istar = find_idx(t);
309 
310  const double ta = istar == 0 ? 0 : end_t_[istar - 1];
311  const double T = end_t_[istar] - ta;
312 
313  const double Del = seg_Del_[istar];
314  const double u = std::clamp<double>(seg_T0_[istar] + Del * (t - ta) / T, 0, 1);
315 
316  constexpr auto Mstatic = detail::cum_coefmat<CSplineType::BEZIER, double, 3>().transpose();
317  Eigen::Map<const Eigen::Matrix<double, 3 + 1, 3 + 1, Eigen::RowMajor>> M(Mstatic[0].data());
318 
319  G g0 = istar == 0 ? G::Identity() : end_g_[istar - 1];
320 
321  // compensate for cropped intervals
322  if (seg_T0_[istar] > 0) {
323  g0 *= cspline_eval_diff<3, G>(vs_[istar], M, seg_T0_[istar]).inverse();
324  }
325 
326  const G g = g0 * cspline_eval_diff<3, G>(vs_[istar], M, u, vel, acc);
327 
328  if (vel.has_value()) { vel.value() *= Del / T; }
329  if (acc.has_value()) { acc.value() *= Del * Del / (T * T); }
330 
331  return g;
332  }

344  Curve crop(double ta, double tb = std::numeric_limits<double>::infinity()) const
345  {
346  ta = std::max<double>(ta, 0);
347  tb = std::min<double>(tb, t_max());
348 
349  if (tb <= ta) { return Curve(); }
350 
351  const std::size_t i0 = find_idx(ta);
352  std::size_t Nseg = find_idx(tb) + 1 - i0;
353 
354  // prevent last segment from being empty
355  if (Nseg >= 2 && end_t_[i0 + Nseg - 2] == tb) { --Nseg; }
356 
357  // state at new from beginning of curve
358  const G ga = eval(ta);
359 
360  std::vector<double> end_t(Nseg);
361  std::vector<G> end_g(Nseg);
362  std::vector<std::array<typename G::Tangent, 3>> vs(Nseg);
363  std::vector<double> seg_T0(Nseg), seg_Del(Nseg);
364 
365  // copy over all relevant segments
366  for (auto i = 0u; i != Nseg; ++i) {
367  if (i == Nseg - 1) {
368  end_t[i] = tb - ta;
369  end_g[i] = ga.inverse() * eval(tb);
370  } else {
371  end_t[i] = end_t_[i0 + i] - ta;
372  end_g[i] = ga.inverse() * end_g_[i0 + i];
373  }
374  vs[i] = vs_[i0 + i];
375  seg_T0[i] = seg_T0_[i0 + i];
376  seg_Del[i] = seg_Del_[i0 + i];
377  }
378 
379  // crop first segment
380  {
381  const double tta = 0;
382  const double ttb = end_t_[i0];
383  const double sa = ta;
384  const double sb = ttb;
385 
386  seg_T0[0] += seg_Del[0] * (sa - tta) / (ttb - tta);
387  seg_Del[0] *= (sb - sa) / (ttb - tta);
388  }
389 
390  // crop last segment
391  {
392  const double tta = Nseg == 1 ? ta : end_t_[Nseg - 2];
393  const double ttb = end_t_[Nseg - 1];
394  const double sa = tta;
395  const double sb = tb;
396 
397  seg_T0[Nseg - 1] += seg_Del[Nseg - 1] * (sa - tta) / (ttb - tta);
398  seg_Del[Nseg - 1] *= (sb - sa) / (ttb - tta);
399  }
400 
401  // create new curve with appropriate body velocities
402  Curve<G> ret;
403  ret.end_t_ = std::move(end_t);
404  ret.end_g_ = std::move(end_g);
405  ret.vs_ = std::move(vs);
406  ret.seg_T0_ = std::move(seg_T0);
407  ret.seg_Del_ = std::move(seg_Del);
408  return ret;
409  }

492  double eval(double t, double & ds, double & d2s) const
493  {
494  if (tt_.size() == 1) {
495  ds = 0;
496  d2s = 0;
497  return std::min(ss_.front(), smax_);
498  }
499 
500  // target: tt_[idx] <= t < tt_[idx + 1]
501  auto it = utils::binary_interval_search(tt_, t);
502 
503  std::size_t idx;
504  if (it != std::ranges::end(tt_)) {
505  idx = it - std::ranges::begin(tt_);
506  } else if (t < tt_.front()) {
507  idx = 0;
508  } else {
509  idx = tt_.size() - 1;
510  }
511 
512  const double tau = t - tt_[idx];
513 
514  ds = vv_[idx];
515  d2s = idx + 2 < tt_.size() ? 0 : (vv_[idx + 1] - vv_[idx]) / (tt_[idx + 1] - tt_[idx]);
516 
517  return std::clamp(ss_[idx] + tau * ds, 0., smax_);
518  }
519 
520 private:
521  double smax_;
522  std::vector<double> tt_, ss_, vv_;
523 };

544 template<LieGroup G>
545 Reparameterization reparameterize_curve(const Curve<G> & curve,
546  const typename G::Tangent & vel_min,
547  const typename G::Tangent & vel_max,
548  const typename G::Tangent & acc_min,
549  const typename G::Tangent & acc_max,
550  double start_vel = 1,
551  double end_vel = std::numeric_limits<double>::infinity(),
552  bool slower_only = false,
553  double dt = 0.05,
554  double eps = 1e-2)
555 {
556  const double zeps = std::numeric_limits<double>::epsilon() / eps;
557 
558  // BACKWARDS PASS
559 
560  Eigen::Vector2d state(curve.t_max(), end_vel); // s, v
561  std::vector<double> xx;
562  std::vector<smooth::T1d> yy;
563 
564  do {
565  typename G::Tangent vel, acc;
566  curve.eval(state.x(), vel, acc);
567 
568  // clamp velocity to not exceed constraints
569  for (auto i = 0u; i != G::Dof; ++i) {
570  if (vel(i) > zeps) {
571  state.y() = std::min<double>(state.y(), vel_max(i) / vel(i));
572  } else if (vel(i) < -zeps) {
573  state.y() = std::min<double>(state.y(), vel_min(i) / vel(i));
574  }
575  }
576  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
577  // ensure v stays positive
578  state.y() = std::max(state.y(), eps);
579 
580  if (xx.empty() || state.x() < xx.back()) {
581  xx.push_back(state.x());
582  yy.push_back(T1d(Eigen::Matrix<double, 1, 1>(state.y())));
583  }
584 
585  if (vel.cwiseAbs().maxCoeff() <= zeps) {
586  // skip over if curve is stationary
587  state.x() -= eps;
588  state.y() = 1. / eps;
589  } else {
590  // figure minimal allowed acceleration
591  double a = -std::numeric_limits<double>::infinity();
592  for (auto i = 0u; i != G::Dof; ++i) {
593  const typename G::Tangent upper = acc_max - acc * state.y() * state.y();
594  const typename G::Tangent lower = acc_min - acc * state.y() * state.y();
595  if (vel(i) > zeps) {
596  a = std::max<double>(a, lower(i) / vel(i));
597  } else if (vel(i) < -zeps) {
598  a = std::max<double>(a, upper(i) / vel(i));
599  }
600  }
601 
602  state.x() -= state.y() * dt - a * dt * dt / 2;
603  state.y() -= dt * a;
604  }
605  } while (state.x() > 0);
606 
607  if (xx.empty() || state.x() < xx.back()) {
608  xx.push_back(state.x());
609  yy.push_back(T1d(Eigen::Matrix<double, 1, 1>(state.y())));
610  }
611 
612  // fit a spline to get v(s)
613 
614  std::reverse(xx.begin(), xx.end());
615  std::reverse(yy.begin(), yy.end());
616  auto v_func = fit_linear_bezier(xx, yy); // linear guarantees positive velocities
617 
618  // FORWARD PASS
619 
620  std::vector<double> tt, ss, vv;
621  state << 0, start_vel;
622 
623  // clamp velocity to not exceed upper bound
624  state.y() = std::min<double>(state.y(), v_func.eval(0).rn().x());
625  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
626 
627  do {
628  typename G::Tangent vel, acc;
629  curve.eval(state.x(), vel, acc);
630 
631  if (vel.cwiseAbs().maxCoeff() <= zeps) {
632  // skip over without storing if curve is stationary
633  state.x() += eps;
634  } else {
635  // figure maximal allowed acceleration
636  double a = std::numeric_limits<double>::infinity();
637  for (auto i = 0u; i != G::Dof; ++i) {
638  const typename G::Tangent upper = acc_max - acc * state.y() * state.y();
639  const typename G::Tangent lower = acc_min - acc * state.y() * state.y();
640  if (vel(i) > zeps) {
641  a = std::min<double>(a, upper(i) / vel(i));
642  } else if (vel(i) < -zeps) {
643  a = std::min<double>(a, lower(i) / vel(i));
644  }
645  }
646 
647  tt.push_back(tt.empty() ? 0 : tt.back() + dt);
648  ss.push_back(state.x());
649  vv.push_back(state.y());
650 
651  state.x() += state.y() * dt;
652  state.y() += dt * a;
653 
654  // clamp velocity to not exceed upper bound
655  state.y() = std::min<double>(state.y(), v_func.eval(state.x()).rn().x());
656  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
657  // ensure v stays positive
658  state.y() = std::max(state.y(), eps);
659  }
660  } while (state.x() < curve.t_max());
661 
662  tt.push_back(tt.empty() ? 0 : tt.back() + dt);
663  ss.push_back(state.x());
664 
665  return Reparameterization(curve.t_max(), std::move(tt), std::move(ss), std::move(vv));
666 }
