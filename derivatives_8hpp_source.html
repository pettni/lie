49template<typename At, typename dAt, typename Bt, typename dBt>
50inline auto d_matrix_product(const At & A, const dAt & dA, const Bt & B, const dBt & dB)
51{
52 using Scalar = std::common_type_t<
53 typename At::Scalar,
54 typename dAt::Scalar,
55 typename Bt::Scalar,
56 typename dBt::Scalar>;
57
58 static constexpr int N = At::ColsAtCompileTime;
59 static constexpr int M = Bt::RowsAtCompileTime;
60 static constexpr int Nvar = []() -> int {
61 if constexpr (dAt::ColsAtCompileTime > 0 && N > 0) {
62 return dAt::ColsAtCompileTime / N;
63 } else if (dBt::ColsAtCompileTime > 0 && M > 0) {
64 return dBt::ColsAtCompileTime / M;
65 } else {
66 return -1;
67 }
68 }();
69
70 const auto n = A.cols();
71 [[maybe_unused]] const auto k = A.rows();
72 const auto m = B.rows();
73 const auto nvar = dA.cols() / (n);
74
75 assert(k == B.cols());
76 assert(nvar == dB.size() / (m * k));
77
78 static constexpr int dAB_cols = (M > 0 && Nvar > 0) ? M * Nvar : -1;
79
80 Eigen::Matrix<Scalar, N, dAB_cols> dAB = B.transpose() * dA;
81 for (auto i = 0u; i < n; ++i) {
82 for (auto j = 0u; j < m; ++j) {
83 dAB.template middleCols<Nvar>(i * nvar, nvar) +=
84 A(i, j) * dB.template middleCols<Nvar>(j * Nvar, nvar);
85 }
86 }
87 return dAB;
88}

100template<typename JfT, typename HfT, typename JgT, typename HgT>
101inline auto d2_fog(const JfT & Jf, const HfT & Hf, const JgT & Jg, const HgT & Hg)
102{
103 using Scalar = std::common_type_t<
104 typename JfT::Scalar,
105 typename HfT::Scalar,
106 typename JgT::Scalar,
107 typename HgT::Scalar>;
108
109 static constexpr int No = JfT::RowsAtCompileTime;
110 static constexpr int Ny = JfT::ColsAtCompileTime;
111 static constexpr int Nx = JgT::ColsAtCompileTime;
112
113 const auto no = Jf.rows();
114 const auto ny = Jf.cols();
115
116 [[maybe_unused]] const auto ni = Jg.rows();
117 const auto nx = Jg.cols();
118
119 // check some dimensions
120 assert(ny == ni);
121 assert(Hf.rows() == ny);
122 assert(Hf.cols() == no * ny);
123 assert(Hg.rows() == nx);
124 assert(Hg.cols() == ni * nx);
125
126 Eigen::Matrix<Scalar, Nx, (No == -1 || Nx == -1) ? -1 : No * Nx> ret(nx, no * nx);
127 ret.setZero();
128
129 for (auto i = 0u; i < no; ++i) {
130 ret.template block<Nx, Nx>(0, i * nx, nx, nx) +=
131 Jg.transpose() * Hf.template middleCols<Ny>(i * ny, ny) * Jg;
132 }
133
134 for (auto i = 0u; i < Jf.outerSize(); ++i) {
135 for (Eigen::InnerIterator it(Jf, i); it; ++it) {
136 ret.template block<Nx, Nx>(0, it.row() * nx) +=
137 it.value() * Hg.template middleCols<Nx>(it.col() * nx, nx);
138 }
139 }
140
141 return ret;
142}

159template<LieGroup G>
160Hessian<G> d2r_rminus(const Tangent<G> & e)
161{
162 const auto J = dr_expinv<G>(e);
163
164 auto res = d2r_expinv<G>(e);
165 for (auto j = 0u; j < Dof<G>; ++j) {
166 res.template block<Dof<G>, Dof<G>>(0, j * e.size(), e.size(), e.size()).applyOnTheRight(J);
167 }
168 return res;
169}

188Eigen::Matrix<Scalar<G>, Dof<G>, Dof<G>> d2r_rminus_squarednorm(const Tangent<G> & e)
189{
190 const TangentMap<G> J1 = dr_rminus<G>(e); // N x N
191 const Hessian<G> H1 = d2r_rminus<G>(e); // N x (N*N)
