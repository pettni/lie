57 auto dr_numerical(auto && f, auto && x)
58 {
59  using Wrt = decltype(x);
60  using Result = decltype(std::apply(f, x));
61  using Scalar = ::smooth::Scalar<Result>;
62 
63  static_assert(Manifold<Result>, "f(x) is not a Manifold");
64 
65  const Scalar eps = std::sqrt(Eigen::NumTraits<Scalar>::epsilon());
66 
67  // arguments are modified below, so we create a copy of those that come in as const
68  auto x_nc = wrt_copy_if_const(std::forward<Wrt>(x));
69  Result val = std::apply(f, x_nc);
70 
71  // static sizes
72  static constexpr Eigen::Index Nx = wrt_Dof<Wrt>();
73  static constexpr Eigen::Index Ny = Dof<Result>;
74 
75  // dynamic sizes
76  Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x_nc);
77  Eigen::Index ny = dof<Result>(val);
78 
79  // output variable
80  Eigen::Matrix<Scalar, Ny, Nx> jac(ny, nx);
81 
82  Eigen::Index index_pos = 0;
83  utils::static_for<std::tuple_size_v<std::decay_t<Wrt>>>([&](auto i) {
84  auto & w = std::get<i>(x_nc);
85  using W = std::decay_t<decltype(w)>;
86 
87  static constexpr Eigen::Index Nx_j = Dof<W>;
88  const int nx_j = dof<W>(w);
89 
90  for (auto j = 0; j != nx_j; ++j) {
91  Scalar eps_j = eps;
92  if constexpr (std::is_base_of_v<Eigen::MatrixBase<W>, W>) {
93  // scale step size if we are in Rn
94  eps_j *= abs(w[j]);
95  if (eps_j == 0.) { eps_j = eps; }
96  }
97  w = rplus<W>(w, (eps_j * Eigen::Vector<Scalar, Nx_j>::Unit(nx_j, j)).eval());
98  jac.col(index_pos + j) = rminus<Result>(std::apply(f, x_nc), val) / eps_j;
99  w = rplus<W>(w, (-eps_j * Eigen::Vector<Scalar, Nx_j>::Unit(nx_j, j)).eval());
100  }
101  index_pos += nx_j;
102  });
103 
104  return std::make_pair(std::move(val), std::move(jac));
105 }

144 template<Type D>
145 auto dr(auto && f, auto && x)
146 {
147  using F = decltype(f);
148  using Wrt = decltype(x);
149 
150  if constexpr (D == Type::NUMERICAL) {
151  return detail::dr_numerical(std::forward<F>(f), std::forward<Wrt>(x));
152  } else if constexpr (D == Type::AUTODIFF) {
153 #ifdef SMOOTH_DIFF_AUTODIFF
154  return dr_autodiff(std::forward<F>(f), std::forward<Wrt>(x));
155 #else
156  static_assert(D != Type::AUTODIFF, "compat/autodiff.hpp header not included");
157 #endif
158  } else if constexpr (D == Type::CERES) {
159 #ifdef SMOOTH_DIFF_CERES
160  return dr_ceres(std::forward<F>(f), std::forward<Wrt>(x));
161 #else
162  static_assert(D != Type::CERES, "compat/ceres.hpp header not included");
163 #endif
164  } else if constexpr (D == Type::ANALYTIC) {
165  return std::apply(f, std::forward<Wrt>(x));
166  } else if constexpr (D == Type::DEFAULT) {
167  return dr<DefaultType>(std::forward<F>(f), std::forward<Wrt>(x));
168  }
169 }
