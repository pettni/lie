57template<std::size_t K = 1>
58 requires(K >= 1 && K <= 2)
59auto dr_numerical(auto && f, auto && x)
60{
61 using Wrt = decltype(x);
62 using Result = decltype(std::apply(f, x));
63 using Scalar = ::smooth::Scalar<Result>;
64
65 static constexpr auto NumArgs = std::tuple_size_v<std::decay_t<Wrt>>;
66
67 static_assert(Manifold<Result>, "f(x) is not a Manifold");
68
69 const Scalar eps = std::sqrt(Eigen::NumTraits<Scalar>::epsilon());
70
71 // arguments are modified below, so we create a copy of those that come in as const
72 auto x_nc = wrt_copy_if_const(std::forward<Wrt>(x));
73 Result fval = std::apply(f, x_nc);
74
75 // static sizes
76 static constexpr Eigen::Index Nx = wrt_Dof<Wrt>();
77 static constexpr Eigen::Index Ny = Dof<Result>;
78
79 // dynamic sizes
80 Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x_nc);
81 Eigen::Index ny = dof<Result>(fval);
82
83 // output variable
84 Eigen::Matrix<Scalar, Ny, Nx> J(ny, nx);
85
86 if constexpr (K == 1) {
87 Eigen::Index I0 = 0;
88 utils::static_for<NumArgs>([&](auto i) {
89 auto & w = std::get<i>(x_nc);
90 using W = std::decay_t<decltype(w)>;
91
92 static constexpr Eigen::Index Nx_j = Dof<W>;
93 const int nx_j = dof<W>(w);
94
95 for (auto j = 0; j != nx_j; ++j) {
96 Scalar eps_j = eps;
97 if constexpr (std::is_base_of_v<Eigen::MatrixBase<W>, W>) {
98 // scale step size if we are in Rn
99 eps_j *= abs(w[j]);
100 if (eps_j == 0.) { eps_j = eps; }
101 }
102 w = rplus<W>(w, (eps_j * Eigen::Vector<Scalar, Nx_j>::Unit(nx_j, j)).eval());
103 J.col(I0 + j) = rminus<Result>(std::apply(f, x_nc), fval) / eps_j;
104 w = rplus<W>(w, (-eps_j * Eigen::Vector<Scalar, Nx_j>::Unit(nx_j, j)).eval());
105 }
106 I0 += nx_j;
107 });
108
109 return std::make_pair(std::move(fval), std::move(J));
110 }
111
112 if constexpr (K == 2) {
113 const auto sqrteps = std::sqrt(eps);
114
115 Eigen::Matrix<Scalar, Nx, std::min(Nx, Ny) == -1 ? -1 : Nx * Ny> H(nx, nx * ny);
116
117 Eigen::Index I0 = 0;
118 utils::static_for<NumArgs>([&](auto i0) {
119 auto & w0 = std::get<i0>(x_nc);
120 using W0 = std::decay_t<decltype(w0)>;
121 static constexpr Eigen::Index Nx_i0 = Dof<W0>;
122 const int nx_i0 = dof<W0>(w0);
123
124 Eigen::Index I1 = 0;
125 utils::static_for<NumArgs>([&](auto i1) {
126 auto & w1 = std::get<i1>(x_nc);
127 using W1 = std::decay_t<decltype(w1)>;
128 static constexpr Eigen::Index Nx_i1 = Dof<W1>;
129 const int nx_i1 = dof<W1>(w1);
130
131 for (auto k0 = 0; k0 != nx_i0; ++k0) {
132 Scalar eps0 = sqrteps;
133 if constexpr (std::is_base_of_v<Eigen::MatrixBase<W0>, W0>) {
134 eps0 *= abs(w0[k0]);
135 if (eps0 == 0.) { eps0 = sqrteps; }
136 }
137
138 w0 = rplus<W0>(w0, eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, k0));
139 const Result F10 = std::apply(f, x_nc);
140 w0 = rplus<W0>(w0, -eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, k0));
141
142 const Eigen::Matrix<Scalar, Ny, 1> d1 = rminus(F10, fval);
143
144 J.col(I0 + k0) = d1 / eps0;
145
146 for (auto k1 = 0; k1 < nx_i1; ++k1) {
147 Scalar eps1 = sqrteps;
148 if constexpr (std::is_base_of_v<Eigen::MatrixBase<W1>, W1>) {
149 eps1 *= abs(w1[k1]);
150 if (eps1 == 0.) { eps1 = sqrteps; }
151 }
152
153 // do this in order to ensure we return to same point on spaces with non-zero brackets
154 w1 = rplus<W1>(w1, eps1 * Eigen::Vector<Scalar, Nx_i1>::Unit(nx_i1, k1));
155 const Result F01 = std::apply(f, x_nc);
156 w0 = rplus<W0>(w0, eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, k0));
157 const Result F11 = std::apply(f, x_nc);
158 w0 = rplus<W0>(w0, -eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, k0));
159 w1 = rplus<W1>(w1, -eps1 * Eigen::Vector<Scalar, Nx_i1>::Unit(nx_i1, k1));
160
161 const Eigen::Matrix<Scalar, Ny, 1> d2 = (rminus(F11, F01) - d1) / eps0 / eps1;
162 for (auto j = 0u; j < ny; ++j) { H(I0 + k0, j * nx + I1 + k1) = d2(j); }
163 }
164 }
165 I1 += nx_i1;
166 });
167 I0 += nx_i0;
168 });
169
170 return std::make_tuple(std::move(fval), std::move(J), std::move(H));
171 }
172}

175template<class F, class Wrt>
176concept diffable_order1 = requires(F && f, Wrt && wrt)
177{
178 {std::apply(f, wrt)};
179 {std::apply(std::bind_front(std::mem_fn(&std::decay_t<decltype(f)>::jacobian), f), wrt)};
180};

183template<class F, class Wrt>
184concept diffable_order2 = diffable_order1<F, Wrt> && requires(F && f, Wrt && wrt)
185{
186 {std::apply(std::bind_front(std::mem_fn(&std::decay_t<decltype(f)>::hessian), f), wrt)};

237template<std::size_t K, Type D>
238auto dr(auto && f, auto && x)
239{
240 using F = decltype(f);
241 using Wrt = decltype(x);
242
243 if constexpr (K == 0u) {
244 // Only function value needed
245
246 return std::make_tuple(std::apply(f, x));
247
248 } else if constexpr (D == Type::Numerical) {
249 // Numerical
250
251 return detail::dr_numerical<K>(std::forward<F>(f), std::forward<Wrt>(x));
252
253 } else if constexpr (D == Type::Autodiff) {
254 // Autodiff
255
256#ifdef SMOOTH_DIFF_AUTODIFF
257 return dr_autodiff<K>(std::forward<F>(f), std::forward<Wrt>(x));
258#else
259 static_assert(D != Type::Autodiff, "compat/autodiff.hpp must be included before diff.hpp");
260#endif
261
262 } else if constexpr (D == Type::Ceres) {
263 // Ceres
264
265#ifdef SMOOTH_DIFF_CERES
266 static_assert(K == 1, "Only K = 1 supported with Ceres");
267 return dr_ceres(std::forward<F>(f), std::forward<Wrt>(x));
268#else
269 static_assert(D != Type::Ceres, "compat/ceres.hpp header must be included before diff.hpp");
270#endif
271
272 } else if constexpr (D == Type::Analytic) {
273 // Analytic
274
275 auto fval = std::apply(f, x);
276 auto dfval =
277 std::apply(std::bind_front(std::mem_fn(&std::decay_t<decltype(f)>::jacobian), f), x);
278 if constexpr (K == 1) {
279 return std::make_tuple(std::move(fval), std::move(dfval));
280 } else if constexpr (K == 2) {
281 auto d2fval =
282 std::apply(std::bind_front(std::mem_fn(&std::decay_t<decltype(f)>::hessian), f), x);
283 return std::make_tuple(fval, dfval, d2fval);
284 }
285
286 } else if constexpr (D == Type::Default) {
287 // Default
288
289 // If analytical derivatives exist
290 if constexpr (K == 1 && detail::diffable_order1<decltype(f), decltype(x)>) {
291 return dr<K, Type::Analytic>(std::forward<F>(f), std::forward<Wrt>(x));
292 } else if constexpr (K == 2 && detail::diffable_order2<decltype(f), decltype(x)>) {
293 return dr<K, Type::Analytic>(std::forward<F>(f), std::forward<Wrt>(x));
294 } else {
295 // Use best available method
296 return dr<K, DefaultType>(std::forward<F>(f), std::forward<Wrt>(x));
297 }
298 }
299}
