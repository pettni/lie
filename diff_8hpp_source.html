57 template<std::size_t K = 1>
58  requires(K >= 1 && K <= 2)
59 auto dr_numerical(auto && f, auto && x)
60 {
61  using Wrt = decltype(x);
62  using Result = decltype(std::apply(f, x));
63  using Scalar = ::smooth::Scalar<Result>;
64 
65  static_assert(Manifold<Result>, "f(x) is not a Manifold");
66 
67  const Scalar eps = std::sqrt(Eigen::NumTraits<Scalar>::epsilon());
68 
69  // arguments are modified below, so we create a copy of those that come in as const
70  auto x_nc = wrt_copy_if_const(std::forward<Wrt>(x));
71  Result F = std::apply(f, x_nc);
72 
73  // static sizes
74  static constexpr Eigen::Index Nx = wrt_Dof<Wrt>();
75  static constexpr Eigen::Index Ny = Dof<Result>;
76 
77  // dynamic sizes
78  Eigen::Index nx = std::apply([](auto &&... args) { return (dof(args) + ...); }, x_nc);
79  Eigen::Index ny = dof<Result>(F);
80 
81  // output variable
82  Eigen::Matrix<Scalar, Ny, Nx> J(ny, nx);
83 
84  if constexpr (K == 1) {
85  Eigen::Index I0 = 0;
86  utils::static_for<std::tuple_size_v<std::decay_t<Wrt>>>([&](auto i) {
87  auto & w = std::get<i>(x_nc);
88  using W = std::decay_t<decltype(w)>;
89 
90  static constexpr Eigen::Index Nx_j = Dof<W>;
91  const int nx_j = dof<W>(w);
92 
93  for (auto j = 0; j != nx_j; ++j) {
94  Scalar eps_j = eps;
95  if constexpr (std::is_base_of_v<Eigen::MatrixBase<W>, W>) {
96  // scale step size if we are in Rn
97  eps_j *= abs(w[j]);
98  if (eps_j == 0.) { eps_j = eps; }
99  }
100  w = rplus<W>(w, (eps_j * Eigen::Vector<Scalar, Nx_j>::Unit(nx_j, j)).eval());
101  J.col(I0 + j) = rminus<Result>(std::apply(f, x_nc), F) / eps_j;
102  w = rplus<W>(w, (-eps_j * Eigen::Vector<Scalar, Nx_j>::Unit(nx_j, j)).eval());
103  }
104  I0 += nx_j;
105  });
106 
107  return std::make_pair(std::move(F), std::move(J));
108  }
109 
110  if constexpr (K == 2) {
111  static_assert(Ny == 1, "2nd derivative only implemented for scalar functions");
112 
113  Eigen::Matrix<Scalar, Nx, Nx> H(nx, nx);
114 
115  Eigen::Index I0 = 0;
116  utils::static_for<std::tuple_size_v<std::decay_t<Wrt>>>([&](auto i0) {
117  auto & w0 = std::get<i0>(x_nc);
118  using W0 = std::decay_t<decltype(w0)>;
119  static constexpr Eigen::Index Nx_i0 = Dof<W0>;
120  const int nx_i0 = dof<W0>(w0);
121 
122  Eigen::Index I1 = 0;
123  utils::static_for<std::tuple_size_v<std::decay_t<Wrt>>>([&](auto i1) {
124  if (Eigen::Index(i1) > I0) { return; }
125 
126  auto & w1 = std::get<i1>(x_nc);
127  using W1 = std::decay_t<decltype(w1)>;
128  static constexpr Eigen::Index Nx_i1 = Dof<W1>;
129  const int nx_i1 = dof<W1>(w1);
130 
131  for (auto i0 = 0; i0 != nx_i0; ++i0) {
132  Scalar eps0 = std::sqrt(eps);
133  if constexpr (std::is_base_of_v<Eigen::MatrixBase<W0>, W0>) {
134  eps0 *= abs(w0[i0]);
135  if (eps0 == 0.) { eps0 = eps; }
136  }
137 
138  w0 = rplus<W0>(w0, eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, i0));
139  const Result F10 = std::apply(f, x_nc);
140  w0 = rplus<W0>(w0, -eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, i0));
141 
142  J(0, I0 + i0) = (F10 - F) / eps0;
143 
144  for (auto i1 = 0; i1 <= i0; ++i1) {
145  Scalar eps1 = std::sqrt(eps);
146  if constexpr (std::is_base_of_v<Eigen::MatrixBase<W1>, W1>) {
147  eps1 *= abs(w1[i1]);
148  if (eps1 == 1.) { eps1 = eps; }
149  }
150 
151  w1 = rplus<W1>(w1, eps1 * Eigen::Vector<Scalar, Nx_i1>::Unit(nx_i1, i1));
152  const Result F01 = std::apply(f, x_nc);
153  w0 = rplus<W0>(w0, eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, i0));
154  const Result F11 = std::apply(f, x_nc);
155  w0 = rplus<W0>(w0, -eps0 * Eigen::Vector<Scalar, Nx_i0>::Unit(nx_i0, i0));
156  w1 = rplus<W1>(w1, -eps1 * Eigen::Vector<Scalar, Nx_i1>::Unit(nx_i1, i1));
157 
158  H(I0 + i0, I1 + i1) = H(I1 + i1, I0 + i0) = (F11 - F01 - F10 + F) / (eps0 * eps1);
159  }
160  }
161  I1 += nx_i1;
162  });
163  I0 += nx_i0;
164  });
165 
166  return std::make_tuple(std::move(F), std::move(J), std::move(H));
167  }

210 template<std::size_t K, Type D>
211 auto dr(auto && f, auto && x)
212 {
213  using F = decltype(f);
214  using Wrt = decltype(x);
215 
216  if constexpr (D == Type::Numerical) {
217  return detail::dr_numerical<K>(std::forward<F>(f), std::forward<Wrt>(x));
218  } else if constexpr (D == Type::Autodiff) {
219 #ifdef SMOOTH_DIFF_AUTODIFF
220  return dr_autodiff<K>(std::forward<F>(f), std::forward<Wrt>(x));
221 #else
222  static_assert(D != Type::Autodiff, "compat/autodiff.hpp header not included");
223 #endif
224  } else if constexpr (D == Type::Ceres) {
225  static_assert(K == 1, "Only K = 1 supported with Ceres");
226 #ifdef SMOOTH_DIFF_CERES
227  return dr_ceres(std::forward<F>(f), std::forward<Wrt>(x));
228 #else
229  static_assert(D != Type::Ceres, "compat/ceres.hpp header not included");
230 #endif
231  } else if constexpr (D == Type::Analytic) {
232  return std::apply(f, std::forward<Wrt>(x));
233  } else if constexpr (D == Type::Default) {
234  return dr<K, DefaultType>(std::forward<F>(f), std::forward<Wrt>(x));
235  }
236 }
