52inline double dubins_angle(const smooth::SO2d & x1, const smooth::SO2d & x2, DubinsSegment s)
53{
54 double d = (x2 - x1).x();
55
56 if (s == DubinsSegment::Right) { d = -d; }
57
58 return d >= 0 ? d : 2 * pi + d;
59}

62inline std::array<double, 3>
63dubins_ccc(const smooth::SE2d & target, double R, DubinsSegment c13, DubinsSegment c2)
64{
65 static constexpr double inf = std::numeric_limits<double>::infinity();
66
67 // circle centers
68 Eigen::Vector2d C1{0, c13 == DubinsSegment::Right ? -R : R};
69 Eigen::Vector2d C3 = target * C1;
70
71 // circle center distance
72 const double d13 = (C3 - C1).norm();
73
74 if (d13 < std::numeric_limits<double>::epsilon()) {
75 // if circles coincide we just follow the circle
76 return {dubins_angle(smooth::SO2d::Identity(), target.so2(), c13), 0, 0};
77 }
78
79 if (4 * R <= d13) {
80 // infeasible case
81 return {inf, inf, inf};
82 }
83
84 // positive angle between lines C1-C3 and C1-C2
85 smooth::SO2d A_13_12 = smooth::SO2d(std::sqrt(1. - d13 * d13 / (16 * R * R)), d13 / (4 * R));
86
87 // positive angle between lines C1-C2 and C3-C2 (180 - 2 * A_13_12)
88 smooth::SO2d A_12_32 = smooth::SO2d(pi) * A_13_12.inverse() * A_13_12.inverse();
89
90 // starting angle with respect to first circle center
91 smooth::SO2d alpha0 = c13 == DubinsSegment::Right ? smooth::SO2d(-pi / 2) : smooth::SO2d(pi / 2);
92
93 Eigen::Vector2d C1_C3 = (C3 - C1).normalized();
94
95 smooth::SO2d theta1, theta2;
96 if (c13 == DubinsSegment::Right) {
97 theta1 = alpha0 * smooth::SO2d(C1_C3.y(), C1_C3.x()) * A_13_12.inverse();
98 theta2 = theta1 * A_12_32.inverse();
99 } else {
100 theta1 = alpha0 * smooth::SO2d(C1_C3.y(), C1_C3.x()) * A_13_12;
101 theta2 = theta1 * A_12_32;
102 }
103
104 return {
105 dubins_angle(smooth::SO2d::Identity(), theta1, c13),
106 dubins_angle(theta1, theta2, c2),
107 dubins_angle(theta2, target.so2(), c13),
108 };
109}

112inline std::array<double, 3>
113dubins_csc(const smooth::SE2d & target, double R, DubinsSegment c1, DubinsSegment c3)
114{
115 static constexpr double inf = std::numeric_limits<double>::infinity();
116
117 // circle centers
118 Eigen::Vector2d C1{0, c1 == DubinsSegment::Right ? -R : R};
119 Eigen::Vector2d C3 = target * Eigen::Vector2d{0, c3 == DubinsSegment::Right ? -R : R};
120
121 // distance between circles
122 const double d13 = (C3 - C1).norm();
123
124 if (d13 < std::numeric_limits<double>::epsilon()) {
125 if (c1 == c3) {
126 // same initial and final, just follow the circle
127 return {dubins_angle(smooth::SO2d::Identity(), target.so2(), c1), 0, 0};
128 } else {
129 return {inf, inf, inf};
130 }
131 }
132
133 // direction C1 -> C3
134 Eigen::Vector2d C1_C3 = (C3 - C1).normalized();
135 smooth::SO2d theta(C1_C3.y(), C1_C3.x());
136
137 if (c1 != c3) {
138 if (d13 <= 2 * R) { return {inf, inf, inf}; }
139
140 // positive angle between C1 -> C3 and line between tangent points
141 smooth::SO2d diff(2 * R / d13, std::sqrt(1. - 4 * R * R / (d13 * d13)));
142
143 if (c1 == DubinsSegment::Right && c3 == DubinsSegment::Left) {
144 theta *= diff.inverse(); // Right -> Left: subtract from nominal for later exit
145 } else {
146 theta *= diff; // Left -> Right: add to nominal for later exit
147 }
148 }
149
150 return {
151 dubins_angle(smooth::SO2d::Identity(), theta, c1),
152 (C3 - C1).dot(Eigen::Vector2d(theta.u1().real(), theta.u1().imag())),
153 dubins_angle(theta, target.so2(), c3),
154 };
155}

165inline DubinsDescription dubins(const smooth::SE2d & target, double R)
166{
167 // try all combinations
168 double min_length = std::numeric_limits<double>::infinity();
169
170 DubinsDescription ret;
171
172 {
173 auto [a1, d2, a3] = detail::dubins_csc(target, R, DubinsSegment::Left, DubinsSegment::Left);
174 double len = d2 + R * (a1 + a3);
175 if (len < min_length) {
176 min_length = len;
177 ret = {
178 std::pair<DubinsSegment, double>{DubinsSegment::Left, a1},
179 std::pair<DubinsSegment, double>{DubinsSegment::Straight, d2},
180 std::pair<DubinsSegment, double>{DubinsSegment::Left, a3},
181 };
182 }
183 }
184
185 {
186 auto [a1, d2, a3] = detail::dubins_csc(target, R, DubinsSegment::Left, DubinsSegment::Right);
187 double len = d2 + R * (a1 + a3);
188 if (len < min_length) {
189 min_length = len;
190 ret = {
191 std::pair<DubinsSegment, double>{DubinsSegment::Left, a1},
192 std::pair<DubinsSegment, double>{DubinsSegment::Straight, d2},
193 std::pair<DubinsSegment, double>{DubinsSegment::Right, a3},
194 };
195 }
196 }
197
198 {
199 auto [a1, d2, a3] = detail::dubins_csc(target, R, DubinsSegment::Right, DubinsSegment::Left);
200 double len = d2 + R * (a1 + a3);
201 if (len < min_length) {
202 min_length = len;
203 ret = {
204 std::pair<DubinsSegment, double>{DubinsSegment::Right, a1},
205 std::pair<DubinsSegment, double>{DubinsSegment::Straight, d2},
206 std::pair<DubinsSegment, double>{DubinsSegment::Left, a3},
207 };
208 }
209 }
210
211 {
212 auto [a1, d2, a3] = detail::dubins_csc(target, R, DubinsSegment::Right, DubinsSegment::Right);
213 double len = d2 + R * (a1 + a3);
214 if (len < min_length) {
215 min_length = len;
216 ret = {
217 std::pair<DubinsSegment, double>{DubinsSegment::Right, a1},
218 std::pair<DubinsSegment, double>{DubinsSegment::Straight, d2},
219 std::pair<DubinsSegment, double>{DubinsSegment::Right, a3},
220 };
221 }
222 }
223
224 {
225 auto [a1, a2, a3] = detail::dubins_ccc(target, R, DubinsSegment::Right, DubinsSegment::Left);
226 double len = R * (a1 + a2 + a3);
227 if (len < min_length) {
228 min_length = len;
229 ret = {
230 std::pair<DubinsSegment, double>{DubinsSegment::Right, a1},
231 std::pair<DubinsSegment, double>{DubinsSegment::Left, a2},
232 std::pair<DubinsSegment, double>{DubinsSegment::Right, a3},
233 };
234 }
235 }
236
237 {
238 auto [a1, a2, a3] = detail::dubins_ccc(target, R, DubinsSegment::Left, DubinsSegment::Right);
239 double len = R * (a1 + a2 + a3);
240 if (len < min_length) {
241 min_length = len;
242 ret = {
243 std::pair<DubinsSegment, double>{DubinsSegment::Left, a1},
244 std::pair<DubinsSegment, double>{DubinsSegment::Right, a2},
245 std::pair<DubinsSegment, double>{DubinsSegment::Left, a3},
246 };
247 }
248 }
249
250 return ret;
251}

261template<std::size_t K = 3>
262 requires(K >= 1)
263Spline<K, smooth::SE2d> dubins_curve(const smooth::SE2d & gb, double R = 1)
264{
265 const auto desc = dubins(gb, R);
266
267 Spline<K, smooth::SE2d> ret;
268 for (auto i = 0u; i != 3; ++i) {
269 const auto & [c, l] = desc[i];
270 if (c == DubinsSegment::Left) {
271 ret += Spline<K, smooth::SE2d>::ConstantVelocity(Eigen::Vector3d(1, 0, 1. / R), R * l);
272 } else if (c == DubinsSegment::Right) {
273 ret += Spline<K, smooth::SE2d>::ConstantVelocity(Eigen::Vector3d(1, 0, -1. / R), R * l);
274 } else {
275 ret += Spline<K, smooth::SE2d>::ConstantVelocity(Eigen::Vector3d(1, 0, 0), l);
276 }
277 }
278 return ret;
279}
