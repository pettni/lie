53 inline double dubins_angle(const smooth::SO2d & x1, const smooth::SO2d & x2, DubinsSegment s)
54 {
55  double d = (x2 - x1).x();
56 
57  if (s == DubinsSegment::Right) { d = -d; }
58 
59  return d >= 0 ? d : 2 * pi + d;
60 }

63 inline std::array<double, 3>
64 dubins_ccc(const smooth::SE2d & target, double R, DubinsSegment c13, DubinsSegment c2)
65 {
66  static constexpr double inf = std::numeric_limits<double>::infinity();
67 
68  // circle centers
69  Eigen::Vector2d C1{0, c13 == DubinsSegment::Right ? -R : R};
70  Eigen::Vector2d C3 = target * C1;
71 
72  // circle center distance
73  const double d13 = (C3 - C1).norm();
74 
75  if (d13 < std::numeric_limits<double>::epsilon()) {
76  // if circles coincide we just follow the circle
77  return {dubins_angle(smooth::SO2d::Identity(), target.so2(), c13), 0, 0};
78  }
79 
80  if (4 * R <= d13) {
81  // infeasible case
82  return {inf, inf, inf};
83  }
84 
85  // positive angle between lines C1-C3 and C1-C2
86  smooth::SO2d A_13_12 = smooth::SO2d(std::sqrt(1. - d13 * d13 / (16 * R * R)), d13 / (4 * R));
87 
88  // positive angle between lines C1-C2 and C3-C2 (180 - 2 * A_13_12)
89  smooth::SO2d A_12_32 = smooth::SO2d(pi) * A_13_12.inverse() * A_13_12.inverse();
90 
91  // starting angle with respect to first circle center
92  smooth::SO2d alpha0 = c13 == DubinsSegment::Right ? smooth::SO2d(-pi / 2) : smooth::SO2d(pi / 2);
93 
94  Eigen::Vector2d C1_C3 = (C3 - C1).normalized();
95 
96  smooth::SO2d theta1, theta2;
97  if (c13 == DubinsSegment::Right) {
98  theta1 = alpha0 * smooth::SO2d(C1_C3.y(), C1_C3.x()) * A_13_12.inverse();
99  theta2 = theta1 * A_12_32.inverse();
100  } else {
101  theta1 = alpha0 * smooth::SO2d(C1_C3.y(), C1_C3.x()) * A_13_12;
102  theta2 = theta1 * A_12_32;
103  }
104 
105  return {
106  dubins_angle(smooth::SO2d::Identity(), theta1, c13),
107  dubins_angle(theta1, theta2, c2),
108  dubins_angle(theta2, target.so2(), c13),
109  };
110 }

113 inline std::array<double, 3>
114 dubins_csc(const smooth::SE2d & target, double R, DubinsSegment c1, DubinsSegment c3)
115 {
116  static constexpr double inf = std::numeric_limits<double>::infinity();
117 
118  // circle centers
119  Eigen::Vector2d C1{0, c1 == DubinsSegment::Right ? -R : R};
120  Eigen::Vector2d C3 = target * Eigen::Vector2d{0, c3 == DubinsSegment::Right ? -R : R};
121 
122  // distance between circles
123  const double d13 = (C3 - C1).norm();
124 
125  if (d13 < std::numeric_limits<double>::epsilon()) {
126  if (c1 == c3) {
127  // same initial and final, just follow the circle
128  return {dubins_angle(smooth::SO2d::Identity(), target.so2(), c1), 0, 0};
129  } else {
130  return {inf, inf, inf};
131  }
132  }
133 
134  // direction C1 -> C3
135  Eigen::Vector2d C1_C3 = (C3 - C1).normalized();
136  smooth::SO2d theta(C1_C3.y(), C1_C3.x());
137 
138  if (c1 != c3) {
139  if (d13 <= 2 * R) { return {inf, inf, inf}; }
140 
141  // positive angle between C1 -> C3 and line between tangent points
142  smooth::SO2d diff(2 * R / d13, std::sqrt(1. - 4 * R * R / (d13 * d13)));
143 
144  if (c1 == DubinsSegment::Right && c3 == DubinsSegment::Left) {
145  theta *= diff.inverse(); // Right -> Left: subtract from nominal for later exit
146  } else {
147  theta *= diff; // Left -> Right: add to nominal for later exit
148  }
149  }
150 
151  return {
152  dubins_angle(smooth::SO2d::Identity(), theta, c1),
153  (C3 - C1).dot(Eigen::Vector2d(theta.u1().real(), theta.u1().imag())),
154  dubins_angle(theta, target.so2(), c3),
155  };
156 }

166 inline DubinsDescription dubins(const smooth::SE2d & target, double R)
167 {
168  // try all combinations
169  double min_length = std::numeric_limits<double>::infinity();
170 
171  DubinsDescription ret;
172 
173  {
174  auto [a1, d2, a3] = detail::dubins_csc(target, R, DubinsSegment::Left, DubinsSegment::Left);
175  double len = d2 + R * (a1 + a3);
176  if (len < min_length) {
177  min_length = len;
178  ret = {
179  std::pair<DubinsSegment, double>{DubinsSegment::Left, a1},
180  std::pair<DubinsSegment, double>{DubinsSegment::Straight, d2},
181  std::pair<DubinsSegment, double>{DubinsSegment::Left, a3},
182  };
183  }
184  }
185

188  double len = d2 + R * (a1 + a3);
189  if (len < min_length) {
190  min_length = len;
191  ret = {

262 template<std::size_t K = 3>
263  requires(K >= 1)
264 Spline<K, smooth::SE2d> dubins_curve(const smooth::SE2d & gb, double R = 1)
265 {
266  const auto desc = dubins(gb, R);
267 
268  Spline<K, smooth::SE2d> ret;
269  for (auto i = 0u; i != 3; ++i) {
270  const auto & [c, l] = desc[i];
271  if (c == DubinsSegment::Left) {
272  ret += Spline<K, smooth::SE2d>::ConstantVelocity(Eigen::Vector3d(1, 0, 1. / R), R * l);
273  } else if (c == DubinsSegment::Right) {
274  ret += Spline<K, smooth::SE2d>::ConstantVelocity(Eigen::Vector3d(1, 0, -1. / R), R * l);
275  } else {
276  ret += Spline<K, smooth::SE2d>::ConstantVelocity(Eigen::Vector3d(1, 0, 0), l);
277  }
278  }
279  return ret;
280 }
