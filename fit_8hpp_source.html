170template<SplineSpec SS>
171constexpr int splinespec_max_deriv()
172{
173 int ret = std::max<int>(0, SS::InnCnt);
174
175 for (const auto & x : SS::LeftDeg) { ret = std::max(ret, x); }
176 for (const auto & x : SS::RghtDeg) { ret = std::max(ret, x); }
177
178 return ret;
179}

204template<SplineSpec SS>
205auto splinespec_project(const SS & ss, std::size_t k)
206{
207 using Scalar = Scalar<typename splinespec_extract<SS>::group>;
208
209 typename splinespec_rebind<SS, Scalar>::type ret;
210 for (auto i = 0u; i < ss.LeftDeg.size(); ++i) {
211 ret.left_values[i] = ss.left_values[i].template segment<1>(k);
212 }
213 for (auto i = 0u; i < ss.RghtDeg.size(); ++i) {
214 ret.rght_values[i] = ss.rght_values[i].template segment<1>(k);
215 }
216 return ret;
217}

239Eigen::VectorXd fit_spline_1d(
240 std::ranges::sized_range auto && dt_r,
241 std::ranges::sized_range auto && dx_r,
242 const SplineSpec auto & ss)
243{
244 using namespace std::views;
245
246 using SS = std::decay_t<decltype(ss)>;
247
248 const std::size_t N = std::min(std::ranges::size(dt_r), std::ranges::size(dx_r));
249
250 // coefficient layout is
251 // [ x0 x1 ... Xn ]
252 // where p_i(t) = \sum_k x_i[k] * b_{i,k}(t) defines p on [tvec(i), tvec(i+1)]
253
254 static constexpr auto K = SS::Degree;
255 static constexpr auto D = detail::splinespec_max_deriv<SS>();
256
257 static_assert(K >= D, "K >= D");
258
259 // compile-time matrix algebra
260 static constexpr auto B_s = polynomial_basis<PolynomialBasis::Bernstein, K>();
261 static constexpr auto U0_s = monomial_derivatives<K, D>(0.);
262 static constexpr auto U1_s = monomial_derivatives<K, D>(1.);
263 static constexpr auto U0tB_s = U0_s * B_s;
264 static constexpr auto U1tB_s = U1_s * B_s;
265
266 Eigen::Map<const Eigen::Matrix<double, D + 1, K + 1, Eigen::RowMajor>> U0tB(U0tB_s[0].data());
267 Eigen::Map<const Eigen::Matrix<double, D + 1, K + 1, Eigen::RowMajor>> U1tB(U1tB_s[0].data());
268
269 // d:th derivative of basis polynomial at 0 (resp. 1) is now U0tB.row(d) * x (resp. U1tB.row(d) *
270 // x), where x are the coefficients.
271
272 const std::size_t N_coef = (K + 1) * N;
273 const std::size_t N_eq = ss.LeftDeg.size() // left endpoint
274 + N // value left-segment
275 + (SS::InnCnt >= 0 ? N : 0) // value rght-segment
276 + (SS::InnCnt > 0 ? (N - 1) * (SS::InnCnt) : 0) // derivative continuity
277 + ss.RghtDeg.size(); // right endpiont
278
279 assert(N_coef >= N_eq);
280
281 // CONSTRAINT MATRICES A, b
282
283 Eigen::SparseMatrix<double, Eigen::ColMajor> A(N_eq, N_coef);
284 Eigen::VectorXi A_pattern(N_coef);
285 A_pattern.head(K + 1).setConstant(1 + ss.LeftDeg.size() + (SS::InnCnt >= 0 ? 1 + SS::InnCnt : 0));
286 if (N >= 2) {
287 A_pattern.segment(K + 1, (N - 2) * (K + 1))
288 .setConstant(1 + (SS::InnCnt >= 0 ? 1 + 2 * SS::InnCnt : 0));
289 }
290 A_pattern.tail(K + 1).setConstant(1 + ss.RghtDeg.size() + (SS::InnCnt >= 0 ? 1 + SS::InnCnt : 0));
291 A.reserve(A_pattern);
292
293 Eigen::VectorXd b = Eigen::VectorXd::Zero(N_eq);
294
295 // current inequality counter
296 std::size_t M = 0;
297
298 // curve beg derivative constraints
299 for (auto i = 0u; i < ss.LeftDeg.size(); ++i) {
300 for (auto j = 0u; j < K + 1; ++j) { A.insert(M, j) = U0tB(ss.LeftDeg[i], j); }
301 b(M++) = ss.left_values[i].x();
302 }
303
304 // interval beg + end value constraint
305 for (const auto & [i, dx] : utils::zip(iota(0u), dx_r)) {
306 for (auto j = 0; j < K + 1; ++j) { A.insert(M, i * (K + 1) + j) = U0tB(0, j); }
307 b(M++) = 0;
308 if (SS::InnCnt >= 0) {
309 for (auto j = 0; j < K + 1; ++j) { A.insert(M, i * (K + 1) + j) = U1tB(0, j); }
310 b(M++) = dx;
311 }
312 }
313
314 // inner derivative continuity constraint
315 for (const auto & [k, dt, dt_next] : utils::zip(iota(0u, N - 1), dt_r, dt_r | drop(1))) {
316 for (auto d = 1; d <= SS::InnCnt; ++d) {
317 const double fac1 = 1. / std::pow(dt, d);
318 const double fac2 = 1. / std::pow(dt_next, d);
319 for (auto j = 0; j < K + 1; ++j) {
320 A.insert(M, k * (K + 1) + j) = U1tB(d, j) * fac1;
321 A.insert(M, (k + 1) * (K + 1) + j) = -U0tB(d, j) * fac2;
322 }
323 b(M++) = 0;
324 }
325 }
326
327 // curve end derivative constraints
328 for (auto i = 0u; i < ss.RghtDeg.size(); ++i) {
329 for (auto j = 0u; j < K + 1; ++j) {
330 A.insert(M, (K + 1) * (N - 1) + j) = U1tB(ss.RghtDeg[i], j);
331 }
332 b(M++) = ss.rght_values[i].x();
333 }
334
335 A.prune(1e-9); // there are typically a lot of zeros (depends on basis)..
336 A.makeCompressed();
337
338 if constexpr (SS::OptDeg < 0) {
339 // No optimization, solve directly
340 assert(N_eq == N_coef);
341 Eigen::SparseLU<decltype(A)> lu(A);
342 return lu.solve(b);
343 } else {
344 static_assert(K >= SS::OptDeg, "K >= OptDeg");
345
346 // COST MATRIX P
347
348 // cost function is ∫ | p^{(D)} (t) |^2 dt, t : 0 -> T,
349 // or (1 / T)^{2D - 1} ∫ | p^{(D)} (u) |^2 du, u : 0 -> 1
350 //
351 // p(u) = u^{(D)}^T B x, so p^{(D)} (u)^2 = x' B' u^{(D)}' u^{(D)} B x
352 //
353 // Let M = \int_{0}^1 u^{(D)} u^{(D)}' du u : 0 -> 1, then the cost matrix P
354 // is (1 / T)^{2D - 1} * B' * M * B
355
356 static constexpr auto Mmat = monomial_integral<K, SS::OptDeg, double>();
357 static constexpr StaticMatrix<double, K + 1, K + 1> P_s = B_s.transpose() * Mmat * B_s;
358
359 Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> P(P_s[0].data());
360
361 // SOLVE QP
362
363 // We solve
364 // min_{x : Ax = b} (1/2) x' Q x
365 // by solving the KKT equations
366 // [Q A'; A 0] [x; l] = [0; b]
367 // via LDLt factorization
368
369 Eigen::SparseMatrix<double> H(N_coef + N_eq, N_coef + N_eq);
370
371 Eigen::Matrix<int, -1, 1> H_pattern(N_coef + N_eq);
372 for (auto i = 0u; i != N_coef; ++i) {
373 H_pattern(i) = (K + 1) + A.outerIndexPtr()[i + 1] - A.outerIndexPtr()[i];
374 }
375 H_pattern.tail(N_eq).setZero();
376
377 H.reserve(H_pattern);
378
379 for (const auto & [i, dt] : utils::zip(iota(0u), dt_r | take(int64_t(N)))) {
380 const double fac = std::pow(dt, 1 - 2 * int(D));
381 for (auto ki = 0u; ki != K + 1; ++ki) {
382 for (auto kj = 0u; kj != K + 1; ++kj) {
383 H.insert(i * (K + 1) + ki, i * (K + 1) + kj) = (ki == kj ? 1e-6 : 0.) + fac * P(ki, kj);
384 }
385 }
386 }
387
388 for (auto col = 0u; col != N_coef; ++col) {
389 for (typename decltype(A)::InnerIterator it(A, col); it; ++it) {
390 H.insert(N_coef + it.index(), col) = it.value();
391 }
392 }
393
394 H.makeCompressed();
395
396 Eigen::VectorXd rhs(N_coef + N_eq);
397 rhs.head(N_coef).setZero();
398 rhs.tail(N_eq) = b;
399
400 const Eigen::SimplicialLDLT<decltype(H), Eigen::Lower> ldlt(H);
401 return ldlt.solve(rhs).head(N_coef);
402 }
403}

417auto fit_spline(
418 std::ranges::random_access_range auto && ts,
419 std::ranges::random_access_range auto && gs,
420 const SplineSpec auto & ss)
421{
422 using namespace std::views;
423
424 using SS = std::decay_t<decltype(ss)>;
425 using G = PlainObject<std::ranges::range_value_t<std::decay_t<decltype(gs)>>>;
426
427 assert(std::ranges::adjacent_find(ts, std::ranges::greater_equal()) == ts.end());
428
429 static constexpr auto K = SS::Degree;
430 const auto N = std::min(std::ranges::size(ts), std::ranges::size(gs));
431
432 assert(N >= 2);
433
434 static constexpr auto sub = [](const auto & x1, const auto & x2) { return x2 - x1; };
435 static constexpr auto sub_lie = [](const auto & x1, const auto & x2) { return rminus(x2, x1); };
436
437 auto dts = ts | utils::views::pairwise_transform(sub);
438 auto dgs = gs | utils::views::pairwise_transform(sub_lie);
439
440 Eigen::Matrix<double, Dof<G>, -1> V(Dof<G>, (N - 1) * (K + 1));
441
442 for (auto k = 0u; k < Dof<G>; ++k) {
443 const auto ss_proj = detail::splinespec_project(ss, k);
444 V.row(k) = fit_spline_1d(dts, dgs | transform([k](const auto & v) { return v(k); }), ss_proj);
445 }
446
447 Spline<K, G> ret;
448 ret.reserve(N);
449
450 for (const auto & [i, dt, g, g_next] : utils::zip(iota(0u), dts, gs, gs | drop(1))) {
451 // spline is in cumulative form, need to get cumulative coefficients
452 Eigen::Matrix<double, Dof<G>, K> cum_coefs =
453 V.template block<Dof<G>, K>(0, i * (K + 1) + 1) - V.template block<Dof<G>, K>(0, i * (K + 1));
454
455 if constexpr (K > 2) {
456 // modify segment to ensure it is interpolating
457 // want exp(v1) * ... * exp(vK) = inv(g) * gnext
458 auto mid = K / 2;
459
460 G midval = composition<G>(::smooth::inverse<G>(g), g_next);
461 for (auto k = 0; k < mid; ++k) {
462 midval = composition<G>(::smooth::exp<G>(-cum_coefs.col(k)), midval);
463 }
464 for (auto k = K - 1; k > mid; --k) {
465 midval = composition<G>(midval, ::smooth::exp<G>(-cum_coefs.col(k)));
466 }
467 cum_coefs.col(mid) = ::smooth::log<G>(midval);
468 }
469
470 ret.concat_global(Spline<K, G>(dt, std::move(cum_coefs), g));
471 }
472
473 ret.concat_global(gs[N - 1]);
474
475 return ret;
476}

499template<std::size_t K, std::ranges::range Rs, std::ranges::range Rg>
500struct fit_bspline_objective
501{
503 using G = std::ranges::range_value_t<Rg>;
504
505 // \cond
506 Rs ts;
507 Rg gs;
508
509 double t0, t1, dt;
510
511 std::size_t NumData, NumPts;
512
513 static constexpr auto M_s = polynomial_cumulative_basis<PolynomialBasis::Bspline, K>();
514 inline static const Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> M =
515 Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>>(M_s[0].data());
516 // \endcond
517
519 fit_bspline_objective(
520 std::ranges::range auto && tsin, std::ranges::range auto && gsin, double dtin)
521 : ts(std::forward<decltype(tsin)>(tsin)), gs(std::forward<decltype(gsin)>(gsin)), dt(dtin)
522 {
523 const auto [rt0, rt1] = std::ranges::minmax(ts);
524
525 t0 = rt0;
526 t1 = rt1;
527
528 NumData = std::min(std::ranges::size(ts), std::ranges::size(gs));
529 NumPts = K + static_cast<std::size_t>((t1 - t0 + dt) / dt);
530 }
531
533 Eigen::VectorXd operator()(const ManifoldVector<G> & var) const
534 {
535 using namespace std::views;
536
537 Eigen::VectorXd ret(Dof<G> * NumData);
538
539 for (const auto & [i, t, g] : utils::zip(iota(0u), ts, gs)) {
540 const int64_t istar = static_cast<int64_t>((t - t0) / dt);
541 const double u = (t - t0 - istar * dt) / dt;
542
543 // gcc 11.1 bug can't handle uint64_t
544 const auto g_spline = cspline_eval<K>(var | drop(istar) | take(int64_t(K + 1)), M, u);
545
546 ret.segment<Dof<G>>(i * Dof<G>) = rminus(g_spline, g);
547 }
548
549 return ret;
550 }

553 Eigen::SparseMatrix<double> jacobian(const ManifoldVector<G> & var) const
554 {
555 using namespace std::views;
556
557 Eigen::SparseMatrix<double, Eigen::RowMajor> Jac;
558 Jac.resize(Dof<G> * NumData, Dof<G> * NumPts);
559 Jac.reserve(Eigen::Matrix<int, -1, 1>::Constant(Dof<G> * NumData, Dof<G> * (K + 1)));
560
561 for (const auto & [i, t, g] : utils::zip(iota(0u), ts, gs)) {
562 const int64_t istar = static_cast<int64_t>((t - t0) / dt);
563 const double u = (t - t0 - istar * dt) / dt;
564
565 Eigen::Matrix<double, Dof<G>, (K + 1) * Dof<G>> d_vali_pts;
566 // gcc 11.1 bug can't handle uint64_t
567 auto g_spline =
568 cspline_eval<K>(var | drop(istar) | take(int64_t(K + 1)), M, u, {}, {}, d_vali_pts);
569
570 const Tangent<G> resi = rminus(g_spline, g);
571
572 const Eigen::Matrix<double, Dof<G>, Dof<G>> d_resi_vali = dr_expinv<G>(resi);
573 const Eigen::Matrix<double, Dof<G>, (K + 1) * Dof<G>> d_resi_pts = d_resi_vali * d_vali_pts;
574
575 for (auto r = 0u; r != Dof<G>; ++r) {
576 for (auto c = 0u; c != Dof<G> * (K + 1); ++c) {
577 Jac.insert(i * Dof<G> + r, istar * Dof<G> + c) = d_resi_pts(r, c);
578 }
579 }
580 }
581
582 Jac.makeCompressed();
583
584 return Jac;
585 }

603template<std::size_t K>
604auto fit_bspline(std::ranges::range auto && ts, std::ranges::range auto && gs, const double dt)
605{
606 using namespace std::views;
607 using G = PlainObject<std::ranges::range_value_t<std::decay_t<decltype(gs)>>>;
608
609 assert(std::ranges::adjacent_find(ts, std::ranges::greater_equal()) == ts.end());
610
611 using obj_t = fit_bspline_objective<K, decltype(ts), decltype(gs)>;
612
613 obj_t obj(std::forward<decltype(ts)>(ts), std::forward<decltype(gs)>(gs), dt);
614
615 // create optimization variable
616 ManifoldVector<G> ctrl_pts(obj.NumPts);
617
618 // create initial guess
619 auto t_iter = std::ranges::begin(ts);
620 auto g_iter = std::ranges::begin(gs);
621 for (auto i = 0u; i != obj.NumPts; ++i) {
622 const double t_target = obj.t0 + (i - static_cast<double>(K - 1) / 2) * dt;
623 while (t_iter + 1 < std::ranges::end(ts)
624 && std::abs(t_target - *(t_iter + 1)) < std::abs(t_target - *t_iter)) {
625 ++t_iter;
626 ++g_iter;
627 }
628 ctrl_pts[i] = *g_iter;
629 }
630
631 // fit to data with loose convergence criteria
632 const MinimizeOptions opts{
633 .ptol = 1e-3,
634 .ftol = 1e-3,
635 .max_iter = 10,
636 .verbose = false,
637 };
638 minimize<diff::Type::Analytic>(obj, smooth::wrt(ctrl_pts), opts);
639
640 return BSpline<K, G>(obj.t0, dt, std::move(ctrl_pts));
641}
