88 template<LieGroup G, std::size_t Terms = std::tuple_size_v<decltype(detail::kBn)>>
89  requires(Terms <= std::tuple_size_v<decltype(detail::kBn)>)
90 detail::hess_t<G> d2r_exp(const Tangent<G> & a)
91 {
92  const TangentMap<G> ad_a_t = ad<G>(a).transpose();
93 
94  detail::hess_t<G> res;
95  res.setZero();
96 
97  for (auto j = 0u; j < Dof<G>; ++j) {
98  double coeff = 1; // hold (-1)^i / i!
99  Tangent<G> gi = Tangent<G>::Unit(j); // (ad_a^T)^i * e_j
100  TangentMap<G> dgi = TangentMap<G>::Zero(); // right derivative of gi w.r.t. a
101 
102  for (auto iter = 0u; iter < Terms; ++iter) {
103  res.template block<Dof<G>, Dof<G>>(0, j * Dof<G>) += coeff * dgi;
104  dgi.applyOnTheLeft(ad_a_t);
105  for (auto i = 0u; i < Dof<G>; ++i) {
106  dgi.row(i).noalias() -= gi.transpose() * detail::algebra_generators<G>[i];
107  }
108  gi.applyOnTheLeft(ad_a_t);
109  coeff *= (-1.) / (iter + 2);
110  }
111  }
112 
113  return res;
114 }
