51 template<typename G>
52 concept LieGroup =
53 requires {
54  // Underlying scalar type
55  typename lie<G>::Scalar;
56  // Default representation
57  typename lie<G>::PlainObject;
58  // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
59  {lie<G>::Dof}->std::convertible_to<Eigen::Index>;
60  // Return the identity element
61  {lie<G>::Identity()}->std::convertible_to<typename lie<G>::PlainObject>;
62  // Return a random element
63  {lie<G>::Random()}->std::convertible_to<typename lie<G>::PlainObject>;
64 } &&
65 // GROUP INTERFACE
66 requires(const G & g1, const G & g2, typename lie<G>::Scalar eps) {
67  // Group adjoint
68  {lie<G>::Ad(g1)}->std::convertible_to<Eigen::Matrix<typename lie<G>::Scalar, lie<G>::Dof, lie<G>::Dof>>;
69  // Group composition
70  {lie<G>::composition(g1, g2)}->std::convertible_to<typename lie<G>::PlainObject>;
71  // Run-time degrees of freedom (tangent space dimension).
72  {lie<G>::dof(g1)}->std::convertible_to<Eigen::Index>;
73  // Group inverse
74  {lie<G>::inverse(g1)}->std::convertible_to<typename lie<G>::PlainObject>;
75  // Check if two elements are (approximately) equal
76  {lie<G>::isApprox(g1, g2, eps)}->std::convertible_to<bool>;
77  // Group logarithm (maps from group to algebra)
78  {lie<G>::log(g1)}->std::convertible_to<Eigen::Matrix<typename lie<G>::Scalar, lie<G>::Dof, 1>>;
79 } &&
80 // TANGENT INTERFACE
81 requires(const Eigen::Matrix<typename lie<G>::Scalar, lie<G>::Dof, 1> & a) {
82  // Algebra adjoint
83  {lie<G>::ad(a)}->std::convertible_to<Eigen::Matrix<typename lie<G>::Scalar, lie<G>::Dof, lie<G>::Dof>>;
84  // Algebra exponential (maps from algebra to group)
85  {lie<G>::exp(a)}->std::convertible_to<typename lie<G>::PlainObject>;
86  // Right derivative of the exponential map
87  {lie<G>::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename lie<G>::Scalar, lie<G>::Dof, lie<G>::Dof>>;
88  // Right derivative of the exponential map inverse
89  {lie<G>::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename lie<G>::Scalar, lie<G>::Dof, lie<G>::Dof>>;
90 } && (
91  // Cast to different scalar type
92  !std::is_convertible_v<typename lie<G>::Scalar, double> ||
93  requires (const G & g) {
94  {lie<G>::template cast<double>(g)};
95  }
96 ) && (
97  !std::is_convertible_v<typename lie<G>::Scalar, float> ||
98  requires (const G & g) {
99  {lie<G>::template cast<double>(g)};
100  }
101 ) &&
102 // PlainObject must be default-constructible
103 std::is_default_constructible_v<typename lie<G>::PlainObject> &&
104 std::is_copy_constructible_v<typename lie<G>::PlainObject> &&
105 // PlainObject must be assignable from G
106 std::is_assignable_v<typename lie<G>::PlainObject &, G>;

290 template<typename G>
291 concept NativeLieGroup = requires
292 {
293  typename G::Scalar;
294  typename G::Tangent;
295  typename G::PlainObject;
296  {G::Dof}->std::convertible_to<Eigen::Index>;
297  {G::Identity()}->std::convertible_to<typename G::PlainObject>;
298  {G::Random()}->std::convertible_to<typename G::PlainObject>;
299 } &&
300 (G::Dof >= 1) &&
301 (G::Tangent::SizeAtCompileTime == G::Dof) &&
302 requires(const G & g1, const G & g2, typename G::Scalar eps) {
303  {g1.dof()}->std::convertible_to<Eigen::Index>; // degrees of freedom at runtime
304  {g1.Ad()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
305  {g1 * g2}->std::convertible_to<typename G::PlainObject>;
306  {g1.inverse()}->std::convertible_to<typename G::PlainObject>;
307  {g1.isApprox(g2, eps)}->std::convertible_to<bool>;
308  {g1.log()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, 1>>;
309 } &&
310 requires(const Eigen::Matrix<typename G::Scalar, G::Dof, 1> & a) {
311  {G::ad(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
312  {G::exp(a)}->std::convertible_to<typename G::PlainObject>;
313  {G::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
314  {G::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
315 };
