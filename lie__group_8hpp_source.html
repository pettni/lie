32template<typename G>
33concept LieGroup =
34requires (Eigen::Index dof) {
35 // Underlying scalar type
36 typename traits::lie<G>::Scalar;
37 // Default representation
38 typename traits::lie<G>::PlainObject;
39 // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
40 {traits::lie<G>::Dof}->std::convertible_to<int>;
41 // Commutativity
42 {traits::lie<G>::IsCommutative}->std::convertible_to<bool>;
43 // Return the identity element (dof = Dof for static size)
44 {traits::lie<G>::Identity(dof)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
45 // Return a random element (dof = Dof for static size)
46 {traits::lie<G>::Random(dof)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
47} &&
48// GROUP INTERFACE
49requires(const G & g1, const G & g2, typename traits::lie<G>::Scalar eps) {
50 // Group adjoint
51 {traits::lie<G>::Ad(g1)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
52 // Group composition
53 {traits::lie<G>::composition(g1, g2)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
54 // Run-time degrees of freedom (tangent space dimension).
55 {traits::lie<G>::dof(g1)}->std::convertible_to<Eigen::Index>;
56 // Group inverse
57 {traits::lie<G>::inverse(g1)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
58 // Check if two elements are (approximately) equal
59 {traits::lie<G>::isApprox(g1, g2, eps)}->std::convertible_to<bool>;
60 // Group logarithm (maps from group to algebra)
61 {traits::lie<G>::log(g1)}->std::convertible_to<Eigen::Vector<typename traits::lie<G>::Scalar, traits::lie<G>::Dof>>;
62} &&
63// TANGENT INTERFACE
64requires(const Eigen::Vector<typename traits::lie<G>::Scalar, traits::lie<G>::Dof> & a) {
65 // Algebra adjoint
66 {traits::lie<G>::ad(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
67 // Algebra exponential (maps from algebra to group)
68 {traits::lie<G>::exp(a)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
69 // Right derivative of the exponential map
70 {traits::lie<G>::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
71 // Right derivative of the exponential map inverse
72 {traits::lie<G>::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
73 // Second right derivative of the exponential map
74 {traits::lie<G>::d2r_exp(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, (traits::lie<G>::Dof > 0 ? traits::lie<G>::Dof * traits::lie<G>::Dof : -1)>>;
75 // Second right derivative of the exponential map inverse
76 {traits::lie<G>::d2r_expinv(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, (traits::lie<G>::Dof > 0 ? traits::lie<G>::Dof * traits::lie<G>::Dof : -1)>>;
77} && (
78 // Cast to different scalar type
79 !std::is_convertible_v<typename traits::lie<G>::Scalar, double> ||
80 requires (const G & g) {
81 {traits::lie<G>::template cast<double>(g)};
82 }
83) && (
84 !std::is_convertible_v<typename traits::lie<G>::Scalar, float> ||
85 requires (const G & g) {
86 {traits::lie<G>::template cast<double>(g)};
87 }
88) &&
89// PlainObject must be default-constructible
90std::is_default_constructible_v<typename traits::lie<G>::PlainObject> &&
91std::is_copy_constructible_v<typename traits::lie<G>::PlainObject> &&
92// PlainObject must be assignable from G
93std::is_assignable_v<typename traits::lie<G>::PlainObject &, G>;

104template<typename G>
105concept NativeLieGroup = requires
106{
107 typename G::Scalar;
108 typename G::Tangent;
109 typename G::PlainObject;
110 {G::Dof}->std::convertible_to<int>;
111 {G::IsCommutative}->std::convertible_to<bool>;
112} &&
113(!(G::Dof > 0) || requires {
114 {G::Identity()}->std::convertible_to<typename G::PlainObject>;
115 {G::Random()}->std::convertible_to<typename G::PlainObject>;
116}) &&
117(!(G::Dof == -1) || requires (Eigen::Index dof) {
118 {G::Identity(dof)}->std::convertible_to<typename G::PlainObject>;
119 {G::Random(dof)}->std::convertible_to<typename G::PlainObject>;
120}) &&
121(G::Tangent::SizeAtCompileTime == G::Dof) &&
122requires(const G & g1, const G & g2, typename G::Scalar eps) {
123 {g1.dof()}->std::convertible_to<Eigen::Index>; // degrees of freedom at runtime
124 {g1.Ad()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
125 {g1 * g2}->std::convertible_to<typename G::PlainObject>;
126 {g1.inverse()}->std::convertible_to<typename G::PlainObject>;
127 {g1.isApprox(g2, eps)}->std::convertible_to<bool>;
128 {g1.log()}->std::convertible_to<Eigen::Vector<typename G::Scalar, G::Dof>>;
129} &&
130requires(const Eigen::Vector<typename G::Scalar, G::Dof> & a) {
131 {G::ad(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
132 {G::exp(a)}->std::convertible_to<typename G::PlainObject>;
133 {G::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
134 {G::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
135 {G::d2r_exp(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof * G::Dof>>;
136 {G::d2r_expinv(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof * G::Dof>>;
137};
