56 template<typename G>
57 concept LieGroup =
58 requires (Eigen::Index dof) {
59  // Underlying scalar type
60  typename traits::lie<G>::Scalar;
61  // Default representation
62  typename traits::lie<G>::PlainObject;
63  // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
64  {traits::lie<G>::Dof}->std::convertible_to<Eigen::Index>;
65  // Return the identity element (dof = Dof for static size)
66  {traits::lie<G>::Identity(dof)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
67  // Return a random element (dof = Dof for static size)
68  {traits::lie<G>::Random(dof)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
69 } &&
70 // GROUP INTERFACE
71 requires(const G & g1, const G & g2, typename traits::lie<G>::Scalar eps) {
72  // Group adjoint
73  {traits::lie<G>::Ad(g1)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
74  // Group composition
75  {traits::lie<G>::composition(g1, g2)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
76  // Run-time degrees of freedom (tangent space dimension).
77  {traits::lie<G>::dof(g1)}->std::convertible_to<Eigen::Index>;
78  // Group inverse
79  {traits::lie<G>::inverse(g1)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
80  // Check if two elements are (approximately) equal
81  {traits::lie<G>::isApprox(g1, g2, eps)}->std::convertible_to<bool>;
82  // Group logarithm (maps from group to algebra)
83  {traits::lie<G>::log(g1)}->std::convertible_to<Eigen::Vector<typename traits::lie<G>::Scalar, traits::lie<G>::Dof>>;
84 } &&
85 // TANGENT INTERFACE
86 requires(const Eigen::Vector<typename traits::lie<G>::Scalar, traits::lie<G>::Dof> & a) {
87  // Algebra adjoint
88  {traits::lie<G>::ad(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
89  // Algebra exponential (maps from algebra to group)
90  {traits::lie<G>::exp(a)}->std::convertible_to<typename traits::lie<G>::PlainObject>;
91  // Right derivative of the exponential map
92  {traits::lie<G>::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
93  // Right derivative of the exponential map inverse
94  {traits::lie<G>::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename traits::lie<G>::Scalar, traits::lie<G>::Dof, traits::lie<G>::Dof>>;
95 } && (
96  // Cast to different scalar type
97  !std::is_convertible_v<typename traits::lie<G>::Scalar, double> ||
98  requires (const G & g) {
99  {traits::lie<G>::template cast<double>(g)};
100  }
101 ) && (
102  !std::is_convertible_v<typename traits::lie<G>::Scalar, float> ||
103  requires (const G & g) {
104  {traits::lie<G>::template cast<double>(g)};
105  }
106 ) &&
107 // PlainObject must be default-constructible
108 std::is_default_constructible_v<typename traits::lie<G>::PlainObject> &&
109 std::is_copy_constructible_v<typename traits::lie<G>::PlainObject> &&
110 // PlainObject must be assignable from G
111 std::is_assignable_v<typename traits::lie<G>::PlainObject &, G>;

122 template<typename G>
123 concept NativeLieGroup = requires
124 {
125  typename G::Scalar;
126  typename G::Tangent;
127  typename G::PlainObject;
128  {G::Dof}->std::convertible_to<Eigen::Index>;
129 } &&
130 (!(G::Dof > 0) || requires {
131  {G::Identity()}->std::convertible_to<typename G::PlainObject>;
132  {G::Random()}->std::convertible_to<typename G::PlainObject>;
133 }) &&
134 (!(G::Dof == -1) || requires (Eigen::Index dof) {
135  {G::Identity(dof)}->std::convertible_to<typename G::PlainObject>;
136  {G::Random(dof)}->std::convertible_to<typename G::PlainObject>;
137 }) &&
138 (G::Tangent::SizeAtCompileTime == G::Dof) &&
139 requires(const G & g1, const G & g2, typename G::Scalar eps) {
140  {g1.dof()}->std::convertible_to<Eigen::Index>; // degrees of freedom at runtime
141  {g1.Ad()}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
142  {g1 * g2}->std::convertible_to<typename G::PlainObject>;
143  {g1.inverse()}->std::convertible_to<typename G::PlainObject>;
144  {g1.isApprox(g2, eps)}->std::convertible_to<bool>;
145  {g1.log()}->std::convertible_to<Eigen::Vector<typename G::Scalar, G::Dof>>;
146 } &&
147 requires(const Eigen::Vector<typename G::Scalar, G::Dof> & a) {
148  {G::ad(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
149  {G::exp(a)}->std::convertible_to<typename G::PlainObject>;
150  {G::dr_exp(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
151  {G::dr_expinv(a)}->std::convertible_to<Eigen::Matrix<typename G::Scalar, G::Dof, G::Dof>>;
152 };
