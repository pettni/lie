80template<LieGroup G>
81inline std::array<Eigen::SparseMatrix<Scalar<G>>, Dof<G>> generators_sparse =
82 []() -> std::array<Eigen::SparseMatrix<Scalar<G>>, Dof<G>> {
83 std::array<Eigen::SparseMatrix<Scalar<G>>, Dof<G>> ret;
84 for (auto i = 0u; i < Dof<G>; ++i) {
85 ret[i] = ad<G>(Tangent<G>::Unit(i)).sparseView();
86 ret[i].makeCompressed();
87 }
88 return ret;
89}();

94template<LieGroup G>
95inline Eigen::SparseMatrix<Scalar<G>> ad_sparse_pattern = [] {
96 Eigen::SparseMatrix<Scalar<G>> ret(Dof<G>, Dof<G>);
97 if constexpr (!IsCommutative<G>) {
98 for (const auto & gen : generators_sparse<G>) { ret += gen; }
99 }
100 ret.makeCompressed();
101 ret.coeffs().setZero();
102 return ret;
103}();

115template<LieGroup G>
116inline void ad_sparse(Eigen::SparseMatrix<Scalar<G>> & sp, const Tangent<G> & a)
117{
118 assert(sp.isCompressed());
119 sp.coeffs().setZero();
120 for (auto k = 0u; k < Dof<G>; ++k) { sp += a(k) * generators_sparse<G>[k]; }
121 assert(sp.isCompressed());
122}

127template<LieGroup G>
128inline Eigen::SparseMatrix<Scalar<G>> d_exp_sparse_pattern = [] {
129 Eigen::SparseMatrix<Scalar<G>> ret(Dof<G>, Dof<G>);
130 if constexpr (IsCommutative<G>) {
131 // identity matrix
132 for (auto i = 0u; i < Dof<G>; ++i) { ret.insert(i, i) = Scalar<G>(1); }
133 } else if constexpr (requires { traits::lie_sparse<G>::d_exp_sparse_pattern; }) {
134 // use specialized method if one exists
135 ret = traits::lie_sparse<G>::d_exp_sparse_pattern;
136 } else {
137 // fall back on dense pattern
138 for (auto i = 0u; i < Dof<G>; ++i) {
139 for (auto j = 0u; j < Dof<G>; ++j) {
140 ret.insert(i, j) = (i == j) ? Scalar<G>(1) : Scalar<G>(0);
141 }
142 }
143 }
144 ret.makeCompressed();
145 return ret;
146}();

162template<LieGroup G, bool Inv = false>
163inline void
164dr_exp_sparse(Eigen::SparseMatrix<Scalar<G>> & sp, const Tangent<G> & a, Eigen::Index i0 = 0)
165{
166 using T = traits::lie_sparse<G>;
167 assert(sp.isCompressed());
168 assert(sp.rows() >= i0 + Dof<G>);
169 assert(sp.cols() >= i0 + Dof<G>);
170 if constexpr (IsCommutative<G>) {
171 // identity matrix
172 for (auto i = 0u; i < a.size(); ++i) { sp.coeffRef(i0 + i, i0 + i) = 1; }
173 } else if constexpr (!Inv && requires { T::dr_exp_sparse(sp, a, i0); }) {
174 // use specialized method if one exists
175 T::dr_exp_sparse(sp, a, i0);
176 } else if constexpr (Inv && requires { T::dr_expinv_sparse(sp, a, i0); }) {
177 // use specialized method if one exists
178 T::dr_expinv_sparse(sp, a, i0);
179 } else {
180 // fall back on dense method+pattern
181 const TangentMap<G> D = [&] {
182 if constexpr (Inv)
183 return dr_expinv<G>(a);
184 else
185 return dr_exp<G>(a);
186 }();
187 for (auto i = 0u; i < Dof<G>; ++i) {
188 for (Eigen::InnerIterator it(d_exp_sparse_pattern<G>, i); it; ++it) {
189 sp.coeffRef(i0 + it.row(), i0 + it.col()) = D(it.row(), it.col());
190 }
191 }
192 }
193 assert(sp.isCompressed());
194}

249template<LieGroup G, bool Inv = false>
250inline void
251d2r_exp_sparse(Eigen::SparseMatrix<Scalar<G>> & sp, const Tangent<G> & a, Eigen::Index i0 = 0)
252{
253 using T = traits::lie_sparse<G>;
254 assert(sp.isCompressed());
255 assert(sp.rows() >= i0 + Dof<G>);
256 assert(sp.cols() >= sp.rows() * (i0 + Dof<G>));
257 if constexpr (IsCommutative<G>) {
258 // zero matrix--do nothing
259 } else if constexpr (!Inv && requires { T::d2r_exp_sparse(sp, a, i0); }) {
260 // use specialized method if one exists
261 T::d2r_exp_sparse(sp, a, i0);
262 } else if constexpr (Inv && requires { T::d2r_expinv_sparse(sp, a, i0); }) {
263 // use specialized method if one exists
264 T::d2r_expinv_sparse(sp, a, i0);
265 } else {
266 // fall back on dense method + pattern
267 const Hessian<G> D = [&] {
268 if constexpr (Inv)
269 return d2r_expinv<G>(a);
270 else
271 return d2r_exp<G>(a);
272 }();
273 for (auto i = 0u; i < Dof<G> * Dof<G>; ++i) {
274 for (Eigen::InnerIterator it(d2_exp_sparse_pattern<G>, i); it; ++it) {
275 const auto block = i0 + (it.col() / Dof<G>);
276 const auto row = i0 + it.row();
277 const auto col = i0 + (it.col() % Dof<G>);
278
279 sp.coeffRef(row, sp.rows() * block + col) = D(it.row(), it.col());
280 }
281 }
282 }
283 assert(sp.isCompressed());
284}
