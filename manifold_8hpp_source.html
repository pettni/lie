49 template<typename M>
50 concept Manifold =
51 requires {
52  // Underlying scalar type
53  typename man<M>::Scalar;
54  // Default representation
55  typename man<M>::PlainObject;
56  // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
57  {man<M>::Dof}->std::convertible_to<Eigen::Index>;
58 } &&
59 requires(const M & m1, const M & m2, const Eigen::Matrix<typename man<M>::Scalar, man<M>::Dof, 1> & a) {
60  // Run-time degrees of freedom (tangent space dimension)
61  {man<M>::dof(m1)}->std::convertible_to<Eigen::Index>;
62  // Right-plus: add a tangent vector to a Manifold element to obtain a new Manifold element
63  {man<M>::rplus(m1, a)}->std::convertible_to<typename man<M>::PlainObject>;
64  // Right-minus: subtract a Manifold element from another to obtain the difference tangent vector
65  {man<M>::rminus(m1, m2)}->std::convertible_to<Eigen::Matrix<typename man<M>::Scalar, man<M>::Dof, 1>>;
66 } && (
67  !std::is_convertible_v<typename man<M>::Scalar, double> ||
68  requires (const M & m) {
69  {man<M>::template cast<double>(m)}->std::convertible_to<typename man<M>::template CastT<double>>;
70  }
71 ) && (
72  !std::is_convertible_v<typename man<M>::Scalar, float> ||
73  requires (const M & m) {
74  {man<M>::template cast<float>(m)}->std::convertible_to<typename man<M>::template CastT<float>>;
75  }
76 ) &&
77 // PlainObject must be default-constructible
78 std::is_default_constructible_v<typename man<M>::PlainObject> &&
79 std::is_copy_constructible_v<typename man<M>::PlainObject> &&
80 // PlainObject must be assignable from M
81 std::is_assignable_v<typename man<M>::PlainObject &, M>;
