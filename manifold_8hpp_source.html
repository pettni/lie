54template<typename M>
55concept Manifold =
56requires (Eigen::Index dof) {
57 // Underlying scalar type
58 typename traits::man<M>::Scalar;
59 // Default representation
60 typename traits::man<M>::PlainObject;
61 // Compile-time degrees of freedom (tangent space dimension). Can be dynamic (equal to -1)
62 {traits::man<M>::Dof}->std::convertible_to<Eigen::Index>;
63 // A default-initialized Manifold object (if Dof > 0 then dof = Dof can be assumed)
64 {traits::man<M>::Default(dof)}->std::convertible_to<typename traits::man<M>::PlainObject>;
65} &&
66requires(const M & m1, const M & m2, const Eigen::Vector<typename traits::man<M>::Scalar, traits::man<M>::Dof> & a) {
67 // Run-time degrees of freedom (tangent space dimension)
68 {traits::man<M>::dof(m1)}->std::convertible_to<Eigen::Index>;
69 // Right-plus: add a tangent vector to a Manifold element to obtain a new Manifold element
70 {traits::man<M>::rplus(m1, a)}->std::convertible_to<typename traits::man<M>::PlainObject>;
71 // Right-minus: subtract a Manifold element from another to obtain the difference tangent vector
72 {traits::man<M>::rminus(m1, m2)}->std::convertible_to<Eigen::Vector<typename traits::man<M>::Scalar, traits::man<M>::Dof>>;
73} && (
74 !std::is_convertible_v<typename traits::man<M>::Scalar, double> ||
75 requires (const M & m) {
76 {traits::man<M>::template cast<double>(m)}->std::convertible_to<typename traits::man<M>::template CastT<double>>;
77 }
78) && (
79 !std::is_convertible_v<typename traits::man<M>::Scalar, float> ||
80 requires (const M & m) {
81 {traits::man<M>::template cast<float>(m)}->std::convertible_to<typename traits::man<M>::template CastT<float>>;
82 }
83) &&
84// PlainObject must be default-constructible
85std::is_default_constructible_v<typename traits::man<M>::PlainObject> &&
86std::is_copy_constructible_v<typename traits::man<M>::PlainObject> &&
87// PlainObject must be assignable from M
88std::is_assignable_v<typename traits::man<M>::PlainObject &, M>;

103template<RnType M>
104struct man<M>
105{
106 // \cond
107 static constexpr int Dof = M::SizeAtCompileTime;
108
109 using Scalar = typename M::Scalar;
110 using PlainObject = Eigen::Vector<Scalar, Dof>;
111 template<typename NewScalar>
112 using CastT = Eigen::Vector<NewScalar, M::SizeAtCompileTime>;
113
114 static inline PlainObject Default(Eigen::Index dof) { return M::Zero(dof); }
115
116 static inline Eigen::Index dof(const M & m) { return m.size(); }
117
118 template<typename NewScalar>
119 static inline CastT<NewScalar> cast(const M & m)
120 {
121 return m.template cast<NewScalar>();
122 }
123
124 template<typename Derived>
125 static inline PlainObject rplus(const M & g, const Eigen::MatrixBase<Derived> & a)
126 {

130 template<typename Derived>
131 static inline Eigen::Vector<Scalar, M::SizeAtCompileTime>
