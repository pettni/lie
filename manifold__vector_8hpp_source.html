95 Eigen::Index dof() const
96 {
97 if constexpr (Dof < M >> 0) {
98 return size() * Dof<M>;
99 } else {
100 return std::accumulate(this->begin(), this->end(), 0u, [](auto s, const auto & item) {
101 return s + ::smooth::dof<M>(item);
102 });
103 }

111 template<typename Derived>
112 ManifoldVector<M> & operator+=(const Eigen::MatrixBase<Derived> & a)
113 {
114 Eigen::Index dof_cntr = 0;
115 for (auto i = 0u; i != this->size(); ++i) {
116 const auto dof_i = ::smooth::dof<M>(this->operator[](i));
117 this->operator[](i) =
118 rplus<M>(this->operator[](i), a.template segment<Dof<M>>(dof_cntr, dof_i));
119 dof_cntr += dof_i;
120 }
121 return *this;
122 }
123
129 template<typename Derived>
130 ManifoldVector<M> operator+(const Eigen::MatrixBase<Derived> & a) const
131 {
132 ManifoldVector<M> ret = *this;
133 ret += a;
134 return ret;
135 }
136
142 Eigen::VectorX<Scalar<M>> operator-(const ManifoldVector<M> & o) const
143 {
144 std::size_t dof_cnts = 0;
145 if (Dof < M >> 0) {
146 dof_cnts = Dof<M> * size();
147 } else {
148 for (auto i = 0u; i != size(); ++i) { dof_cnts += ::smooth::dof<M>(this->operator[](i)); }
149 }
150
151 Eigen::VectorX<Scalar<M>> ret(dof_cnts);
152 Eigen::Index idx = 0;
153 for (auto i = 0u; i != size(); ++i) {
154 const auto & size_i = ::smooth::dof<M>(this->operator[](i));
155 ret.template segment<Dof<M>>(idx, size_i) = rminus<M>(this->operator[](i), o[i]);
156 idx += size_i;
157 }
158
159 return ret;
160 }
161};

166template<Manifold M>

175 static constexpr Eigen::Index Dof = -1;

177 static inline Eigen::Index dof(const ManifoldVector<M> & m) { return m.dof(); }

179 static inline PlainObject Default(Eigen::Index dof)
180 {
182 const Eigen::Index mdof = ::smooth::Dof<M> != -1 ? ::smooth::Dof<M> : 1;
183 const Eigen::Index size = dof / mdof;

185 return PlainObject(size, Default<M>(mdof));
186 }
