125 Eigen::VectorX<Scalar<M>> operator-(const ManifoldVector<M> & o) const
126 {
127 std::size_t dof_cnts = 0;
128 if (Dof < M >> 0) {
129 dof_cnts = Dof<M> * size();

131 for (auto i = 0u; i != size(); ++i) { dof_cnts += ::smooth::dof<M>(this->operator[](i)); }
132 }
133
134 Eigen::VectorX<Scalar<M>> ret(dof_cnts);
135 Eigen::Index idx = 0;
136 for (auto i = 0u; i != size(); ++i) {
137 const auto & size_i = ::smooth::dof<M>(this->operator[](i));
138 ret.template segment<Dof<M>>(idx, size_i) = rminus<M>(this->operator[](i), o[i]);
139 idx += size_i;
140 }
