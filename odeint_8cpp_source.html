53 int main(int, char const **)
54 {
55  using state_t = smooth::Bundle<smooth::SO3d, Eigen::Vector3d>;
56  using deriv_t = typename state_t::Tangent;
57 
58  std::srand(5);
59 
60  // equilibrium point
61  const smooth::SO3d Xdes = smooth::SO3d::Identity();
62 
63  // "control" proportional and derivative gains
64  constexpr double kp = 1;
65  constexpr double kd = 1;
66 
67  auto ode = [&](const state_t & state, deriv_t & deriv, double) {
68  deriv.template head<3>() = state.part<1>();
69  deriv.template tail<3>() = -kp * (state.part<0>() - Xdes) - kd * state.part<1>();
70  };
71 
72  state_t state;
73  state.part<0>() = smooth::SO3d(Eigen::Quaterniond(0, 0.8, 0, 0.1));
74  state.part<1>() = Eigen::Vector3d(0, 0, 2);
75 
76  std::vector<double> tvec;
77  std::vector<state_t> gvec;
78 
79  auto stepper = boost::numeric::odeint::
80  runge_kutta4<state_t, double, deriv_t, double, boost::numeric::odeint::vector_space_algebra>();
81 
82  boost::numeric::odeint::integrate_const(
83  stepper, ode, state, 0., 10., 0.01, [&tvec, &gvec](const state_t & s, double t) {
84  tvec.push_back(t);
85  gvec.push_back(s);
86  });

88 #ifdef ENABLE_PLOTTING

100  }
101  // plot the trajectory
