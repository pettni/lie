49template<std::size_t K>
50constexpr auto pow_s(const auto x) noexcept
51{
52 const auto f = [&x]<auto... Is>(std::index_sequence<Is...>)
53 {
54 return ((static_cast<void>(Is), x) * ... * 1.);
55 };
56 return f(std::make_index_sequence<K>{});
57}

60template<std::size_t K>
61constexpr auto factorial_s() noexcept
62{
63 const auto f = []<auto... Is>(std::index_sequence<Is...>) { return ((Is + 1) * ... * 1.); };
64 return f(std::make_index_sequence<K>{});
65}

68constexpr auto cos_s(const auto x) noexcept
69{
70 const auto f = [&x]<auto... Is>(std::index_sequence<Is...>)
71 {
72 return ((pow_s<Is>(-1.) * pow_s<2 * Is>(x) / factorial_s<2 * Is>()) + ...);
73 };
74 return f(std::make_index_sequence<8>{});
75}

85template<std::size_t K>
86constexpr std::array<double, K> cgr_nodes()
87{
88 std::array<double, K> x;
89 for (auto i = 0u; i < K; ++i) { x[i] = -detail::cos_s(2. * pi_v<double> * i / (2 * K - 1)); }
90 return x;
91}

114template<std::size_t K, std::size_t I = 8>
115 requires(K <= 40)
116constexpr std::pair<std::array<double, K>, std::array<double, K>> lgr_nodes()
117{
118 // initial guess
119 auto xs = cgr_nodes<K>();
120
121 // initialize weights
122 std::array<double, K> ws;
123 ws[0] = 2. / (K * K);
124
125 // two rightmost column of legendre basis matrix
126 constexpr auto B =
127 polynomial_basis<PolynomialBasis::Legendre, K>().template block<K + 1, 2>(0, K - 1);
128
129 for (auto i = 1u; i < K; ++i) {
130 StaticMatrix<double, 1, 2> U_B;
131 for (auto iter = 0u; iter < I; ++iter) {
132 U_B = monomial_derivative<K>(xs[i], 0) * B;
133 const auto dU_B = monomial_derivative<K>(xs[i], 1) * B;
134 const double fxi = U_B[0][0] + U_B[0][1];
135 const double dfxi = dU_B[0][0] + dU_B[0][1];
136 xs[i] -= fxi / dfxi;
137 }
138 ws[i] = (1. - xs[i]) / (K * K * U_B[0][0] * U_B[0][0]);
139 }
140
141 return {xs, ws};
142}
