58 Spline<2, double> reparameterize_spline(
59  const SplineType auto & spline,
60  const auto & vel_min,
61  const auto & vel_max,
62  const auto & acc_min,
63  const auto & acc_max,
64  double start_vel = 1,
65  double end_vel = std::numeric_limits<double>::infinity(),
66  const std::size_t N = 100)
67 {
68  using G = std::invoke_result_t<decltype(spline), double>;
69 
70  static_assert(Dof<G> > 0, "reparameterize_spline only supports static-sized types");
71 
72  assert(vel_min.size() == Dof<G>);
73  assert(vel_max.size() == Dof<G>);
74  assert(acc_min.size() == Dof<G>);
75  assert(acc_max.size() == Dof<G>);
76 
77  assert(vel_min.maxCoeff() < 0);
78  assert(vel_max.minCoeff() > 0);
79  assert(acc_min.maxCoeff() < 0);
80  assert(acc_max.minCoeff() > 0);
81 
82  constexpr auto sq = [](auto x) { return x * x; };
83  constexpr auto eps = 1e-8;
84  constexpr auto inf = std::numeric_limits<double>::infinity();
85 
86  const double s0 = spline.t_min();
87  const double sf = spline.t_max();
88 
89  const double ds = (sf - s0) / N;
90 
91  // REVERSE PASS WITH MINIMAL ACCELERATION
92 
93  Eigen::VectorXd v2max(N + 1);
94  v2max(N) = end_vel * end_vel;
95 
96  for (auto i = 0u; i < N + 1; ++i) {
97  const auto Nmi = N - i; // iterating backwards
98 
99  Tangent<G> vel, acc;
100  spline(s0 + ds * Nmi, vel, acc);
101 
102  if (Nmi + 1 < N + 1) {
103  // APPROACH 1: Solve linear program in y = v^2
104  //
105  // max y
106  // s.t. y - 2 ds a \leq y(Nmi + 1) [1]
107  // vel_min.^2 \leq vel.^2 y \leq vel_max.^2 [2]
108  // acc_min \leq acc * y + vel * a \leq acc_max [3]
109  //
110  // and use acceleration a at Nmi
111  //
112  // TODO: implement, use lp2d, and evaluate
113 
114  // APPROACH 2: calculate acceleration constraints from Nmi + 1 and apply them at Nmi
115 
116  // figure minimal allowed acceleration at Nmi + 1
117  double a = -inf;
118  if (vel.cwiseAbs().maxCoeff() > eps) {
119  const Tangent<G> upper = (acc_max - acc * v2max(Nmi + 1)).cwiseMax(Tangent<G>::Zero());
120  const Tangent<G> lower = (acc_min - acc * v2max(Nmi + 1)).cwiseMin(Tangent<G>::Zero());
121  for (auto j = 0u; j != Dof<G>; ++j) {
122  if (vel(j) > eps) {
123  a = std::max<double>(a, lower(j) / vel(j));
124  } else if (vel(j) < -eps) {
125  a = std::max<double>(a, upper(j) / vel(j));
126  }
127  }
128  }
129 
130  // maximal allowed velocity at Nmi
131  v2max(Nmi) = v2max(Nmi + 1) - 2 * ds * a;
132  }
133 
134  // clamp velocity to not exceed constraints
135  for (auto j = 0u; j != Dof<G>; ++j) {
136  if (vel(j) > eps) {
137  v2max(Nmi) = std::min<double>(v2max(Nmi), sq(vel_max(j) / vel(j)));
138  } else if (vel(j) < -eps) {
139  v2max(Nmi) = std::min<double>(v2max(Nmi), sq(vel_min(j) / vel(j)));
140  }
141 
142  if (acc(j) > eps) {
143  v2max(Nmi) = std::min<double>(v2max(Nmi), acc_max(j) / acc(j));
144  } else if (acc(j) < -eps) {
145  v2max(Nmi) = std::min<double>(v2max(Nmi), acc_min(j) / acc(j));
146  }
147  }
148  }
149 
150  // FORWARD PASS WITH MAXIMAL ACCELERATION
151 
152  Spline<2, double> ret;
153  ret.reserve(N + 1);
154 
155  // "current" squared velocity
156  double v2m = std::min(start_vel * start_vel, v2max(0));
157 
158  for (auto i = 0u; i < N + 1; ++i) {
159  const double si = s0 + ds * i;
160 
161  Tangent<G> vel, acc;
162  spline(si, vel, acc);
163 
164  // velocity at si
165  const double vi2 = v2m;
166  const double vi = std::sqrt(vi2);
167 
168  // figure maximal allowed acceleration at (si, vi)
169  double ai = inf;
170  const Tangent<G> upper = (acc_max - acc * vi2).cwiseMax(Tangent<G>::Zero());
171  const Tangent<G> lower = (acc_min - acc * vi2).cwiseMin(Tangent<G>::Zero());
172  for (auto j = 0u; j != Dof<G>; ++j) {
173  if (vel(j) > eps) {
174  ai = std::min<double>(ai, upper(j) / vel(j));
175  } else if (vel(j) < -eps) {
176  ai = std::min<double>(ai, lower(j) / vel(j));
177  }
178  }
179 
180  // do not exceed velocity at next step
181  if (i + 1 < N + 1) { ai = std::min<double>(ai, (v2max(i + 1) - vi2) / (2 * ds)); }
182 
183  if (ai != inf && i < N) {
184  const double dt = std::abs(ai) < eps
185  ? ds / vi
186  : (-vi + std::sqrt(std::max<double>(eps, vi2 + 2 * ds * ai))) / ai;
187 
188  // add segment to spline
189  ret.concat_global(Spline<2, double>{
190  dt,
191  Eigen::Vector2d{dt * vi / 2, dt * (dt * ai + vi) / 2},
192  s0 + i * ds,
193  });
194 
195  // update squared velocity with value at end of new segment
196  v2m = ai == inf ? vi2 : std::max<double>(eps, vi2 + 2 * ai * ds);
197  }
198  }
199 
200  // reparameterization attains t_max
201  ret.concat_global(Spline<2, double>(spline.t_max()));
202 
203  return ret;
204 }
