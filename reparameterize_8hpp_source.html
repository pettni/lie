93  double operator()(double t, double & ds, double & d2s) const
94  {
95  const auto it =
96  utils::binary_interval_search(d_, t, [](const Data & d, double t) { return d.t <=> t; });
97 
98  if (it == d_.end()) {
99  ds = 0;
100  d2s = 0;
101  return d_.empty() ? 0. : std::clamp<double>(d_.front().s, 0., smax_);
102  }
103 
104  const double tau = t - it->t;
105 
106  ds = it->v + it->a * tau;
107  d2s = it->a;
108 
109  return std::clamp(it->s + it->v * tau + it->a * tau * tau / 2, 0., smax_);
110  }

137 template<LieGroup G>
138 Reparameterization reparameterize_spline(
139  const Spline<3, G> & curve,
140  const Tangent<G> & vel_min,
141  const Tangent<G> & vel_max,
142  const Tangent<G> & acc_min,
143  const Tangent<G> & acc_max,
144  double start_vel = 1,
145  double end_vel = std::numeric_limits<double>::infinity(),
146  bool slower_only = false,
147  double dt = 0.05,
148  double eps = 1e-2)
149 {
150  // BACKWARDS PASS WITH MINIMAL ACCELERATION
151 
152  Eigen::Vector2d state(curve.t_max(), end_vel); // s, v
153  std::vector<double> xx, yy;
154 
155  do {
156  Tangent<G> vel, acc;
157  curve(state.x(), vel, acc);
158 
159  // clamp velocity to not exceed constraints
160  for (auto i = 0u; i != Dof<G>; ++i) {
161  // regular velocity constraint
162  if (vel(i) > eps) {
163  state.y() = std::min<double>(state.y(), vel_max(i) / vel(i));
164  } else if (vel(i) < -eps) {
165  state.y() = std::min<double>(state.y(), vel_min(i) / vel(i));
166  }
167 
168  // ensure a = 0 is feasible for acceleration constraint
169  if (acc(i) > eps) {
170  state.y() = std::min<double>(state.y(), std::sqrt(acc_max(i) / acc(i)));
171  } else if (acc(i) < -eps) {
172  state.y() = std::min<double>(state.y(), std::sqrt(acc_min(i) / acc(i)));
173  }
174  }
175 
176  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
177  // ensure v stays positive
178  state.y() = std::max(state.y(), eps);
179 
180  if (xx.empty() || state.x() < xx.back()) {
181  xx.push_back(state.x());
182  yy.push_back(state.y());
183  }
184 
185  double a = 0;
186  if (vel.cwiseAbs().maxCoeff() > eps) {
187  // figure minimal allowed acceleration
188  a = -std::numeric_limits<double>::infinity();
189  const Tangent<G> upper = (acc_max - acc * state.y() * state.y()).cwiseMax(Tangent<G>::Zero());
190  const Tangent<G> lower = (acc_min - acc * state.y() * state.y()).cwiseMin(Tangent<G>::Zero());
191  for (auto i = 0u; i != Dof<G>; ++i) {
192  if (vel(i) > eps) {
193  a = std::max<double>(a, lower(i) / vel(i));
194  } else if (vel(i) < -eps) {
195  a = std::max<double>(a, upper(i) / vel(i));
196  }
197  }
198  }
199 
200  state.x() -= state.y() * dt - a * dt * dt / 2;
201  state.y() -= dt * a;
202  } while (state.x() > 0);
203 
204  if (xx.empty() || state.x() < xx.back()) {
205  xx.push_back(state.x());
206  yy.push_back(state.y());
207  }
208 
209  // fit a spline to get v(s)
210 
211  std::reverse(xx.begin(), xx.end());
212  std::reverse(yy.begin(), yy.end());
213 
214  // linear spline guarantees positive velocities
215  auto v_func = fit_spline(xx, yy, spline_specs::PiecewiseLinear<double>{});
216 
217  // FORWARD PASS WITH MAXIMAL ACCELERATION
218 
219  std::vector<Reparameterization::Data> dd;
220  state << 0, start_vel;
221 
222  // clamp velocity to not exceed upper bound
223  state.y() = std::clamp<double>(state.y(), eps, v_func(0));
224  if (slower_only) { state.y() = std::min<double>(state.y(), 1); }
225 
226  do {
227  Tangent<G> vel, acc;
228  curve(state.x(), vel, acc);
229 
230  double a = 0;
231  if (vel.cwiseAbs().maxCoeff() > eps) {
232  // figure maximal allowed acceleration
233  a = std::numeric_limits<double>::infinity();
234  const Tangent<G> upper = (acc_max - acc * state.y() * state.y()).cwiseMax(Tangent<G>::Zero());
235  const Tangent<G> lower = (acc_min - acc * state.y() * state.y()).cwiseMin(Tangent<G>::Zero());
236  for (auto i = 0u; i != Dof<G>; ++i) {
237  if (vel(i) > eps) {
238  a = std::min<double>(a, upper(i) / vel(i));
239  } else if (vel(i) < -eps) {
240  a = std::min<double>(a, lower(i) / vel(i));
241  }
242  }
243 
244  const double new_s = state.x() + state.y() * dt + a * dt * dt / 2;
245  double new_v = state.y() + dt * a;
246 
247  // clamp velocity to not exceed upper bound
248  new_v = std::clamp<double>(new_v, eps, v_func(new_s));
249  if (slower_only) { new_v = std::min<double>(new_v, 1); }
250 
251  a = (new_v - state.y()) / dt;
252  }
253 
254  dd.push_back({
255  .t = dd.empty() ? 0 : dd.back().t + dt,
256  .s = state.x(),
257  .v = state.y(),
258  .a = a,
259  });
260 
261  state.x() += dd.back().v * dt + dd.back().a * dt * dt / 2;
262  state.y() += dd.back().a * dt;
263  } while (state.x() < curve.t_max());
264 
265  dd.push_back({
266  .t = dd.empty() ? 0 : dd.back().t + dt,
267  .s = state.x(),
268  .v = state.y(),
269  .a = 0,
270  });
271 
272  return Reparameterization(curve.t_max(), std::move(dd));
273 }
