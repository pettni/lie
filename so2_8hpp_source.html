163{

176 SO2(const Scalar & qz, const Scalar & qw)
177 {
178 using std::sqrt;
179
180 const Scalar n = sqrt(qw * qw + qz * qz);
181 coeffs_.x() = qz / n;
182 coeffs_.y() = qw / n;
183 }
184
190 explicit SO2(const Scalar & angle)
191 {
192 using std::cos, std::sin;
193
194 coeffs_.x() = sin(angle);
195 coeffs_.y() = cos(angle);
196 }

205 SO2(const std::complex<Scalar> & c)
206 {
207 using std::sqrt;
208
209 const Scalar n = sqrt(c.imag() * c.imag() + c.real() * c.real());
210 coeffs_.x() = c.imag() / n;
211 coeffs_.y() = c.real() / n;
212 }
