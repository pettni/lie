152{

155
156public:
165 SO2(const Scalar & qz, const Scalar & qw)
166 {
167 using std::sqrt;
168
169 const Scalar n = sqrt(qw * qw + qz * qz);
170 coeffs_.x() = qz / n;
171 coeffs_.y() = qw / n;
172 }
173
179 explicit SO2(const Scalar & angle)
180 {
181 using std::cos, std::sin;
182
183 coeffs_.x() = sin(angle);
184 coeffs_.y() = cos(angle);

194 SO2(const std::complex<Scalar> & c)
195 {
196 using std::sqrt;
197
198 const Scalar n = sqrt(c.imag() * c.imag() + c.real() * c.real());
199 coeffs_.x() = c.imag() / n;
200 coeffs_.y() = c.real() / n;
201 }
