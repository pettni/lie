164 {
165 using std::sqrt;
166
167 const Scalar n = sqrt(qw * qw + qz * qz);
168 m_coeffs.x() = qz / n;
169 m_coeffs.y() = qw / n;
170 }

192 SO2(const std::complex<Scalar> & c)
193 {
194 using std::sqrt;
