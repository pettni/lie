138 SO2<Scalar> project_so2() const
139 {
140 using std::atan2;
141
142 const auto q = quat();
143 Scalar yaw = atan2(
144 Scalar(2) * (q.w() * q.z() + q.x() * q.y()),
145 Scalar(1) - Scalar(2) * (q.y() * q.y() + q.z() * q.z()));
146 return SO2<Scalar>(yaw);
147 }
