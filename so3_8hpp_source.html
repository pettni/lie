127 SO2<Scalar> project_so2() const
128 {
129 using std::atan2;
130
131 const auto q = quat();
132 Scalar yaw = atan2(
133 Scalar(2) * (q.w() * q.z() + q.x() * q.y()),
134 Scalar(1) - Scalar(2) * (q.y() * q.y() + q.z() * q.z()));
135 return SO2<Scalar>(yaw);
