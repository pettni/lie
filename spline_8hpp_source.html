354  template<typename S = double>
355  CastT<S, G> operator()(
356  const S & t,
357  detail::OptTangent<CastT<S, G>> vel = {},
358  detail::OptTangent<CastT<S, G>> acc = {}) const
359  {
360  if (empty() || t < 0) {
361  if (vel.has_value()) { vel.value().setZero(); }
362  if (acc.has_value()) { acc.value().setZero(); }
363  return cast<S>(g0_);
364  }
365 
366  if (t > t_max()) {
367  if (vel.has_value()) { vel.value().setZero(); }
368  if (acc.has_value()) { acc.value().setZero(); }
369  return cast<S>(end_g_.back());
370  }
371 
372  const auto istar = find_idx(static_cast<double>(t));
373 
374  const double ta = istar == 0 ? 0 : end_t_[istar - 1];
375  const double T = end_t_[istar] - ta;
376 
377  const double Del = seg_Del_[istar];
378  const S u = std::clamp<S>(S(seg_T0_[istar]) + S(Del) * (t - S(ta)) / S(T), S(0.), S(1.));
379 
380  G g0 = istar == 0 ? g0_ : end_g_[istar - 1];
381 
382  if constexpr (K == 0) {
383  // piecewise constant, nothing to evaluate
384  if (vel.has_value()) { vel.value().setZero(); }
385  if (acc.has_value()) { acc.value().setZero(); }
386  return g0;
387  } else {
388  // compensate for cropped intervals
389  if (seg_T0_[istar] > 0) {
390  g0 = composition(
391  g0, inverse(cspline_eval_diff<K, G>(Vs_[istar].colwise(), B_, seg_T0_[istar])));
392  }
393  const CastT<S, G> add = cspline_eval_diff<K, CastT<S, G>>(
394  Vs_[istar].template cast<S>().colwise(), B_.template cast<S>(), u, vel, acc);
395  if (vel.has_value()) { vel.value() *= S(Del / T); }
396  if (acc.has_value()) { acc.value() *= S(Del * Del / (T * T)); }
397  return composition(cast<S>(g0), add);
398  }
399  }

412  Tangent<G> arclength(double t) const requires(K == 3)
413  {
414  Tangent<G> ret = Tangent<G>::Zero();
415 
416  for (auto i = 0u; i < end_t_.size(); ++i) {
417  // check if we have reached t
418  if (i > 0 && t <= end_t_[i - 1]) { break; }
419 
420  // polynomial coefficients a0 + a1 x + a2 x2 + a3 x3
421  const Eigen::Matrix<double, K + 1, Dof<G>> coefs = B_.rightCols(K) * Vs_[i].transpose();
422 
423  const double ta = i == 0 ? 0 : end_t_[i - 1];
424  const double tb = end_t_[i];
425 
426  const double ua = seg_T0_[i];
427  const double ub = ua + seg_Del_[i] * (std::min<double>(t, tb) - ta) / (tb - ta);
428 
429  for (auto k = 0u; k < Dof<G>; ++k) {
430  // derivative b0 + b1 x + b2 x2 has coefficients [b0, b1, b2] = [a1, 2a2, 3a3]
431  ret(k) +=
432  integrate_absolute_polynomial(ua, ub, 3 * coefs(3, k), 2 * coefs(2, k), coefs(1, k));
433  }
434  }
435 
436  return ret;
437  }

455  Spline
456  crop(double ta, double tb = std::numeric_limits<double>::infinity(), bool localize = true) const
457  {
458  ta = std::max<double>(ta, 0);
459  tb = std::min<double>(tb, t_max());
460 
461  if (tb <= ta) { return Spline(); }
462 
463  const std::size_t i0 = find_idx(ta);
464  std::size_t Nseg = find_idx(tb) + 1 - i0;
465 
466  // prevent last segment from being empty
467  if (Nseg >= 2 && end_t_[i0 + Nseg - 2] == tb) { --Nseg; }
468 
469  // state at new from beginning of Spline
470  G ga = operator()(ta);
471 
472  std::vector<double> end_t(Nseg);
473  std::vector<G> end_g(Nseg);
474  std::vector<Eigen::Matrix<double, Dof<G>, K>> vs(Nseg);
475  std::vector<double> seg_T0(Nseg), seg_Del(Nseg);
476 
477  // copy over all relevant segments
478  for (auto i = 0u; i != Nseg; ++i) {
479  if (i == Nseg - 1) {
480  end_t[i] = tb - ta;
481  end_g[i] = composition(inverse(ga), operator()(tb));
482  } else {
483  end_t[i] = end_t_[i0 + i] - ta;
484  end_g[i] = composition(inverse(ga), end_g_[i0 + i]);
485  }
486  vs[i] = Vs_[i0 + i];
487  seg_T0[i] = seg_T0_[i0 + i];
488  seg_Del[i] = seg_Del_[i0 + i];
489  }
490 
491  // crop first segment
492  {
493  const double tta = 0;
494  const double ttb = end_t_[i0];
495  const double sa = ta;
496  const double sb = ttb;
497 
498  seg_T0[0] += seg_Del[0] * (sa - tta) / (ttb - tta);
499  seg_Del[0] *= (sb - sa) / (ttb - tta);
500  }
501 
502  // crop last segment
503  {
504  const double tta = Nseg == 1 ? ta : end_t_[Nseg - 2];
505  const double ttb = end_t_[Nseg - 1];
506  const double sa = tta;
507  const double sb = tb;
508 
509  seg_T0[Nseg - 1] += seg_Del[Nseg - 1] * (sa - tta) / (ttb - tta);
510  seg_Del[Nseg - 1] *= (sb - sa) / (ttb - tta);
511  }
512 
513  // create new Spline with appropriate body velocities
514  Spline<K, G> ret;
515  ret.g0_ = localize ? Identity<G>() : std::move(ga);
516  ret.end_t_ = std::move(end_t);
517  ret.end_g_ = std::move(end_g);
518  ret.Vs_ = std::move(vs);
519  ret.seg_T0_ = std::move(seg_T0);
520  ret.seg_Del_ = std::move(seg_Del);
521  return ret;
522  }

525  std::size_t find_idx(double t) const
526  {
527  // target condition:
528  // end_t_[istar - 1] <= t < end_t_[istar]
529 
530  std::size_t istar = 0;
531 
532  auto it = utils::binary_interval_search(end_t_, t);
533  if (it != end_t_.end()) {
534  istar = std::min<std::size_t>((it - end_t_.begin()) + 1, end_t_.size() - 1);
535  }

543  Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>>(B_s_[0].data());

545  // segment i is defined by
546  //
547  // - time interval: end_t_[i-1], end_t_[i]
