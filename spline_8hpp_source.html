168 static Spline ConstantVelocity(const Tangent<G> & v, double T = 1, const G & ga = Identity<G>())

186 static Spline FixedCubic(
187 const G & gb,
188 const Tangent<G> & va,
189 const Tangent<G> & vb,
190 double T = 1,
191 const G & ga = Identity<G>()) requires(K == 3)
192 {
193 Eigen::Matrix<double, Dof<G>, K> V;
194 V.col(0) = T * va / 3;
195 V.col(2) = T * vb / 3;
196 V.col(1) = log(composition(
197 ::smooth::exp<G>(-V.col(0)), composition(inverse(ga), gb), ::smooth::exp<G>(-V.col(2))));
198 return Spline(T, std::move(V), ga);

253 Spline & concat_global(const Spline & other)
254 {
255 const std::size_t N1 = size();
256 const std::size_t N2 = other.size();
257
258 const double tend = t_max();
259
260 if (empty()) {
261 g0_ = other.g0_;
262 } else {
263 end_g_[N1 - 1] = other.g0_;
264 }
265
266 end_t_.resize(N1 + N2);
267 end_g_.resize(N1 + N2);
268 Vs_.resize(N1 + N2);
269 seg_T0_.resize(N1 + N2);
270 seg_Del_.resize(N1 + N2);
271
272 for (auto i = 0u; i < N2; ++i) {
273 end_t_[N1 + i] = tend + other.end_t_[i];
274 end_g_[N1 + i] = other.end_g_[i];
275 Vs_[N1 + i] = other.Vs_[i];
276 seg_T0_[N1 + i] = other.seg_T0_[i];

364 template<typename S = double>
365 CastT<S, G> operator()(
366 const S & t,
367 detail::OptTangent<CastT<S, G>> vel = {},
368 detail::OptTangent<CastT<S, G>> acc = {}) const
369 {
370 if (empty() || t < S(0)) {
371 if (vel.has_value()) { vel.value().setZero(); }
372 if (acc.has_value()) { acc.value().setZero(); }
373 return cast<S>(g0_);
374 } else if (t > S(t_max())) {
375 if (vel.has_value()) { vel.value().setZero(); }
376 if (acc.has_value()) { acc.value().setZero(); }
377 return cast<S>(end_g_.back());
378 }
379
380 const auto istar = find_idx(static_cast<double>(t));
381
382 const double ta = istar == 0 ? 0 : end_t_[istar - 1];
383 const double T = end_t_[istar] - ta;
384
385 const double Del = seg_Del_[istar];
386 const S u = std::clamp<S>(S(seg_T0_[istar]) + S(Del) * (t - S(ta)) / S(T), S(0.), S(1.));
387
388 G g0 = istar == 0 ? g0_ : end_g_[istar - 1];
389
390 if constexpr (K == 0) {
391 // piecewise constant, nothing to evaluate
392 if (vel.has_value()) { vel.value().setZero(); }
393 if (acc.has_value()) { acc.value().setZero(); }
394 return g0;
395 } else {
396 // compensate for cropped intervals
397 if (seg_T0_[istar] > 0) {
398 g0 = composition(
399 g0, inverse(cspline_eval_diff<K, G>(Vs_[istar].colwise(), B_, seg_T0_[istar])));
400 }
401 const CastT<S, G> add = cspline_eval_diff<K, CastT<S, G>>(
402 Vs_[istar].template cast<S>().colwise(), B_.template cast<S>(), u, vel, acc);
403 if (vel.has_value()) { vel.value() *= S(Del / T); }
404 if (acc.has_value()) { acc.value() *= S(Del * Del / (T * T)); }
405 return composition(cast<S>(g0), add);
406 }
407 }

420 [[nodiscard]] Tangent<G> arclength(double t) const requires(K == 3)
421 {
422 Tangent<G> ret = Tangent<G>::Zero();
423
424 for (auto i = 0u; i < end_t_.size(); ++i) {
425 // check if we have reached t
426 if (i > 0 && t <= end_t_[i - 1]) { break; }
427
428 // polynomial coefficients a0 + a1 x + a2 x2 + a3 x3
429 const Eigen::Matrix<double, K + 1, Dof<G>> coefs = B_.rightCols(K) * Vs_[i].transpose();
430
431 const double ta = i == 0 ? 0 : end_t_[i - 1];
432 const double tb = end_t_[i];
433
434 const double ua = seg_T0_[i];
435 const double ub = ua + seg_Del_[i] * (std::min<double>(t, tb) - ta) / (tb - ta);
436
437 for (auto k = 0u; k < Dof<G>; ++k) {
438 // derivative b0 + b1 x + b2 x2 has coefficients [b0, b1, b2] = [a1, 2a2, 3a3]
439 ret(k) +=
440 integrate_absolute_polynomial(ua, ub, 3 * coefs(3, k), 2 * coefs(2, k), coefs(1, k));
441 }
442 }
443
444 return ret;
445 }

463 [[nodiscard]] Spline
464 crop(double ta, double tb = std::numeric_limits<double>::infinity(), bool localize = true) const
465 {
466 ta = std::max<double>(ta, 0);
467 tb = std::min<double>(tb, t_max());
468
469 if (tb <= ta) { return Spline(); }
470
471 const std::size_t i0 = find_idx(ta);
472 std::size_t Nseg = find_idx(tb) + 1 - i0;
473
474 // prevent last segment from being empty
475 if (Nseg >= 2 && end_t_[i0 + Nseg - 2] == tb) { --Nseg; }
476
477 if (Nseg == 0) { return Spline(); } // appease santizer
478
479 // state at new from beginning of Spline
480 G ga = operator()(ta);
481
482 std::vector<double> end_t(Nseg);
483 std::vector<G> end_g(Nseg);
484 std::vector<Eigen::Matrix<double, Dof<G>, K>> vs(Nseg);
485 std::vector<double> seg_T0(Nseg), seg_Del(Nseg);
486
487 // copy over all relevant segments
488 for (auto i = 0u; i < Nseg; ++i) {
489 if (i == Nseg - 1) {
490 end_t[i] = tb - ta;
491 end_g[i] = composition(inverse(ga), operator()(tb));
492 } else {
493 end_t[i] = end_t_[i0 + i] - ta;
494 end_g[i] = composition(inverse(ga), end_g_[i0 + i]);
495 }
496 vs[i] = Vs_[i0 + i];
497 seg_T0[i] = seg_T0_[i0 + i];
498 seg_Del[i] = seg_Del_[i0 + i];
499 }
500
501 // crop first segment
502 {
503 const double tta = 0;
504 const double ttb = end_t_[i0];
505 const double sa = ta;
506 const double sb = ttb;
507
508 seg_T0[0] += seg_Del[0] * (sa - tta) / (ttb - tta);
509 seg_Del[0] *= (sb - sa) / (ttb - tta);
510 }
511
512 // crop last segment
513 {
514 const double tta = Nseg == 1 ? ta : end_t_[Nseg - 2];
515 const double ttb = end_t_[Nseg - 1];
516 const double sa = tta;
517 const double sb = tb;
518
519 seg_T0[Nseg - 1] += seg_Del[Nseg - 1] * (sa - tta) / (ttb - tta);
520 seg_Del[Nseg - 1] *= (sb - sa) / (ttb - tta);
521 }
522
523 // create new Spline with appropriate body velocities
524 Spline<K, G> ret;
525 ret.g0_ = localize ? Identity<G>() : std::move(ga);
526 ret.end_t_ = std::move(end_t);
527 ret.end_g_ = std::move(end_g);
528 ret.Vs_ = std::move(vs);
529 ret.seg_T0_ = std::move(seg_T0);
530 ret.seg_Del_ = std::move(seg_Del);
531 return ret;
532 }
533
534private:
535 std::size_t find_idx(double t) const
536 {
537 // target condition:
538 // end_t_[istar - 1] <= t < end_t_[istar]
539
540 std::size_t istar = 0;
541
542 auto it = utils::binary_interval_search(end_t_, t);
543 if (it != end_t_.end()) {
544 istar = std::min<std::size_t>((it - end_t_.begin()) + 1, end_t_.size() - 1);
545 }
546
547 return istar;
548 }

550 // cumulative basis functions
551 static constexpr auto B_s_ = polynomial_cumulative_basis<PolynomialBasis::Bernstein, K, double>();
552 inline static const Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> B_ =
553 Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>>(B_s_[0].data());
554
555 // segment i is defined by
556 //
557 // - time interval: end_t_[i-1], end_t_[i]
558 // - g interval: end_g_[i-1], end_g_[i]
559 // - velocities: Vs_[i]
560 // - crop: seg_T0_[i], seg_Del_[i]
561 //
562 // s.t.
563 //
564 // u(t) = T0[i] + Del[i] * (t - end_t[i-1]) / (end_t[i] - end_t[i-1])
565 //
566 // x(t) = xu(u(t)) where xu is spline defined in [0, 1]
567
568 // Spline starting point
569 G g0_;
570
571 // segment end times
572 std::vector<double> end_t_;
573
574 // segment end points
575 std::vector<G> end_g_;
576
577 // segment bezier velocities
578 std::vector<Eigen::Matrix<double, Dof<G>, K>> Vs_;
579
580 // segment crop information
581 std::vector<double> seg_T0_, seg_Del_;
582};
