162 const G & gb,

165 double T = 1,
166 const G & ga = Identity<G>()) requires(K == 3)

339 template<typename S = double>
340 CastT<S, G>
341 operator()(const S & t, OptTangent<CastT<S, G>> vel = {}, OptTangent<CastT<S, G>> acc = {}) const
342 {
343 if (empty() || t < S(0)) {
344 if (vel.has_value()) { vel.value().setZero(); }
345 if (acc.has_value()) { acc.value().setZero(); }
346 return cast<S>(g0_);
347 } else if (t > S(t_max())) {
348 if (vel.has_value()) { vel.value().setZero(); }
349 if (acc.has_value()) { acc.value().setZero(); }
350 return cast<S>(end_g_.back());
351 }
352
353 const auto istar = find_idx(static_cast<double>(t));
354
355 const double ta = istar == 0 ? 0 : end_t_[istar - 1];
356 const double T = end_t_[istar] - ta;
357
358 const double Del = seg_Del_[istar];
359 const S u = std::clamp<S>(S(seg_T0_[istar]) + S(Del) * (t - S(ta)) / S(T), S(0.), S(1.));
360
361 G g0 = istar == 0 ? g0_ : end_g_[istar - 1];
362
363 if constexpr (K == 0) {
364 // piecewise constant, nothing to evaluate
365 if (vel.has_value()) { vel.value().setZero(); }
366 if (acc.has_value()) { acc.value().setZero(); }
367 return g0;
368 } else {
369 // compensate for cropped intervals
370 if (seg_T0_[istar] > 0) {
371 g0 =
372 composition(g0, inverse(cspline_eval_vs<K, G>(Vs_[istar].colwise(), B_, seg_T0_[istar])));
373 }
374 const CastT<S, G> add = cspline_eval_vs<K, CastT<S, G>>(
375 Vs_[istar].template cast<S>().colwise(), B_.template cast<S>(), u, vel, acc);
376 if (vel.has_value()) { vel.value() *= S(Del / T); }
377 if (acc.has_value()) { acc.value() *= S(Del * Del / (T * T)); }
378 return composition(cast<S>(g0), add);
379 }
380 }

393 [[nodiscard]] Tangent<G> arclength(double t) const requires(K == 3)
394 {
395 Tangent<G> ret = Tangent<G>::Zero();
396
397 for (auto i = 0u; i < end_t_.size(); ++i) {
398 // check if we have reached t
399 if (i > 0 && t <= end_t_[i - 1]) { break; }
400
401 // polynomial coefficients a0 + a1 x + a2 x2 + a3 x3
402 const Eigen::Matrix<double, K + 1, Dof<G>> coefs = B_.rightCols(K) * Vs_[i].transpose();
403
404 const double ta = i == 0 ? 0 : end_t_[i - 1];
405 const double tb = end_t_[i];
406
407 const double ua = seg_T0_[i];
408 const double ub = ua + seg_Del_[i] * (std::min<double>(t, tb) - ta) / (tb - ta);
409
410 for (auto k = 0u; k < Dof<G>; ++k) {
411 // derivative b0 + b1 x + b2 x2 has coefficients [b0, b1, b2] = [a1, 2a2, 3a3]
412 ret(k) +=
413 integrate_absolute_polynomial(ua, ub, 3 * coefs(3, k), 2 * coefs(2, k), coefs(1, k));
414 }
415 }
416
417 return ret;
418 }

436 [[nodiscard]] Spline
437 crop(double ta, double tb = std::numeric_limits<double>::infinity(), bool localize = true) const
438 {
439 ta = std::max<double>(ta, 0);
440 tb = std::min<double>(tb, t_max());
441
442 if (tb <= ta) { return Spline(); }
443
444 const std::size_t i0 = find_idx(ta);
445 std::size_t Nseg = find_idx(tb) + 1 - i0;
446
447 // prevent last segment from being empty
448 if (Nseg >= 2 && end_t_[i0 + Nseg - 2] == tb) { --Nseg; }
449
450 if (Nseg == 0) { return Spline(); } // appease santizer
451
452 // state at new from beginning of Spline
453 G ga = operator()(ta);
454
455 std::vector<double> end_t(Nseg);
456 std::vector<G> end_g(Nseg);
457 std::vector<Eigen::Matrix<double, Dof<G>, K>> vs(Nseg);
458 std::vector<double> seg_T0(Nseg), seg_Del(Nseg);
459
460 // copy over all relevant segments
461 for (auto i = 0u; i < Nseg; ++i) {
462 if (i == Nseg - 1) {
463 end_t[i] = tb - ta;
464 end_g[i] = composition(inverse(ga), operator()(tb));
465 } else {
466 end_t[i] = end_t_[i0 + i] - ta;
467 end_g[i] = composition(inverse(ga), end_g_[i0 + i]);
468 }
469 vs[i] = Vs_[i0 + i];
470 seg_T0[i] = seg_T0_[i0 + i];
471 seg_Del[i] = seg_Del_[i0 + i];
472 }
473
474 // crop first segment
475 {
476 const double tta = 0;
477 const double ttb = end_t_[i0];
478 const double sa = ta;
479 const double sb = ttb;
480
481 seg_T0[0] += seg_Del[0] * (sa - tta) / (ttb - tta);
482 seg_Del[0] *= (sb - sa) / (ttb - tta);
483 }
484
485 // crop last segment
486 {
487 const double tta = Nseg == 1 ? ta : end_t_[Nseg - 2];
488 const double ttb = end_t_[Nseg - 1];
489 const double sa = tta;
490 const double sb = tb;
491
492 seg_T0[Nseg - 1] += seg_Del[Nseg - 1] * (sa - tta) / (ttb - tta);
493 seg_Del[Nseg - 1] *= (sb - sa) / (ttb - tta);
494 }
495
496 // create new Spline with appropriate body velocities
497 Spline<K, G> ret;
498 ret.g0_ = localize ? Identity<G>() : std::move(ga);
499 ret.end_t_ = std::move(end_t);
500 ret.end_g_ = std::move(end_g);
501 ret.Vs_ = std::move(vs);
502 ret.seg_T0_ = std::move(seg_T0);
503 ret.seg_Del_ = std::move(seg_Del);
504 return ret;
505 }

507private:
508 std::size_t find_idx(double t) const
509 {
510 // target condition:
511 // end_t_[istar - 1] <= t < end_t_[istar]
512
513 std::size_t istar = 0;
514
515 auto it = utils::binary_interval_search(end_t_, t);
516 if (it != end_t_.end()) {
517 istar = std::min(
518 static_cast<std::size_t>(std::distance(end_t_.begin(), it)) + 1, end_t_.size() - 1);
519 }
520
521 return istar;
522 }

524 // cumulative basis functions
525 static constexpr auto B_s_ = polynomial_cumulative_basis<PolynomialBasis::Bernstein, K, double>();
526 inline static const Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>> B_ =
527 Eigen::Map<const Eigen::Matrix<double, K + 1, K + 1, Eigen::RowMajor>>(B_s_[0].data());
528
529 // segment i is defined by
530 //
531 // - time interval: end_t_[i-1], end_t_[i]
532 // - g interval: end_g_[i-1], end_g_[i]
533 // - velocities: Vs_[i]
534 // - crop: seg_T0_[i], seg_Del_[i]
535 //
536 // s.t.
537 //
538 // u(t) = T0[i] + Del[i] * (t - end_t[i-1]) / (end_t[i] - end_t[i-1])
539 //
540 // x(t) = xu(u(t)) where xu is spline defined in [0, 1]
541
542 // Spline starting point
543 G g0_;
544
545 // segment end times
546 std::vector<double> end_t_;
547
548 // segment end points
549 std::vector<G> end_g_;
550
551 // segment bezier velocities
552 std::vector<Eigen::Matrix<double, Dof<G>, K>> Vs_;
553
554 // segment crop information
555 std::vector<double> seg_T0_, seg_Del_;
556};
