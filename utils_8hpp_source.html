61 template<std::ranges::range _R, typename _T, typename _WO>
62 auto binary_interval_search(const _R & r, const _T & t, _WO && wo) noexcept
63 {
64  auto left = std::ranges::cbegin(r);
65  auto rght = std::ranges::cend(r);
66 
67  if (std::ranges::empty(r) || wo(*left, t) > 0) {
68  return rght;
69  } else if (wo(*(rght - 1), t) <= 0) {
70  return rght - 1;
71  }
72 
73  decltype(left) pivot;
74 
75  while (left + 1 < rght) {
76  double alpha;
77  if constexpr (
78  std::is_convertible_v<
79  std::ranges::range_value_t<_R>,
80  double> && std::is_convertible_v<_T, double>) {
81  alpha = (static_cast<double>(t) - static_cast<double>(*left))
82  / static_cast<double>(*(rght - 1) - *left);
83  } else {
84  alpha = 0.5;
85  }
86 
87  pivot = left + static_cast<intptr_t>(static_cast<double>(rght - 1 - left) * alpha);
88 
89  if (wo(*(pivot + 1), t) <= 0) {
90  left = pivot + 1;
91  } else if (wo(*pivot, t) > 0) {
92  rght = pivot + 1;
93  } else {
94  break;
95  }
96  }
97 
98  return pivot;
99 }

104 template<std::ranges::range _R, typename _T, typename _S = std::ranges::range_value_t<_R>>
105 auto binary_interval_search(const _R & r, const _T & t) noexcept
106 {
107  return binary_interval_search(r, t, [](const _S & _s, const _T & _t) { return _s <=> _t; });
108 }

117 template<std::size_t _I, typename _F>
118 inline static constexpr auto static_for(_F && f)
119 {
120  const auto f_caller = [&]<std::size_t... _Idx>(std::index_sequence<_Idx...>)
121  {
122  return (std::invoke(f, std::integral_constant<std::size_t, _Idx>()), ...);
123  };
