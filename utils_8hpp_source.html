57constexpr auto
58binary_interval_search(std::ranges::random_access_range auto && r, auto && t, auto && wo) noexcept
59{
60 using T = std::decay_t<decltype(t)>;
61 using Rv = std::ranges::range_value_t<std::decay_t<decltype(r)>>;
62
63 auto left = std::ranges::cbegin(r);
64 auto rght = std::ranges::cend(r);
65
66 if (std::ranges::empty(r) || wo(*left, t) > 0) {
67 return rght;
68 } else if (wo(*(rght - 1), t) <= 0) {
69 return rght - 1;
70 }
71
72 auto pivot = left;
73
74 while (left + 1 < rght) {
75 double alpha = 0.5;
76 if constexpr (std::is_convertible_v<Rv, double> && std::is_convertible_v<T, double>) {
77 alpha = (static_cast<double>(t) - static_cast<double>(*left))
78 / static_cast<double>(*(rght - 1) - *left);
79 }
80
81 pivot = std::ranges::next(left, alpha * std::distance(left, rght - 1), rght - 2);
82
83 if (wo(*next(pivot), t) <= 0) {
84 left = pivot + 1;
85 } else if (wo(*pivot, t) > 0) {
86 rght = pivot + 1;
87 } else {
88 break;
89 }
90 }
91
92 return pivot;
93}

98constexpr auto
99binary_interval_search(std::ranges::random_access_range auto && r, auto && t) noexcept
100{
101 return binary_interval_search(
102 std::forward<decltype(r)>(r),
103 std::forward<decltype(t)>(t),
104 [](const auto & _s, const auto & _t) { return _s <=> _t; });
105}

114template<std::size_t _I>
115constexpr auto
116static_for(auto && f) noexcept(noexcept(std::invoke(f, std::integral_constant<std::size_t, 0>())))
117{
118 const auto f_caller = [&]<std::size_t... _Idx>(std::index_sequence<_Idx...>)
119 {
120 return (std::invoke(f, std::integral_constant<std::size_t, _Idx>()), ...);
121 };
122
123 return f_caller(std::make_index_sequence<_I>{});
124}

133template<typename _T, std::size_t _L>
134constexpr std::array<_T, _L + 1> array_psum(const std::array<_T, _L> & x) noexcept
135{
136 std::array<_T, _L + 1> ret;
137 ret[0] = _T(0);
138 std::partial_sum(x.begin(), x.end(), std::next(ret.begin(), 1));
139 return ret;
140}

151template<std::ranges::input_range R, std::copy_constructible F>
152 requires std::ranges::view<R>
153class pairwise_transform_view : public std::ranges::view_interface<pairwise_transform_view<R, F>>
154{
155public:
156 class _Iterator
157 {
158 private:
159 const pairwise_transform_view * parent_;
160 std::ranges::iterator_t<const R> it1_, it2_;
161
162 public:
163 using value_type = std::remove_cvref_t<
164 std::
165 invoke_result_t<F &, std::ranges::range_reference_t<R>, std::ranges::range_reference_t<R>>>;
166 using difference_type = std::ranges::range_difference_t<R>;
167
168 _Iterator() = default;
169
170 constexpr _Iterator(const pairwise_transform_view * parent, const R & r)
171 : parent_(parent), it1_(std::ranges::begin(r)), it2_(std::ranges::begin(r))
172 {
173 if (it2_ != std::ranges::end(r)) { ++it2_; }
174 }
175
176 constexpr decltype(auto) operator*() const { return std::invoke(parent_->f_, *it1_, *it2_); }
177
178 constexpr _Iterator & operator++()
179 {
180 ++it1_, ++it2_;
181 return *this;
182 }
183
184 constexpr void operator++(int) { ++it1_, ++it2_; }
185
186 constexpr _Iterator operator++(int) requires std::ranges::forward_range<R>
187 {
188 _Iterator tmp = *this;
189 ++this;
190 return tmp;
191 }
192
193 constexpr _Iterator & operator--() requires std::ranges::bidirectional_range<R>
194 {
195 --it1_, --it2_;
196 return *this;
197 }
198
199 constexpr _Iterator operator--(int) requires std::ranges::bidirectional_range<R>
200 {
201 auto tmp = *this;
202 --this;
203 return tmp;
204 }
205
206 friend constexpr bool operator==(const _Iterator & x, const _Iterator & y)
207 {
208 return x.it1_ == y.it1_;
209 }
210
211 friend constexpr bool
212 operator==(const _Iterator & x, const std::ranges::sentinel_t<const R> & y)
213 {
214 return x.it2_ == y;
215 }
216 };
217
218private:
219 R base_{};
220 F f_{};
221
222public:
223 constexpr pairwise_transform_view() = default;
224
225 template<typename Fp>
226 constexpr pairwise_transform_view(R base, Fp && f) : base_(base), f_(std::forward<Fp>(f))
227 {}
228
229 constexpr _Iterator begin() const { return _Iterator(this, base_); }
230
231 constexpr std::ranges::sentinel_t<const R> end() const { return std::ranges::end(base_); }
232
233 constexpr auto size() const requires std::ranges::sized_range<const R>
234 {
235 const auto s = std::ranges::size(base_);
236 return (s >= 2) ? s - 1 : 0;
237 }
238};

242pairwise_transform_view(R &&, F) -> pairwise_transform_view<std::views::all_t<R>, F>;

246template<typename F>
247struct PairwiseTransformClosure
248{
249 F f_;
250
251 constexpr PairwiseTransformClosure(F && f) : f_(std::forward<F>(f)) {}
252
253 template<std::ranges::viewable_range R>
254 constexpr auto operator()(R && r) const
255 {
256 return pairwise_transform_view(std::forward<R>(r), f_);
257 }
258};

260struct PairwiseTransform
261{
262 template<std::ranges::viewable_range R, typename F>
263 constexpr auto operator()(R && r, F && f) const
264 {
265 return pairwise_transform_view(std::forward<R>(r), std::forward<F>(f));
266 }
267
268 template<typename F>
269 constexpr auto operator()(F && f) const
270 {
271 return PairwiseTransformClosure<F>(std::forward<F>(f));
272 }
273};

276constexpr auto operator|(R && r, const PairwiseTransformClosure<F> & closure)

297template<std::ranges::input_range... View>
298 requires(std::ranges::view<View> &&...)
299class zip_view : public std::ranges::view_interface<zip_view<View...>>
300{
301public:
302 template<bool Const>
303 class _Iterator
304 {
305 private:
306 std::tuple<std::ranges::iterator_t<std::conditional_t<Const, const View, View>>...> its_;
307
308 public:
309 using value_type = std::tuple<std::ranges::range_value_t<View>...>;
310 using difference_type = std::common_type_t<std::ranges::range_difference_t<View>...>;
311
312 _Iterator() = default;
313
314 constexpr _Iterator(auto &&... its) : its_(its...) {}
315
316 constexpr _Iterator(_Iterator<!Const> i) requires Const : its_(i.its_) {}
317
318 constexpr decltype(auto) operator*() const
319 {
320 return [this]<std::size_t... Idx>(std::index_sequence<Idx...>)
321 {
322 return std::tuple<std::ranges::range_reference_t<View>...>(
323 std::get<Idx>(its_).operator*()...);
324 }
325 (std::make_index_sequence<sizeof...(View)>{});
326 }
327
328 constexpr _Iterator & operator++()
329 {
330 [this]<std::size_t... Idx>(std::index_sequence<Idx...>) { (++std::get<Idx>(its_), ...); }
331 (std::make_index_sequence<sizeof...(View)>{});
332 return *this;
333 }
334
335 constexpr void operator++(int)
336 {
337 [this]<std::size_t... Idx>(std::index_sequence<Idx...>) { (++std::get<Idx>(its_), ...); }
338 (std::make_index_sequence<sizeof...(View)>{});
339 }
340
341 constexpr _Iterator operator++(int) requires(std::ranges::forward_range<View> &&...)
342 {
343 _Iterator tmp = *this;
344 ++this;
345 return tmp;
346 }
347
348 friend constexpr bool operator==(const _Iterator & x, const _Iterator & y)
349 {
350 return x.its_ == y.its_;
351 }
352
353 friend constexpr bool operator==(
354 const _Iterator & x,
355 const std::tuple<std::ranges::sentinel_t<std::conditional_t<Const, const View, View>>...> & y)
356 {
357 return [&x, &y ]<std::size_t... Idx>(std::index_sequence<Idx...>)
358 {
359 return ((std::get<Idx>(x.its_) == std::get<Idx>(y)) || ...);
360 }
361 (std::make_index_sequence<sizeof...(View)>{});
362 }
363 }; // _Iterator
364
365private:
366 std::tuple<View...> bases_{};
367
368public:
369 constexpr zip_view() = default;
370
371 constexpr zip_view(View... base) : bases_(base...) {}
372
373 constexpr _Iterator<false> begin()
374 {
375 return [this]<std::size_t... Idx>(std::index_sequence<Idx...>)
376 {
377 return _Iterator<false>(std::ranges::begin(std::get<Idx>(bases_))...);
378 }
379 (std::make_index_sequence<sizeof...(View)>{});
380 }
381
382 // some views only support mutable iteration (e.g. drop view over non-random access view). zip can
383 // only allow const iteration if all underlying views support it
384 constexpr _Iterator<true> begin() const requires(std::ranges::range<const View> &&...)
385 {
386 return [this]<std::size_t... Idx>(std::index_sequence<Idx...>)
387 {
388 return _Iterator<true>(std::ranges::begin(std::get<Idx>(bases_))...);
389 }
390 (std::make_index_sequence<sizeof...(View)>{});
391 }
392
393 constexpr decltype(auto) end()
394 {
395 return [this]<std::size_t... Idx>(std::index_sequence<Idx...>)
396 {
397 return std::make_tuple<std::ranges::sentinel_t<View>...>(
398 std::ranges::end(std::get<Idx>(bases_))...);
399 }
400 (std::make_index_sequence<sizeof...(View)>{});
401 }
402
403 // see comment for const begin
404 constexpr decltype(auto) end() const requires(std::ranges::range<const View> &&...)
405 {
406 return [this]<std::size_t... Idx>(std::index_sequence<Idx...>)
407 {
408 return std::make_tuple<std::ranges::sentinel_t<const View>...>(
409 std::ranges::end(std::get<Idx>(bases_))...);
410 }
411 (std::make_index_sequence<sizeof...(View)>{});
412 }
413
414 constexpr auto size() const requires(std::ranges::sized_range<View> &&...)
415 {
416 return [this]<std::size_t... Idx>(std::index_sequence<Idx...>)
417 {
418 return std::min({std::ranges::size(std::get<Idx>(bases_))...});
419 }
420 (std::make_index_sequence<sizeof...(View)>{});
421 }
422};
