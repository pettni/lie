53template<typename Wrt>
54constexpr int wrt_Dof()
55{
56 using DWrt = std::decay_t<Wrt>;
57
58 const auto f = [&]<std::size_t... Idx>(std::index_sequence<Idx...>)->int
59 {
60 constexpr auto min_dof = std::min<int>({Dof<std::decay_t<std::tuple_element_t<Idx, DWrt>>>...});
61
62 if constexpr (min_dof == -1) {
63 return -1;
64 } else {
65 return (Dof<std::decay_t<std::tuple_element_t<Idx, DWrt>>> + ...);
66 }
67 };
68
69 return f(std::make_index_sequence<std::tuple_size_v<DWrt>>{});
70}

75template<typename Scalar>
76auto wrt_cast(auto && wrt)
77{
78 using Wrt = std::decay_t<decltype(wrt)>;
79
80 const auto f = [&]<std::size_t... Idx>(std::index_sequence<Idx...>)
81 {
82 return std::make_tuple(cast<Scalar>(std::get<Idx>(wrt))...);
83 };

85 return f(std::make_index_sequence<std::tuple_size_v<Wrt>>{});

91template<typename Derived>
92auto wrt_rplus(auto && wrt, const Eigen::MatrixBase<Derived> & a)
93{
94 using Wrt = std::decay_t<decltype(wrt)>;
95
96 const auto f = [&]<std::size_t... Idx>(std::index_sequence<Idx...>)
97 {
98 static constexpr std::array<Eigen::Index, sizeof...(Idx)> Nx{
99 Dof<std::decay_t<std::tuple_element_t<Idx, Wrt>>>...};
100
101 const std::array<Eigen::Index, sizeof...(Idx)> ilen{dof(std::get<Idx>(wrt))...};
102 const auto ibeg = utils::array_psum(ilen);
103
104 // clang-format off
105 return std::make_tuple(
106 rplus(
107 std::get<Idx>(wrt),
108 a.template segment<std::get<Idx>(Nx)>(std::get<Idx>(ibeg), std::get<Idx>(ilen))
109 )...
110 );
111 // clang-format on
112 };
113
114 return f(std::make_index_sequence<std::tuple_size_v<Wrt>>{});
115}
